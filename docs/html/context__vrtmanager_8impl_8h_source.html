137 template <typename VcT, typename MsgT, ActiveVrtTypedFnType<MsgT, VcT> *f>
138 messaging::PendingSend VirtualContextManager::sendSerialMsg(
139  VirtualProxyType const& toProxy, MsgT *const msg
140 ) {
141  auto base_msg = promoteMsg(msg).template to<BaseMsgType>();
142  ByteType msg_sz = sizeof(MsgT);
143 
144  if (theContext()->getWorker() == worker_id_comm_thread) {
145  NodeType const& home_node = VirtualProxyBuilder::getVirtualNode(toProxy);
146  // register the user's handler
147  HandlerType const& han = auto_registry::makeAutoHandlerVC<VcT,MsgT,f>();
148  // save the user's handler in the message
149  msg->setVrtHandler(han);
150  msg->setProxy(toProxy);
151 
152  vt_debug_print(
153  vrt, node,
154  "sending serialized msg to VC: msg={}, han={}, home_node={}, toProxy={}\n",
155  print_ptr(msg), han, home_node, toProxy
156  );
157 
158  using SerialMsgT = SerializedEagerMsg<MsgT, VirtualMessage>;
159 
160  // route the message to the destination using the location manager
161  messaging::PendingSend pending(
162  base_msg, msg_sz, [=](MsgPtr<BaseMsgType> mymsg){
163  // Uses special implementation overload not exposed in theMsg..
164  MsgT* typed_msg = reinterpret_cast<MsgT*>(mymsg.get());
165  auto sendSerialHan = auto_registry::makeAutoHandler<MsgT,virtualTypedMsgHandler<MsgT>>();
166  SerializedMessenger::sendSerialMsgSendImpl<MsgT, VirtualMessage>(
167  typed_msg,
168  sendSerialHan,
169  // custom send lambda to route the message
170  [=](MsgSharedPtr<SerialMsgT> innermsg) -> messaging::PendingSend {
171  innermsg->setProxy(toProxy);
172  theLocMan()->vrtContextLoc->routeMsgHandler<
173  SerialMsgT, SerializedMessenger::payloadMsgHandler
174  >(toProxy, home_node, innermsg.get());
175  return messaging::PendingSend(nullptr);
176  },
177  // custom data transfer lambda if above the eager threshold
178  [=](ActionNodeSendType action) -> messaging::PendingSend {
179  auto captured_action = [=](NodeType node){ action(node); };
180  theLocMan()->vrtContextLoc->routeNonEagerAction(
181  toProxy, home_node, captured_action
182  );
183  return messaging::PendingSend(nullptr);
184  }
185  );
186  }
187  );
188  return pending;
189  } else {
190  return messaging::PendingSend(
191  base_msg, msg_sz, [=](MsgPtr<BaseMsgType> mymsg) {
192  theWorkerGrp()->enqueueCommThread([=]{
193  auto typed_msg = reinterpret_cast<MsgT*>(mymsg.get());
194  theVirtualManager()->sendSerialMsg<VcT, MsgT, f>(toProxy, typed_msg);
195  });
196  }
197  );
198  }
199 }

201 template <typename VrtContextT, typename... Args>
202 VirtualProxyType VirtualContextManager::makeVirtualRemote(
203  NodeType const& dest, bool isImmediate, ActionProxyType action, Args&&... args
204 ) {
205  using ArgsTupleType = std::tuple<typename std::decay<Args>::type...>;
206  using MsgType = VrtConstructMsg<RemoteVrtInfo, ArgsTupleType, VrtContextT>;
207 
208  auto sys_msg = makeMessage<MsgType>(ArgsTupleType{std::forward<Args>(args)...});
209 
210  auto const& this_node = theContext()->getNode();
211  std::unique_ptr<RemoteVrtInfo> info = nullptr;
212  VirtualProxyType return_proxy = no_vrt_proxy;
213 
214  if (isImmediate) {
215  auto const& remote_id = generateNewRemoteID(dest);
216  return_proxy = VirtualProxyBuilder::createRemoteProxy(
217  remote_id, myNode_, dest, false, false
218  );
219  info = std::make_unique<RemoteVrtInfo>(this_node, return_proxy);
220  } else {
221  auto next_req = cur_request_id++;
222 
223  // insert a pending request to trigger the action
224  pending_request_.emplace(
225  std::piecewise_construct,
226  std::forward_as_tuple(next_req),
227  std::forward_as_tuple(PendingRequestType{next_req, action})
228  );
229 
230  info = std::make_unique<RemoteVrtInfo>(this_node, next_req);
231  }
232 
233  sys_msg->info = *info.get();
234 
235  theMsg()->sendMsg<MsgType,remoteConstructVrt<MsgType>>(dest, sys_msg);
236 
237  return return_proxy;
238 }

261 template <typename VrtContextT, mapping::ActiveSeedMapFnType fn, typename... Args>
262 VirtualProxyType VirtualContextManager::makeVirtualMap(Args... args) {
263  SeedType next_seed = no_seed;
264  HandlerType core_map_handle = uninitialized_handler;
265  bool const& has_workers = theContext()->hasWorkers();
266 
267  vt_debug_print(
268  vrt, node,
269  "makeVirtualMap: has_workers={}\n", print_bool(has_workers)
270  );
271 
272  if (has_workers) {
273  next_seed = cur_seed_++;
274  core_map_handle = auto_registry::makeAutoHandlerSeedMap<fn>();
275 
276  auto const& num_workers = theContext()->getNumWorkers();
277  auto const& mapped_core = fn(next_seed, num_workers);
278 
279  vt_debug_print(
280  vrt, node,
281  "seed={}, mapped_core={}, num_workers={}\n",
282  next_seed, mapped_core, num_workers
283  );
284 
285  if (mapped_core != worker_id_comm_thread) {
286  using TupleType = std::tuple<Args...>;
287 
288  auto proxy = makeVirtualPlaceholder();
289  auto vrt_info = getVirtualInfoByProxy(proxy);
290  setupMappedVirutalContext(proxy, next_seed, mapped_core, core_map_handle);
291 
292  auto cl = new VirtualMakeClosure<VrtContextT, Args...>(
293  TupleType{std::forward<Args>(args)...}, proxy, vrt_info
294  );
295 
296  // work to defer to the worker thread
297  auto work_unit = [=]{ cl->make(); delete cl; };
298  theWorkerGrp()->enqueueForWorker(mapped_core, work_unit);
299 
300  return proxy;
301  }
302  }
303 
304  return makeVirtualMapComm<VrtContextT>(
305  next_seed, core_map_handle, std::forward<Args>(args)...
306  );
307 }
