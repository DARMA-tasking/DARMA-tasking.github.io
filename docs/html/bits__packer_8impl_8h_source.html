113 template <typename BitType, typename BitField>
114 /*static*/ inline void
115 BitPacker::setFieldDynamic(
116  FieldType start, FieldType len, BitField& field, BitType const& segment
117 ) {
118  // make_unsigned UB before C++20 for non-integral or bool types (ill-formed after C++20)
119  static_assert(std::is_integral<BitField>::value && !std::is_same<BitField, bool>::value,
120  "BitField must be non-bool integral type");
121  // sizeof(UnsignedBitField) is guaranteed to be the same as sizeof(BitField)
122  using UnsignedBitField = std::make_unsigned_t<BitField>;
123  auto const nbits = (sizeof(BitField) * 8) - len;
124  field = field & ~(gen_bit_mask(len) << start);
125  auto seg = bit_cast<UnsignedBitField>(static_cast<BitField>(segment));
126  UnsignedBitField bits = ((seg << nbits) >> nbits) << start;
127  field = field | bit_cast<BitField>( bits );
128 }
