103  vt_debug_print(

106  print_bool(iter != lookup_.end()), lookup_.size()
107  );

109  if (iter == lookup_.end()) {
110  if (lookup_.size() + 1 > max_size_) {
111  auto last_iter = cache_.crbegin();
112  lookup_.erase(std::get<0>(*last_iter));
113  cache_.pop_back();
114  }

116  cache_.push_front(std::make_tuple(key, value));
117 
118  lookup_.emplace(
119  std::piecewise_construct,
120  std::forward_as_tuple(key),
121  std::forward_as_tuple(cache_.begin())
122  );
123  } else {
124  std::get<1>(*iter->second) = value;
125  cache_.splice(cache_.cbegin(), cache_, iter->second);
126  }
