60 template <typename T>
61 struct ProcessBuffer {
62  using ProcessFnType = std::function<void(T&)>;
63 
64  void push(T const& elm) {
65  LockGuardPtrType lock(getMutex());
66  buffer_.push_back(elm);
67  progressEngine(true);
68  }
69 
70  void emplace(T&& elm) {
71  LockGuardPtrType lock(getMutex());
72  buffer_.emplace_back(std::forward<T>(elm));
73  progressEngine(true);
74  }
75 
76  void attach(ProcessFnType fn, WorkerIDType worker = no_worker_id) {
77  LockGuardPtrType lock(getMutex());
78  process_fn_ = fn;
79  worker_ = worker;
80  progressEngine(true);
81  }
82 
83  inline void progress() { progressEngine(false); }
84 
85 private:
86  void apply(ProcessFnType fn, bool locked) {
87  bool found = true;
88  T elm_out;
89  do {
90  {
91  LockGuardPtrType lock(locked ? nullptr : getMutex());
92  if (buffer_.size() > 0) {
93  found = true;
94  T elm(std::move(buffer_.front()));
95  elm_out = std::move(elm);
96  buffer_.pop_front();
97  } else {
98  found = false;
99  }
100  }
101  if (found) {
102  fn(elm_out);
103  }
104  } while (found);
105  }
106 
107 private:
108  inline void progressEngine(bool has_lock) {
109  if (process_fn_ && isProcessWorker()) { apply(process_fn_, has_lock); }
110  }
111  inline bool isProcessWorker() const {
112  return worker_ == no_worker_id || worker_ == ::vt::theContext()->getWorker();
113  }
114  inline MutexType* getMutex() { return needs_lock_ ? &mutex_: nullptr; }
115 
116 private:
117  std::list<T> buffer_;
118  bool needs_lock_ = true;
119  MutexType mutex_{};
120  WorkerIDType worker_ = no_worker_id;
121  ProcessFnType process_fn_ = nullptr;
122 };
