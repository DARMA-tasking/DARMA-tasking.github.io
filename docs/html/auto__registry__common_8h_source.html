77 template <typename MsgT, typename HandlerT, typename ObjT>
78 struct HandlersDispatcher final : BaseHandlersDispatcher {
79  using ColTypedFnType = vrt::collection::ActiveColTypedFnType<MsgT, ObjT>;
80  using ColMemberTypedFnType = vrt::collection::ActiveColMemberTypedFnType<MsgT, ObjT>;
81 
82  explicit HandlersDispatcher(HandlerT in_fn_ptr) : fp(in_fn_ptr) { }
83 
84  template <typename U>
85  using ColMsgTrait = typename U::IsCollectionMessage;
86  template <typename U>
87  using IsColMsgTrait =
88  detection::is_detected_convertible<std::true_type, ColMsgTrait, U>;
89 
90  template <typename U>
91  using ColTrait = typename U::IsCollectionType;
92  template <typename U>
93  using IsColTrait =
94  detection::is_detected_convertible<std::true_type, ColTrait, U>;
95 
96 public:
97  void dispatch(messaging::BaseMsg* base_msg, void* object) const override {
98  using T = HandlerT;
99 
100  [[maybe_unused]] auto msg = static_cast<MsgT*>(base_msg);
101  [[maybe_unused]] auto elm = static_cast<ObjT*>(object);
102 
103  if constexpr (std::is_same_v<T, ActiveVoidFnType*>) {
104  fp();
105  } else if constexpr (std::is_same_v<T, ActiveTypedFnType<MsgT>*>) {
106  fp(msg);
107  } else if constexpr (std::is_same_v<T, ColTypedFnType*>) {
108  if constexpr (IsColTrait<ObjT>::value and not IsColMsgTrait<MsgT>::value) {
109  auto wrap_msg = static_cast<
110  vrt::collection::ColMsgWrap<ObjT, MsgT, vt::Message>*
111  >(base_msg);
112  fp(elm, &wrap_msg->getMsg());
113  } else {
114  fp(elm, msg);
115  }
116  } else if constexpr (std::is_same_v<T, ColMemberTypedFnType>) {
117  if constexpr (IsColTrait<ObjT>::value and not IsColMsgTrait<MsgT>::value) {
118  auto wrap_msg = static_cast<
119  vrt::collection::ColMsgWrap<ObjT, MsgT, vt::Message>*
120  >(base_msg);
121  (elm->*fp)(&wrap_msg->getMsg());
122  } else {
123  (elm->*fp)(msg);
124  }
125  } else if constexpr (std::is_same_v<ObjT, SentinelObject>) {
126  std::apply(fp, msg->getTuple());
127  } else {
128  std::apply(fp, std::tuple_cat(std::make_tuple(elm), msg->getTuple()));
129  }
130  }
131 
132 private:
133  HandlerT fp = nullptr;
134 };
