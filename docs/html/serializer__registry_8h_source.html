100 template <typename ObjT, typename SerializerT>
101 Registrar<ObjT, SerializerT>::Registrar() {
102  auto& reg = getRegistry<ObjT>();
103  index = static_cast<TypeIdx>(reg.size());
104 
105  debug_checkpoint(
106  "serializer registrar: %zu, ObjT=%s SerializerT=%s\n",
107  reg.size(),
108  typeid(ObjT).name(),
109  typeid(SerializerT).name()
110  );
111 
112  reg.emplace_back(
113  SerializerEntry<ObjT>{
114  index,
115  no_type_idx, // Set later when linkDerivedToBase is called
116  [=](void* s, ObjT& obj) {
117  auto& ser = *reinterpret_cast<SerializerT*>(s);
118  // Disable virtual serializer dispatch because we are already in a
119  // virtualSerialize and otherwise we will recurse indefinitely
120  ser.setVirtualDisabled(true);
121  ser | obj;
122  }
123  }
124  );
125 }
