77 template <typename Serializer, typename T, typename VectorAllocator>
78 void serialize(Serializer& s, std::vector<T, VectorAllocator>& vec) {
79  serializeVectorMeta(s, vec);
80  dispatch::serializeArray(s, vec.data(), vec.size());
81 
82  // make sure to account for reserved space when footprinting
83  s.addBytes(sizeof(T) * (vec.capacity() - vec.size()));
84 }

86 template <typename Serializer, typename VectorAllocator>
87 void serialize(Serializer& s, std::vector<bool, VectorAllocator>& vec) {
88  if (s.isFootprinting()) {
89  s.countBytes(vec);
90  return;
91  }
92 
93  serializeVectorMeta(s, vec);
94 
95  if (!s.isUnpacking()) {
96  for (bool elt : vec) {
97  s | elt;
98  }
99  } else {
100  for (size_t i = 0; i < vec.size(); ++i) {
101  bool elt = false;
102  s | elt;
103  vec[i] = elt;
104  }
105  }
106 }
