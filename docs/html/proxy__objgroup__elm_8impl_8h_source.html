80 template <typename ObjT>
81 template <auto f, typename... Params>
82 typename ProxyElm<ObjT>::PendingSendType
83 ProxyElm<ObjT>::send(Params&&... params) const {
84  using MsgT = typename ObjFuncTraits<decltype(f)>::MsgT;
85  if constexpr (std::is_same_v<MsgT, NoMsg>) {
86  using Tuple = typename ObjFuncTraits<decltype(f)>::TupleType;
87  using SendMsgT = messaging::ParamMsg<Tuple>;
88  auto msg = vt::makeMessage<SendMsgT>();
89  msg->setParams(std::forward<Params>(params)...);
90  auto const ctrl = proxy::ObjGroupProxy::getID(proxy_);
91  auto const han = auto_registry::makeAutoHandlerObjGroupParam<
92  ObjT, decltype(f), f, SendMsgT
93  >(ctrl);
94  return theObjGroup()->send(msg, han, node_);
95  } else {
96  auto msg = makeMessage<MsgT>(std::forward<Params>(params)...);
97  return sendMsg<MsgT, f>(msg);
98  }
99 
100  // Silence nvcc warning (no longer needed for CUDA 11.7 and up)
101  return ProxyElm<ObjT>::PendingSendType{std::nullptr_t{}};
102 }
