127 template <typename T, HandleEnum E, typename ProxyT>
128 Handle<T, E> Manager::makeHandleCollectiveObjGroup(
129  ProxyT proxy, std::size_t size, bool uniform_size
130 ) {
131  auto proxy_bits = proxy.getProxy();
132  ElemType sub = 0;
133  auto next_handle = ++cur_handle_obj_group_[proxy_bits];
134  auto key = HandleKey{typename HandleKey::ObjGroupTag{}, proxy_bits, next_handle};
135  auto han = Handle<T, E>{typename Handle<T, E>::NodeTagType{}, key, size};
136  holder_<T,E>[key].template addHandle<ObjGroupProxyType>(key, sub, han, size, uniform_size);
137  holder_footprint_[key] = [&holder = holder_<T,E>[key]]() {
138  return checkpoint::getMemoryFootprint(holder);
139  };
140  auto cb = vt::theCB()->makeBcast<
141  Manager, impl::ConstructMsg<T, E, ProxyT>, &Manager::finishMake<T, E, ProxyT>
142  >(proxy_);
143  auto msg = vt::makeMessage<impl::ConstructMsg<T, E, ProxyT>>(
144  impl::HandleData{key, size, 1}
145  );
146  proxy_.template reduce<vt::collective::PlusOp<impl::HandleData>>(msg.get(),cb);
147  return han;
148 }

150 template <typename T, HandleEnum E, typename ProxyT, typename LookupT>
151 HandleSet<T> Manager::makeHandleSetCollectiveObjGroup(
152  ProxyT,
153  LookupT max_lookup,
154  std::unordered_map<LookupT, std::size_t> const& map,
155  bool dense_start_with_zero,
156  bool uniform_size
157 ) {
158  using LookupType = LookupT;
159  using IndexType = typename HandleSet<T>::IndexType;
160  using SubType = SubHandle<T, vt::rdma::HandleEnum::StaticSize, IndexType>;
161  auto const num_nodes = vt::theContext()->getNumNodes();
162  IndexType range(static_cast<LookupType>(num_nodes), max_lookup);
163  auto proxy = SubType::template construct<Manager::staticHandleMap>(
164  false, range, dense_start_with_zero
165  );
166  HandleSet<T> set(typename HandleSet<T>::BuildSetTagType{});
167  auto const this_node = static_cast<LookupT>(vt::theContext()->getNode());
168  for (auto&& elm : map) {
169  IndexType idx(this_node, elm.first);
170  set.addHandle(idx, proxy.get()->addLocalIndex(idx, elm.second));
171  }
172  proxy.get()->makeSubHandles();
173  set.finishedInserts();
174  return set;

201  using SubType = SubHandle<T,E,IndexT>;
202  auto proxy = objgroup::proxy::Proxy<SubType>{han.proxy_};
203  proxy.get()->deleteHandle();
204  vt_debug_print(
205  normal, rdma,
206  "deleteHandleCollection: num deleted={}, num active={}\n",
207  proxy.get()->getNumDeletedHandles(),
208  proxy.get()->getNumActiveHandles()
209  );
210  // If all of the handles mapped here are deleted, destroy the sub-handle
211  // manager
212  if (proxy.get()->getNumDeletedHandles() == proxy.get()->getNumActiveHandles()) {
213  vt_debug_print(
214  normal, rdma,
215  "deleteHandleCollection: deleting handle collection: num={}\n",
216  proxy.get()->getNumDeletedHandles()
217  );
218  SubType::destroyCollective(proxy);
219  }
