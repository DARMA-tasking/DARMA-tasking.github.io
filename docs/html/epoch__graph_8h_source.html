102  template <typename SerializerT>
103  void serialize(SerializerT& s) {
104  s | epoch_;
105  s | user_label_;
106  std::size_t nc = successors_.size();
107  s | nc;
108  for (std::size_t i = 0; i < nc; i++) {
109  // This is inefficient because it potentially serializes vertices multiple
110  // times (if the structure is not a tree). But, given that this only
111  // executes as a diagnostic, it's not worth the trouble of keeping a
112  // lookup table
113  if (s.isUnpacking()) {
114  EpochGraph et;
115  s | et;
116  successors_.push_back(std::make_shared<EpochGraph>(std::move(et)));
117  } else {
118  s | *successors_[i];
119  }
120  }
121  s.skip(successors_);
122  }
