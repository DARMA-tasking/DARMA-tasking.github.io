279 template <typename SerializerT, typename T, typename... Args>
280 inline void serialize_impl(SerializerT& s, Kokkos::DynRankView<T,Args...>& view) {
281  using ViewType = Kokkos::DynRankView<T,Args...>;
282  using ArrayLayoutType = typename ViewType::traits::array_layout;
283 
284  // Serialize the label for the view which is used to construct a new view with
285  // the same label. Labels may not be unique and are for debugging Kokkos::View
286  auto const label = serializeViewLabel(s,view);
287 
288  // Serialize the total number of runtime dimensions
289  unsigned dims = 0;
290  if (!s.isUnpacking()) {
291  dims = view.rank();
292  }
293  s | dims;
294 
295  // Serialize the Kokkos layout data, including the extents, strides
296  ArrayLayoutType layout;
297 
298  // Make sure we serialize all 7 dimensions, instead of just `dims`.
299  if (s.isUnpacking()) {
300  serializeLayout<SerializerT>(s, 7, layout);
301  } else {
302  ArrayLayoutType layout_cur = view.layout();
303 
304  // We must set these to the invalid index (they are not set in the layout by
305  // default!). This ensures that when the DynRankView comes out after
306  // de-serialization, the number of ranks is correct.
307  for (int i = dims; i < 8; i++) {
308  layout_cur.dimension[i] = KOKKOS_INVALID_INDEX;
309  }
310 
311  serializeLayout<SerializerT>(s, 7, layout_cur);
312  }
313 
314  // Serialize the total number of elements in the Kokkos::View
315  size_t num_elms = view.size();
316  s | num_elms;
317  bool is_uninitialized = dims == 0 and num_elms == 0;
318  // Construct a view with the layout and use operator= to propagate out
319  if (s.isUnpacking()) {
320  if (is_uninitialized) {
321  view = ViewType{};
322  } else if (dims == 0) {
323  view = constructRankedView<ViewType,0>(label, std::make_tuple(layout));
324  } else if (dims == 1) {
325  view = constructRankedView<ViewType,1>(label, std::make_tuple(layout));
326  } else if (dims == 2) {
327  view = constructRankedView<ViewType,2>(label, std::make_tuple(layout));
328  } else if (dims == 3) {
329  view = constructRankedView<ViewType,3>(label, std::make_tuple(layout));
330  } else if (dims == 4) {
331  view = constructRankedView<ViewType,4>(label, std::make_tuple(layout));
332  } else if (dims == 5) {
333  view = constructRankedView<ViewType,5>(label, std::make_tuple(layout));
334  } else if (dims == 6) {
335  view = constructRankedView<ViewType,6>(label, std::make_tuple(layout));
336  } else if (dims == 7) {
337  view = constructRankedView<ViewType,7>(label, std::make_tuple(layout));
338  } else {
339  checkpointAssert(
340  false,
341  "Serializing Kokkos::DynRankView is only supported up to 7 dimensions"
342  );
343  }
344  }
345 
346  if (is_uninitialized) {
347  return;
348  }
349 
350  // Serialize whether the view is contiguous or not. Is this required?
351  bool is_contig = view.span_is_contiguous();
352  s | is_contig;
353 
354  DEBUG_PRINT_CHECKPOINT(
355  s, "label=%s: contig=%s, size=%zu, dims=%d\n",
356  label.c_str(), is_contig ? "true" : "false", num_elms, dims
357  );
358 
359  bool init = false;
360  if (!s.isUnpacking()) {
361  init = (view.use_count() > 0 || view.data() != nullptr);
362  }
363  s | init;
364 
365  if (init) {
366  auto host_view = Kokkos::create_mirror_view(CHECKPOINT_KOKKOS_WITHOUTINIT view);
367  using HostViewType = decltype(host_view);
368 
369  if (s.isPacking()) {
370  deepCopyWithLocalFence(host_view, view);
371  }
372 
373  // Serialize the actual data owned by the Kokkos::View
374  if (is_contig) {
375  // Serialize the data directly out of the data buffer
376  dispatch::serializeArray(s, host_view.data(), num_elms);
377  } else {
378  if (dims == 1) {
379  TraverseManual<SerializerT,HostViewType,1>::apply(s,host_view);
380  } else if (dims == 2) {
381  TraverseManual<SerializerT,HostViewType,2>::apply(s,host_view);
382  } else if (dims == 3) {
383  TraverseManual<SerializerT,HostViewType,3>::apply(s,host_view);
384  } else if (dims == 4) {
385  TraverseManual<SerializerT,HostViewType,4>::apply(s,host_view);
386  } else if (dims == 5) {
387  TraverseManual<SerializerT,HostViewType,5>::apply(s,host_view);
388  } else if (dims == 6) {
389  TraverseManual<SerializerT,HostViewType,6>::apply(s,host_view);
390  } else if (dims == 7) {
391  TraverseManual<SerializerT,HostViewType,7>::apply(s,host_view);
392  } else {
393  checkpointAssert(
394  false,
395  "Serializing Kokkos::DynRankView is only supported up to 7 dimensions"
396  " for non-contiguous views"
397  );
398  }
399  }
400 
401  if (s.isUnpacking()) {
402  deepCopyWithLocalFence(view, host_view);
403  }
404  }
405 }

407 template <typename SerializerT, typename T, typename... Args>
408 inline void serialize_impl(SerializerT& s, Kokkos::View<T,Args...>& view) {
409  using ViewType = Kokkos::View<T,Args...>;
410  using ArrayLayoutType = typename ViewType::traits::array_layout;
411 
412  static constexpr auto const rank_val = ViewType::Rank;
413 
414  checkpointAssert(
415  ViewType::traits::is_managed,
416  "Serialization not implemented for unmanaged views"
417  );
418 
419  // Serialize the label for the view which is used to construct a new view with
420  // the same label. Labels may not be unique and are for debugging Kokkos::View
421  auto const label = serializeViewLabel(s,view);
422 
423  // Serialize the total number of dimensions, including runtime+static dims
424  int rt_dim = 0;
425  if (!s.isUnpacking()) {
426  rt_dim = CountDims<ViewType, T>::numDims(view);
427  }
428  s | rt_dim;
429 
430 #if CHECKPOINT_KOKKOS_PACK_LAYOUT
431  // Serialize the Kokkos layout data, including the extents, strides
432  ArrayLayoutType layout;
433 
434  // This is ordered as so because the view.layout() might fail before proper
435  // initialization
436  if (s.isUnpacking()) {
437  serializeLayout<SerializerT>(s, rt_dim, layout);
438  } else {
439  ArrayLayoutType layout_cur = view.layout();
440  serializeLayout<SerializerT>(s, rt_dim, layout_cur);
441  }
442 
443  // Construct a view with the layout and use operator= to propagate out
444  if (s.isUnpacking()) {
445  view = constructView<ViewType>(label, std::make_tuple(layout));
446  }
447 #else
448  //
449  // This code for now is disabled by default
450  //
451  // Works only for Kokkos::LayoutLeft and Kokkos::LayoutRight
452  //
453  // Instead of serializing the layout struct data, serialize the extents that
454  // get propagate to the View from the layout
455  //
456  // Note: enabling this option will *not* work with Kokkos::LayoutStride. It
457  // will fail to compile with a static_assert: because LayoutStide is not
458  // extent constructible: traits::array_layout::is_extent_constructible!
459  //
460  constexpr auto dyn_dims = CountDims<ViewType, T>::dynamic;
461 
462  std::array<size_t, dyn_dims> extents_array;
463 
464  if (!s.isUnpacking()) {
465  // Set up the extents array
466  for (auto i = 0; i < dyn_dims; i++) {
467  extents_array[i] = view.extent(i);
468  }
469  }
470 
471  s | extents_array;
472 
473  // Construct a view with the layout and use operator= to propagate out
474  if (s.isUnpacking()) {
475  view = constructView<ViewType>(label, extents_array);
476  }
477 #endif
478 
479  // Serialize the total number of elements in the Kokkos::View
480  size_t num_elms = view.size();
481  s | num_elms;
482 
483  // Serialize whether the view is contiguous or not. Is this required?
484  bool is_contig = view.span_is_contiguous();
485  s | is_contig;
486 
487  DEBUG_PRINT_CHECKPOINT(
488  s, "label=%s: contig=%s, size=%zu, rt_dim=%d\n",
489  label.c_str(), is_contig ? "true" : "false", num_elms, rt_dim
490  );
491 
492  bool init = false;
493  if (!s.isUnpacking()) {
494  init = (view.use_count() > 0 || view.data() != nullptr);
495  }
496  s | init;
497 
498  if (init) {
499  auto host_view = Kokkos::create_mirror_view(view);
500  using HostViewType = decltype(host_view);
501 
502  if (s.isPacking()) {
503  deepCopyWithLocalFence(host_view, view);
504  }
505 
506  // Serialize the actual data owned by the Kokkos::View
507  if (is_contig) {
508  // Serialize the data directly out of the data buffer
509  dispatch::serializeArray(s, host_view.data(), num_elms);
510  } else {
511  // Serialize manually traversing the data with Kokkos::View::operator()(...)
512 
513 #if CHECKPOINT_KOKKOS_NDIM_TRAVERSE
514  using CountDimType = CountDims<ViewType>;
515  using BaseType = typename CountDimType::BaseT;
516 
517  constexpr auto dims = CountDimType::dynamic;
518  auto fn = [&s](BaseType& elm){
519  s | elm;
520  };
521 
522  TraverseRecursive<HostViewType,T,dims,decltype(fn)>::apply(host_view,fn);
523 #else
524  TraverseManual<SerializerT,HostViewType,rank_val>::apply(s,host_view);
525 #endif
526  }
527 
528  if (s.isUnpacking()) {
529  deepCopyWithLocalFence(view, host_view);
530  }
531  }
532 }
