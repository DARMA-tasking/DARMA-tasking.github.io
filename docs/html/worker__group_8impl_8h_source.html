158 template <typename WorkerT>
159 void WorkerGroupAny<WorkerT>::spawnWorkers() {
160  using namespace std::placeholders;
161 
162  debug_print(
163  worker, node,
164  "WorkerGroup: spawnWorkers: num_workers_={}\n", num_workers_
165  );
166 
167  vtAssert(workers_.size() >= num_workers_, "Must be correct size");
168 
169  for (int i = 0; i < num_workers_; i++) {
170  WorkerIDType const worker_id = i;
171  workers_[i] = std::make_unique<WorkerT>(
172  worker_id, num_workers_, finished_fn_
173  );
174  }
175 
176  for (auto&& elm : workers_) {
177  elm->spawn();
178  }
179 
180  initialized_ = true;
181 
182  // Give every worker at least one work unit (for termination purposes)
183  auto initial_work_unit = []{};
184  enqueueAllWorkers(initial_work_unit);
185 }
