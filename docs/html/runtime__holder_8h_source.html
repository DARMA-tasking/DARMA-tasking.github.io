55 /*
56  * Holder class for `vt::runtime::Runtime' that essentially acts as a
57  * std::unique_ptr<Runtime>, but specialized for the Runtime struct. It enables
58  * user management of Runtime instances with a value-semantic container for
59  * safety.
60  */
61 struct RuntimeHolder {
62  using PointerType = vt::runtime::Runtime*;
63  using ReferenceType = vt::runtime::Runtime&;

66  : rt_(in_rt), owner_(in_owner)
67  { }
68  RuntimeHolder(std::nullptr_t) : rt_(nullptr) { }
69  RuntimeHolder(RuntimeHolder&&) = default;
70  RuntimeHolder(RuntimeHolder const&) = delete;
71 
72  virtual ~RuntimeHolder() { release(); }
73 
74  RuntimeHolder& operator=(std::nullptr_t) { release(); return *this; }
75  bool operator==(RuntimeHolder const& other) const { return rt_ == other.rt_; }
76  bool operator!=(RuntimeHolder const& other) const { return rt_ != other.rt_; }
77  bool operator==(std::nullptr_t) const { return rt_ == nullptr; }
78  bool operator!=(std::nullptr_t) const { return rt_ != nullptr; }
79  operator bool() const { return rt_ != nullptr; }

81  RuntimeHolder& operator=(RuntimeHolder&& holder) {
82  release();
83  if (holder.rt_) {

92  }

98 private:

102  vtAssert(rt_, "RuntimeHolder: holder is null: can not be dereferenced!");
103  return *rt_;
104  }
105 
106  void release() {
107  if (owner_ && rt_ && rt_->isFinalizable() && rt_->hasSchedRun() && rt_->isTerminated()) {
108  rt_->finalize();
109  }
110  rt_ = nullptr;
111  }
112 
113 private:
