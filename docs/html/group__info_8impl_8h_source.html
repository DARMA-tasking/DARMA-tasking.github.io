93  );
94 
95  if (new_size < min_spanning_tree_size) {
96  vt_debug_print(
97  normal, group,

99  new_size
100  );
101

107  );

111  "Info::groupSetupHandler: op={}, parent={}\n",
112  op_id, parent
113  );

153  if (parent != this_node) {
154  theMsg()->sendMsg<GroupOnlyMsg, Info::groupTriggerHandler>(
155  parent, contmsg

160  }

164  auto const& c = region->head();

168 
169  vt_debug_print(

172  );
173 
174  auto op1 = theGroup()->registerContinuation(parent_cont);
175  auto l1 = static_cast<RangeType*>(region.get());
176  auto c_msg = makeMessage<MsgT>(
177  c, c_size, group, op1, group_total_size, this_node, l1
178  );
179  if (c != this_node) {

181  } else {
182  groupSetupHandler(c_msg.get());
183  }
184  });
185

191  );
192 
193  theGroup()->initializeRemoteGroup(
194  group, std::move(region), is_static, group_size
195  );
196  // end parent region
197 
198  auto iter = theGroup()->remote_group_info_.find(group);
199  vtAssertExpr(iter != theGroup()->remote_group_info_.end());
200  auto info = iter->second.get();
201  info->wait_count_ += num_children;
202  }
203 }
