53 template <typename ColT>
54 std::tuple<EpochType, VirtualProxyType> CollectionManager::makeCollection(
55  param::ConstructParams<ColT>& po
56 ) {
57  auto const is_migratable = po.migratable_;
58  auto const is_collective = po.collective_;
59 
60  // Generate a new proxy for this new collection
61  auto const proxy_bits = makeCollectionProxy(is_collective, is_migratable);
62  po.proxy_bits_ = proxy_bits;
63 
64  if (not is_collective) {
65  auto ep = theTerm()->makeEpochRooted(
66  "collection construction", term::UseDS{false}
67  );
68  theMsg()->pushEpoch(ep);
69  using MsgType = param::ConstructParamMsg<ColT>;
70  auto m = makeMessage<MsgType>(po);
71  theMsg()->broadcastMsg<MsgType, makeCollectionHandler>(m);
72  theMsg()->popEpoch(ep);
73  theTerm()->finishedEpoch(ep);
74  return std::make_tuple(ep, proxy_bits);
75  } else {
76  auto ep = theTerm()->makeEpochCollective("collection construction");
77  theMsg()->pushEpoch(ep);
78  makeCollectionImpl(po);
79  theMsg()->popEpoch(ep);
80  theTerm()->finishedEpoch(ep);
81  return std::make_tuple(ep, proxy_bits);
82  }
83 }

104 template <typename ColT>
105 void CollectionManager::makeCollectionImpl(param::ConstructParams<ColT>& po) {
106  using IndexType = typename ColT::IndexType;
107 
108  if (not po.has_bounds_ and po.bulk_inserts_.size() == 1) {
109  po.bounds_ = po.bulk_inserts_[0];
110  po.has_bounds_ = true;
111  }
112 
113  auto const proxy = po.proxy_bits_;
114  auto const has_dynamic_membership = po.dynamic_membership_;
115  auto const this_node = theContext()->getNode();
116  auto const has_bounds = po.has_bounds_;
117  auto const bounds = has_bounds ? po.bounds_ : IndexType{};
118  auto const label = po.label_;
119 
120  // Setup a proper default map if none is explicitly specified by the user
121  if (po.map_han_ == uninitialized_handler and po.map_object_ == no_obj_group) {
122  if (has_bounds) {
123  po.map_han_ = CollectionManager::getDefaultMap<ColT>();
124  } else {
125  po.map_object_ = mapping::UnboundedDefaultMap<IndexType>::construct();
126  }
127  }
128 
129  auto const map_han = po.map_han_;
130  auto const map_object = po.map_object_;
131 
132  // Invoke getCollectionLM() to create a new location manager instance for
133  // this collection
134  theLocMan()->getCollectionLM<IndexType>(proxy);
135 
136  // Insert action on cleanup for this collection
137  addCleanupFn<ColT>(proxy);
138 
139  // Insert some typeless basic meta-data about the collection
140  insertCollectionInfo(proxy, map_han);
141 
142  // Insert the typed meta-data for this new collection, along with creating
143  // the meta-data collection holder for elements
144  insertMetaCollection<ColT>(
145  label, proxy, map_han, has_dynamic_membership, map_object, has_bounds, bounds
146  );
147 
148  std::size_t global_constructed_elms = 0;
149 
150  if (po.bulk_insert_bounds_) {
151  vtAssert(
152  po.bulk_inserts_.size() == 0,
153  "To bulk insert bounds, the collection must have no other bulk insertions"
154  );
155  vtAssert(
156  po.list_inserts_.size() == 0,
157  "To bulk insert bounds, the collection must have no other list insertions"
158  );
159  }
160 
161  if (po.bulk_insert_bounds_ and po.has_bounds_) {
162  po.bulk_inserts_.push_back(po.bounds_);
163  }
164 
165  if (!po.bulk_inserts_.empty() || !po.list_inserts_.empty()) {
166  auto cons_fn = po.template getConsFn<ColT>();
167 
168  // Do all bulk insertions
169  for (auto&& range : po.bulk_inserts_) {
170  range.foreach([&](IndexType idx) {
171  if (elementMappedHere(map_han, map_object, idx, bounds)) {
172  makeCollectionElement<ColT>(proxy, idx, this_node, cons_fn);
173  }
174  global_constructed_elms++;
175  });
176  }
177 
178  // Do all list insertions
179  for (auto&& list_fn : po.list_inserts_) {
180  list_fn([&](IndexType idx) {
181  if (elementMappedHere(map_han, map_object, idx, bounds)) {
182  makeCollectionElement<ColT>(proxy, idx, this_node, cons_fn);
183  }
184  global_constructed_elms++;
185  });
186  }
187  }
188 
189  // Do all 'here' insertions
190  for (auto&& elm : po.list_insert_here_) {
191  auto const idx = std::get<0>(elm);
192  detail::ContainableElementFn<ColT> c{std::move(std::get<1>(elm))};
193  makeCollectionElement<ColT>(proxy, idx, this_node, std::move(c));
194  }
195 
196  if (global_constructed_elms != 0) {
197  // Construct a underlying group for the collection
198  constructGroup<ColT>(proxy);
199  }
200 }

202 template <typename ColT, typename Callable>
203 void CollectionManager::makeCollectionElement(
204  VirtualProxyType const proxy, typename ColT::IndexType idx,
205  NodeType const mapped_node, Callable&& cons_fn, bool zero_reduce_stamp
206 ) {
207  using IndexType = typename ColT::IndexType;
208  using IdxContextHolder = CollectionContextHolder<IndexType>;
209 
210  auto prev_index = IdxContextHolder::index();
211  auto prev_proxy = IdxContextHolder::proxy();
212 
213  // Set the current context index to `idx`. This enables the user to
214  // query the index of their collection element in the constructor
215  IdxContextHolder::set(&idx, proxy);
216 
217  // Invoke the construct function for the collection element
218  auto elm = cons_fn(idx);
219 
220  // Through the attorney, setup all the properties on the newly constructed
221  // collection element: index and proxy
222  CollectionTypeAttorney::setup(elm.get(), idx, proxy);
223 
224  // If true, set the reduce stamp to zero so we can correctly identify new
225  // collection elements during an insertion epoch to compute the min stamp
226  // after insertions complete
227  if (zero_reduce_stamp) {
228  elm->zeroReduceStamp();
229  }
230 
231  // Insert the element into the managed holder for elements
232  insertCollectionElement<ColT>(std::move(elm), proxy, idx, mapped_node);
233 
234  // Clear the current index context
235  IdxContextHolder::clear();
236 
237  // Restore back the previous context for recursive constructions
238  IdxContextHolder::set(prev_index, prev_proxy);
239 }

250 template <typename IdxT>
251 NodeType CollectionManager::getElementMapping(
252  HandlerType map_han, ObjGroupProxyType map_object, IdxT idx, IdxT bounds
253 ) {
254  if (map_han != uninitialized_handler) {
255  // Get the map handler function
256  bool const is_functor =
257  auto_registry::HandlerManagerType::isHandlerFunctor(map_han);
258  auto const& map_fn = is_functor ?
259  auto_registry::getAutoHandlerFunctorMap(map_han) :
260  auto_registry::getAutoHandlerMap(map_han);
261  auto const num_nodes = theContext()->getNumNodes();
262  auto const mapped_node = map_fn->dispatch(&idx, &bounds, num_nodes);
263 
264  return mapped_node;
265  }
266 
267  if (map_object != no_obj_group) {
268  objgroup::proxy::Proxy<mapping::BaseMapper<IdxT>> p{map_object};
269  auto map_obj_ptr = p.get();
270  auto num_nodes = theContext()->getNumNodes();
271  return map_obj_ptr->map(&idx, idx.ndims(), num_nodes);
272  }
273 
274  vtAbort("No valid map fn or object group specified for the collection");
275  return uninitialized_destination;
276 }
