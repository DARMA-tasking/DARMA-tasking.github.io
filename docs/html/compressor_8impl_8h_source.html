57 template <typename StreamLike>
58 bool Compressor::writeImpl(
59  StreamLike& s, uint8_t const* buffer, std::size_t const size, bool finish_
60 ) {
61  vtAssert(enc_, "Must have a valid compressor");
62 
63  uint8_t const* cur = buffer;
64  std::size_t rem = size;
65 
66  while (rem > 0 or (finish_ and not BrotliEncoderIsFinished(enc_))) {
67  std::size_t avail_out = buf_size_;
68  uint8_t* next_out = out_buf_.get();
69 
70  auto op = finish_ ? BROTLI_OPERATION_FINISH : BROTLI_OPERATION_PROCESS;
71  bool ret = BrotliEncoderCompressStream(
72  enc_, op, &rem, &cur, &avail_out, &next_out, nullptr
73  );
74  if (!ret) {
75  vtAbort("Failed to stream compression data out\n");
76  return false;
77  }
78  s.write(reinterpret_cast<char*>(out_buf_.get()), buf_size_ - avail_out);
79  }
80 
81  return true;
82 }
