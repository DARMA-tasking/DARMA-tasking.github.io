309 template <typename U>
310 RunnableMaker<U> makeRunnable(
311  MsgSharedPtr<U> const& msg, bool is_threaded, HandlerType handler,
312  NodeType from, auto_registry::RegistryTypeEnum han_type =
313  auto_registry::RegistryTypeEnum::RegGeneral
314 ) {
315  auto r = std::make_unique<RunnableNew>(msg, is_threaded);
316  if (han_type == auto_registry::RegistryTypeEnum::RegVrt or
317  han_type == auto_registry::RegistryTypeEnum::RegGeneral) {
318  r->template addContext<ctx::Trace>(msg, handler, from, han_type);
319  }
320  r->template addContext<ctx::FromNode>(from);
321  r->template addContext<ctx::SetContext>(r.get());
322  return RunnableMaker<U>{std::move(r), msg, handler, han_type, from};
323 }
