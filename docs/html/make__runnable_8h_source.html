336 template <typename U>
337 RunnableMaker<U> makeRunnable(
338  MsgSharedPtr<U> const& msg, bool is_threaded, HandlerType handler, NodeType from
339 ) {
340  auto r = new RunnableNew(msg, is_threaded);
341 #if vt_check_enabled(trace_enabled)
342  auto const han_type = HandlerManager::getHandlerRegistryType(handler);
343  if (han_type == auto_registry::RegistryTypeEnum::RegVrt or
344  han_type == auto_registry::RegistryTypeEnum::RegGeneral or
345  han_type == auto_registry::RegistryTypeEnum::RegObjGroup) {
346  r->addContextTrace(msg, handler, from);
347  }
348 #endif
349  r->addContextSetContext(r, from);
350  return RunnableMaker<U>{r, msg, handler, from};
351 }
