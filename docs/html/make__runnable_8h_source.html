308 template <typename U>
309 RunnableMaker<U> makeRunnable(
310  MsgSharedPtr<U> const& msg, bool is_threaded, HandlerType handler, NodeType from
311 ) {
312  auto r = new RunnableNew(msg, is_threaded);
313 #if vt_check_enabled(trace_enabled)
314  auto const han_type = HandlerManager::getHandlerRegistryType(handler);
315  if (han_type == auto_registry::RegistryTypeEnum::RegVrt or
316  han_type == auto_registry::RegistryTypeEnum::RegGeneral or
317  han_type == auto_registry::RegistryTypeEnum::RegObjGroup) {
318  r->addContextTrace(msg, handler, from);
319  }
320 #endif
321  r->addContextSetContext(r, from);
322  return RunnableMaker<U>{r, msg, handler, from};
323 }
