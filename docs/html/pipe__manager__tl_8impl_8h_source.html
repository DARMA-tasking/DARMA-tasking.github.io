144 template <auto f, bool is_bcast, typename ProxyT>
145 auto PipeManagerTL::makeCallbackProxy(ProxyT proxy) {
146  bool const persist = true;
147  bool const send_back = false;
148  bool const dispatch = true;
149  auto const pipe_id = makePipeID(persist, send_back);
150  newPipeState(pipe_id, persist, dispatch, -1, -1, 0);
151 
152  using Trait = ObjFuncTraits<decltype(f)>;
153  using ObjT = typename Trait::ObjT;
154  using MsgT = typename Trait::MsgT;
155  using RetType = typename Trait::template WrapType<CallbackRetType>;
156 
157  if constexpr (detection::is_detected<hasIdx_t, ProxyT>::value) {
158  if constexpr (std::is_same_v<MsgT, NoMsg>) {
159  using Tuple = typename Trait::TupleType;
160  using NormMsgT = messaging::ParamMsg<Tuple>;
161  using WrapMsgT = vrt::collection::ColMsgWrap<ObjT, NormMsgT, vt::Message>;
162  auto han = auto_registry::makeAutoHandlerCollectionMemParam<
163  ObjT, decltype(f), f, WrapMsgT
164  >();
165  if constexpr (is_bcast) {
166  auto vrt_handler = vrt::collection::makeVrtDispatch<NormMsgT, ObjT>();
167  return RetType{
168  callback::cbunion::RawBcastColDirTag, pipe_id, han, vrt_handler,
169  proxy.getProxy()
170  };
171  } else {
172  auto cb = RetType(callback::cbunion::RawSendColMsgTag, pipe_id);
173  addListenerAny<NormMsgT>(
174  cb.getPipe(),
175  std::make_unique<callback::CallbackProxySend<ObjT, NormMsgT>>(han, proxy)
176  );
177  return cb;
178  }
179  } else {
180  HandlerType han = uninitialized_handler;
181  if constexpr (Trait::is_member) {
182  han = auto_registry::makeAutoHandlerCollectionMem<ObjT, MsgT, f>();
183  } else {
184  han = auto_registry::makeAutoHandlerCollection<ObjT, MsgT, f>();
185  }
186  if constexpr (is_bcast) {
187  auto vrt_handler = vrt::collection::makeVrtDispatch<MsgT, ObjT>();
188  return RetType{
189  callback::cbunion::RawBcastColDirTag, pipe_id, han, vrt_handler,
190  proxy.getProxy()
191  };
192  } else {
193  auto cb = RetType(callback::cbunion::RawSendColMsgTag, pipe_id);
194  addListenerAny<MsgT>(
195  cb.getPipe(),
196  std::make_unique<callback::CallbackProxySend<ObjT, MsgT>>(han, proxy)
197  );
198  return cb;
199  }
200  }
201  } else {
202  auto const proxy_bits = proxy.getProxy();
203  auto const ctrl = objgroup::proxy::ObjGroupProxy::getID(proxy_bits);
204  HandlerType han = uninitialized_handler;
205  if constexpr (std::is_same_v<MsgT, NoMsg>) {
206  using Tuple = typename Trait::TupleType;
207  using PMsgT = messaging::ParamMsg<Tuple>;
208  han = auto_registry::makeAutoHandlerObjGroupParam<
209  ObjT, decltype(f), f, PMsgT
210  >(ctrl);
211  } else {
212  han = auto_registry::makeAutoHandlerObjGroup<ObjT,MsgT,f>(ctrl);
213  }
214  if constexpr (is_bcast) {
215  return RetType{
216  callback::cbunion::RawBcastObjGrpTag, pipe_id, han, proxy_bits
217  };
218  } else {
219  auto const dest_node = proxy.getNode();
220  return RetType{
221  callback::cbunion::RawSendObjGrpTag, pipe_id, han, proxy_bits, dest_node
222  };
223  }
224  }
225 
226  // unnecessary, but to make nvcc happy
227  return RetType{};
228 }

230 template <auto f, bool is_bcast>
231 auto PipeManagerTL::makeCallbackSingle([[maybe_unused]] NodeType node) {
232  auto const new_pipe_id = makePipeID(true,false);
233 
234  using Trait = FuncTraits<decltype(f)>;
235  using MsgT = typename Trait::MsgT;
236 
237  HandlerType han = uninitialized_handler;
238  if constexpr (std::is_same_v<MsgT, NoMsg>) {
239  using SendMsgT = messaging::ParamMsg<typename Trait::TupleType>;
240  han = auto_registry::makeAutoHandlerParam<decltype(f), f, SendMsgT>();
241  } else {
242  han = auto_registry::makeAutoHandler<MsgT,f>();
243  }
244 
245  using RetType = typename Trait::template WrapType<CallbackRetType>;
246  if constexpr (is_bcast) {
247  return RetType{callback::cbunion::RawBcastMsgTag, new_pipe_id, han};
248  } else {
249  return RetType{callback::cbunion::RawSendMsgTag, new_pipe_id, han, node};
250  }
251 
252  // unnecessary, but to make nvcc happy
253  return RetType{};
254 }
