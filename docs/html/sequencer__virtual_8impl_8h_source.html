129 template <typename SeqTag, template <typename> class SeqTrigger>
130 template <typename VcT, typename MsgT, ActiveVrtTypedFnType<MsgT, VcT> *f>
131 void TaggedSequencerVrt<SeqTag, SeqTrigger>::sequenceVrtMsg(
132  MsgT* msg, VcT* vrt
133 ) {
134  auto const& is_tag_type = envelopeIsTagType(msg->env);
135  TagType const& msg_tag = is_tag_type ? envelopeGetTag(msg->env) : no_tag;
136 
137  debug_print(
138  sequence, node,
139  "sequenceVrtMsg: arrived: msg={}, tag={}\n", print_ptr(msg), msg_tag
140  );
141 
142  bool const has_match =
143  SeqStateMatcherType<VcT, MsgT, f>::hasMatchingAction(msg_tag);
144 
145  debug_print(
146  sequence, node,
147  "sequenceVrtMsg: arriving: msg={}, has_match={}, tag={}\n",
148  print_ptr(msg), print_bool(has_match), msg_tag
149  );
150 
151  // reference the arrival message to keep it alive past normal lifetime, in
152  // case it's buffered or put into the scheduler
153  auto pmsg = promoteMsg(msg);
154 
155  if (has_match) {
156  auto action = SeqStateMatcherType<VcT, MsgT, f>::getMatchingAction(msg_tag);
157  auto handle_msg_action = [this,action,pmsg,vrt]{
158  this->lookupContextExecute(
159  action.seq_id, action.generateCallable(pmsg.get(), vrt)
160  );
161  };
162 
163  // skip the work queue and directly execute
164  if (seq_skip_queue) {
165  handle_msg_action();
166  } else {
167  theVirtualSeq()->enqueue(handle_msg_action);
168  }
169  } else {
170  // nothing was found so the message must be buffered and wait an action
171  // being posted
172 
173  // buffer the unmatched messaged until a trigger is posted for it that
174  // matches
175  SeqStateMatcherType<VcT, MsgT, f>::bufferUnmatchedMessage(pmsg, msg_tag);
176  }
177 }

198 template <typename SeqTag, template <typename> class SeqTrigger>
199 template <typename VcT, typename MsgT, ActiveVrtTypedFnType<MsgT, VcT> *f>
200 void TaggedSequencerVrt<SeqTag, SeqTrigger>::wait_on_trigger(
201  TagType const& tag, SeqActionType<MsgT, VcT> action
202 ) {
203  theTerm()->produce();
204 
205  this->assertValidContext();
206 
207  SeqType const seq_id = this->getSeqID();
208  SeqNodePtrType node = this->getNode(seq_id);
209 
210  vtAssert(node != nullptr, "Must have node from context");
211 
212  bool const seq_ready = node->isReady();
213 
214  debug_print(
215  sequence_vrt, node,
216  "SequencerVrt: wait: tag={}: context seq id={}, node={}, blocked={}, "
217  "ready={}\n",
218  tag, seq_id, PRINT_SEQ_NODE_PTR(node),
219  print_bool(node->isBlockedNode()), print_bool(seq_ready)
220  );
221 
222  auto deferred_wait_action = [tag,action,node,seq_id]() -> bool {
223  bool const has_match = SeqStateMatcherType<VcT,MsgT,f>::hasMatchingMsg(tag);
224 
225  if (has_match) {
226  auto msg = SeqStateMatcherType<VcT, MsgT, f>::getMatchingMsg(tag);
227 
228  auto const& cur_proxy = theVirtualSeq()->getCurrentVirtualProxy();
229  auto vrt_context = theVirtualManager()->getVirtualByProxy(cur_proxy);
230  action.runAction(static_cast<VcT*>(vrt_context), msg.get());
231  }
232 
233  debug_print(
234  sequence_vrt, node,
235  "SequencerVrt: {}: tag={}: node={}, has_match={}, "
236  "is_blocked={}\n",
237  has_match ? "wait ran *immediately*" : "wait registered", tag,
238  PRINT_SEQ_NODE_PTR(node), print_bool(has_match),
239  print_bool(node->isBlockedNode())
240  );
241 
242  bool const should_block = not has_match;
243 
244  node->setBlockedOnNode(eSeqConstructType::WaitConstruct, should_block);
245 
246  if (has_match) {
247  debug_print(
248  sequence_vrt, node,
249  "SequencerVrt: activating next node: seq={}, node={}, blocked={}\n",
250  seq_id, PRINT_SEQ_NODE_PTR(node), print_bool(node->isBlockedNode())
251  );
252 
253  node->activate();
254  } else {
255  // buffer the action to wait for a matching message
256 
257  auto msg_recv_trigger = [node,seq_id,action,tag](MsgT* msg, VcT* vc){
258  debug_print(
259  sequence_vrt, node,
260  "SequencerVrt: msg_recv_trigger: seq={}, tag={}, node={}, blocked={}, "
261  "msg={}\n",
262  seq_id, tag, PRINT_SEQ_NODE_PTR(node),
263  print_bool(node->isBlockedNode()), print_ptr(msg)
264  );
265 
266  action.runAction(vc, msg);
267 
268  vtAssert(node != nullptr, "node must not be nullptr");
269 
270  node->setBlockedOnNode(eSeqConstructType::WaitConstruct, false);
271  node->activate();
272  };
273 
274  // buffer the action because there is not a matching message to trigger
275  // the message
276  SeqStateMatcherType<VcT, MsgT, f>::bufferUnmatchedAction(
277  msg_recv_trigger, seq_id, tag
278  );
279  }
280 
281  return should_block;
282  };
283 
284  if (seq_ready) {
285  // run it here, right now
286  bool const has_match = not deferred_wait_action();
287 
288  debug_print(
289  sequence_vrt, node,
290  "SequencerVrt: executed wait: has_match={}: seq_id={}\n",
291  print_bool(has_match), seq_id
292  );
293  } else {
294  node->addSequencedClosure(deferred_wait_action);
295  }
296 }
