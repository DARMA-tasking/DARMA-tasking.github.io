69 template <
70  typename T,
71  HandleEnum E,
72  typename ColT,
73  typename ProxyT,
74  typename IndexT
75 >
76 Handle<T, E, IndexT> Manager::makeCollectionHandles(
77  ProxyT collection_proxy, std::size_t idx_size, bool uniform_size,
78  RDMA_HandleType in_next_handle, vt::HandlerType map_han, IndexT in_range
79 ) {
80  using ElementListener = vrt::collection::listener::ElementEventEnum;
81  using SubType = SubHandle<T, vt::rdma::HandleEnum::StaticSize, IndexT>;
82  using BaseIdxType = vt::index::BaseIndex;
83  using ListenerType = vrt::collection::listener::ListenFnType<IndexT>;
84 
85  RDMA_HandleType next_handle = in_next_handle;
86  auto proxy_bits = collection_proxy.getCollectionProxy();
87  auto idx = collection_proxy.getElementProxy().getIndex();
88  IndexT range = in_range;
89  if (in_next_handle == no_rdma_handle) {
90  range = theCollection()->getRange<ColT>(proxy_bits);
91  auto lin = vt::mapping::linearizeDenseIndexRowMajor(&idx, &range);
92  next_handle = ++cur_handle_collection_[proxy_bits][lin];
93  }
94 
95  debug_print(
96  rdma, node,
97  "CollectionHandle: next_handle={}, idx={}, range={}\n",
98  next_handle, idx, range
99  );
100 
101  objgroup::proxy::Proxy<SubType> sub_proxy;
102  auto iter = collection_to_manager_[proxy_bits].find(next_handle);
103  if (iter == collection_to_manager_[proxy_bits].end()) {
104  // First time this handle is being created on this node
105  if (map_han == -1) {
106  map_han = vrt::collection::UniversalIndexHolder<>::getMap(proxy_bits);
107  }
108  sub_proxy = SubType::construct(true, range, true, map_han);
109  // Register the migration listener
110  ListenerType fn = [=](ElementListener event, IndexT lidx) {
111  if (event == ElementListener::ElementMigratedOut) {
112  sub_proxy.get()->migratedOutIndex(lidx);
113  } else if (event == ElementListener::ElementMigratedIn) {
114  sub_proxy.get()->migratedInIndex(lidx);
115  }
116  };
117  theCollection()->template registerElementListener<ColT>(proxy_bits, fn);
118 
119  // Count the number of local handles that should exist here. We can't use the
120  // cached value in the collection manager since this might be invoked in the
121  // an element's constructor before they are all created.
122  auto this_node = theContext()->getNode();
123  auto num_nodes = theContext()->getNumNodes();
124  auto min_node_mapped = theContext()->getNumNodes();
125  std::size_t local_count = 0;
126  auto map_fn = auto_registry::getHandlerMap(map_han);
127  range.foreach([&](IndexT cur_idx) {
128  auto* cur_idx_ptr = static_cast<BaseIdxType*>(&cur_idx);
129  auto* range_ptr = static_cast<BaseIdxType*>(&range);
130  auto home_node = map_fn(cur_idx_ptr, range_ptr, num_nodes);
131  if (home_node == this_node) {
132  local_count++;
133  }
134  min_node_mapped = std::min(home_node, min_node_mapped);
135  });
136 
137  // If LB is enabled then we need to register an afterLB listener
138 # if backend_check_enabled(lblite)
139  debug_print(
140  rdma, node,
141  "CollectionHandle: registering LB listener\n"
142  );
143  theLBManager()->registerListenerAfterLB([=](PhaseType){
144  sub_proxy.get()->afterLB();
145  });
146 # endif
147 
148  // Set the number of expected handles for next time we run through this code
149  sub_proxy.get()->setCollectionExpected(local_count);
150 
151  if (local_count == 0) {
152  sub_proxy.get()->makeSubHandles();
153  }
154 
155  collection_to_manager_[proxy_bits][next_handle] = sub_proxy.getProxy();
156 
157  // If this node is the min node mapped, it is responsible for informing all
158  // other nodes about the RDMA handle creation, since it uses a collective
159  // ObjGroup and all nodes might not have a collection element mapped to
160  // them.
161  if (this_node == min_node_mapped and in_next_handle == no_rdma_handle) {
162  auto msg = makeMessage<impl::InformRDMAMsg<ProxyT,IndexT>>(
163  collection_proxy, next_handle, uniform_size, map_han, range
164  );
165  proxy_.template broadcast<
166  impl::InformRDMAMsg<ProxyT,IndexT>,
167  &Manager::informCollectionRDMA<T,E,ProxyT,ColT>
168  >(msg.get());
169  }
170  } else {
171  auto sub_proxy_bits = iter->second;
172  sub_proxy = objgroup::proxy::Proxy<SubType>(sub_proxy_bits);
173  }
174 
175  if (in_next_handle == no_rdma_handle) {
176  // Add the handle
177  auto handle = sub_proxy.get()->addLocalIndex(idx, idx_size);
178 
179  // Check if all mapped handles are now created on this node
180  auto expected = sub_proxy.get()->getCollectionExpected();
181  auto current = sub_proxy.get()->getNumHandles();
182 
183  debug_print(
184  rdma, node,
185  "CollectionHandle: expected={}, current={}\n", expected, current
186  );
187 
188  // All the local handles have been added
189  if (current == expected) {
190  sub_proxy.get()->makeSubHandles();
191  }
192 
193  return handle;
194  } else {
195  return Handle<T, E, IndexT>();
196  }
197 }
