69 template <
70  typename T,
71  HandleEnum E,
72  typename ColT,
73  typename ProxyT,
74  typename IndexT
75 >
76 Handle<T, E, IndexT> Manager::makeCollectionHandles(
77  ProxyT collection_proxy, std::size_t idx_size, bool uniform_size,
78  RDMA_HandleType in_next_handle, vt::HandlerType map_han, IndexT in_range
79 ) {
80  using ElementListener = vrt::collection::listener::ElementEventEnum;
81  using SubType = SubHandle<T, vt::rdma::HandleEnum::StaticSize, IndexT>;
82  using ListenerType = vrt::collection::listener::ListenFnType<IndexT>;
83 
84  RDMA_HandleType next_handle = in_next_handle;
85  auto proxy_bits = collection_proxy.getCollectionProxy();
86  auto idx = collection_proxy.getElementProxy().getIndex();
87  IndexT range = in_range;
88  if (in_next_handle == no_rdma_handle) {
89  range = theCollection()->getRange<ColT>(proxy_bits);
90  auto lin = vt::mapping::linearizeDenseIndexRowMajor(&idx, &range);
91  next_handle = ++cur_handle_collection_[proxy_bits][lin];
92  }
93 
94  vt_debug_print(
95  normal, rdma,
96  "CollectionHandle: next_handle={}, idx={}, range={}\n",
97  next_handle, idx, range
98  );
99 
100  objgroup::proxy::Proxy<SubType> sub_proxy;
101  auto iter = collection_to_manager_[proxy_bits].find(next_handle);
102  if (iter == collection_to_manager_[proxy_bits].end()) {
103  // First time this handle is being created on this node
104  if (map_han == -1) {
105  map_han = theCollection()->getTypelessHolder().getMap(proxy_bits);
106  }
107  sub_proxy = SubType::construct(true, range, true, map_han);
108  // Register the migration listener
109  ListenerType fn = [=](ElementListener event, IndexT lidx, NodeType) {
110  if (event == ElementListener::ElementMigratedOut) {
111  sub_proxy.get()->migratedOutIndex(lidx);
112  } else if (event == ElementListener::ElementMigratedIn) {
113  sub_proxy.get()->migratedInIndex(lidx);
114  }
115  };
116  theCollection()->template registerElementListener<ColT>(proxy_bits, fn);
117 
118  // Count the number of local handles that should exist here. We can't use the
119  // cached value in the collection manager since this might be invoked in the
120  // an element's constructor before they are all created.
121  auto this_node = theContext()->getNode();
122  auto num_nodes = theContext()->getNumNodes();
123  auto min_node_mapped = theContext()->getNumNodes();
124  std::size_t local_count = 0;
125  auto const& map_fn = auto_registry::getHandlerMap(map_han);
126  range.foreach ([&](IndexT cur_idx) {
127  auto home_node = map_fn->dispatch(&cur_idx, &range, num_nodes);
128  if (home_node == this_node) {
129  local_count++;
130  }
131  min_node_mapped = std::min(home_node, min_node_mapped);
132  });
133 
134  // If LB is enabled then we need to register an afterLB listener
135 # if vt_check_enabled(lblite)
136  vt_debug_print(
137  verbose, rdma,
138  "CollectionHandle: registering LB listener\n"
139  );
140  thePhase()->registerHookCollective(phase::PhaseHook::EndPostMigration, [=]{
141  sub_proxy.get()->afterLB();
142  });
143 # endif
144 
145  // Set the number of expected handles for next time we run through this code
146  sub_proxy.get()->setCollectionExpected(local_count);
147 
148  if (local_count == 0) {
149  sub_proxy.get()->makeSubHandles();
150  }
151 
152  collection_to_manager_[proxy_bits][next_handle] = sub_proxy.getProxy();
153 
154  // If this node is the min node mapped, it is responsible for informing all
155  // other nodes about the RDMA handle creation, since it uses a collective
156  // ObjGroup and all nodes might not have a collection element mapped to
157  // them.
158  if (this_node == min_node_mapped and in_next_handle == no_rdma_handle) {
159  proxy_.template broadcast<
160  impl::InformRDMAMsg<ProxyT,IndexT>,
161  &Manager::informCollectionRDMA<T,E,ProxyT,ColT>
162  >(collection_proxy, next_handle, uniform_size, map_han, range);
163  }
164  } else {
165  auto sub_proxy_bits = iter->second;
166  sub_proxy = objgroup::proxy::Proxy<SubType>(sub_proxy_bits);
167  }
168 
169  if (in_next_handle == no_rdma_handle) {
170  // Add the handle
171  auto handle = sub_proxy.get()->addLocalIndex(idx, idx_size);
172 
173  // Check if all mapped handles are now created on this node
174  auto expected = sub_proxy.get()->getCollectionExpected();
175  auto current = sub_proxy.get()->getNumHandles();
176 
177  vt_debug_print(
178  verbose, rdma,
179  "CollectionHandle: expected={}, current={}\n", expected, current
180  );
181 
182  // All the local handles have been added
183  if (current == expected) {
184  sub_proxy.get()->makeSubHandles();
185  }
186 
187  return handle;
188  } else {
189  return Handle<T, E, IndexT>();
190  }
191 }
