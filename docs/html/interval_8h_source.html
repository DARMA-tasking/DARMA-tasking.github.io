114  /*
115  * Determine if a input interval is tangent to this interval within a
116  * specified gap.
117  *
118  * TangentLeft TangentRight
119  * |------i-----| lb_ ub_ |------i-----|
120  * |----this----|
121  * ^ ^
122  * gap gap
123  */
124 
125  PositionEnum tangent(IntervalType const& i, DomainT gap = 1) const {
126  if (lb_ == i.ub_ + gap) {
127  return PositionEnum::TangentLeft;
128  } else if (ub_ == i.lb_ - gap) {
129  return PositionEnum::TangentRight;
130  } else {
131  return PositionEnum::NotTangent;
132  }
133  }

135  void join(IntervalType const& i, PositionEnum pos) {
136  switch (pos) {
137  case PositionEnum::TangentLeft:
138  vtAssertExpr(lb_ == i.upper() + 1);
139  lb_ = i.lower();
140  break;
141  case PositionEnum::TangentRight:
142  vtAssertExpr(ub_ == i.lower() - 1);
143  ub_ = i.upper();
144  break;
145  default:
146  vtAssert(false, "Must have tangent position to join");
147  }
148  }

159  /*
160  * Determine if two intervals intersect:
161  *
162  *
163  *. [i1.lower(), ..., i1.upper()]
164  * ^ ^
165  * | |
166  * [i2.lower(), ..., i2.upper()]
167  *
168  * Two intervals intersect iff:
169  * (i2.lower() <= i1.upper()) && (i1.lower() <= i2.upper())
170  */
171  template <typename IntT, typename IntU>
172  static bool intersects(IntT&& i1, IntU&& i2) {
173  bool ret = i1.lower() <= i2.upper() and i2.lower() <= i1.upper();
174  vt_debug_print(
175  verbose, gen,
176  "Interval intersects: i1={}, i2={}, op:{}<={}={} and {}<={}={} => {}\n",
177  i1, i2,
178  i1.lower(), i2.upper(), i1.lower() <= i2.upper(),
179  i2.lower(), i1.upper(), i2.lower() <= i1.upper(),
180  ret
181  );

183  }
