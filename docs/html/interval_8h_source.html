105  enum struct PositionEnum : int {
106  TangentLeft = -1,
107  NotTangent = 0,
108  TangentRight = 1

111  /*
112  * Determine if a input interval is tangent to this interval within a
113  * specified gap.
114  *
115  * TangentLeft TangentRight
116  * |------i-----| lb_ ub_ |------i-----|
117  * |----this----|
118  * ^ ^
119  * gap gap
120  */
121 
122  PositionEnum tangent(IntervalType const& i, DomainT gap = 1) const {
123  if (lb_ == i.ub_ + gap) {
124  return PositionEnum::TangentLeft;
125  } else if (ub_ == i.lb_ - gap) {
126  return PositionEnum::TangentRight;
127  } else {
128  return PositionEnum::NotTangent;
129  }
130  }

132  void join(IntervalType const& i, PositionEnum pos) {
133  switch (pos) {
134  case PositionEnum::TangentLeft:
135  vtAssertExpr(lb_ == i.upper() + 1);
136  lb_ = i.lower();
137  break;
138  case PositionEnum::TangentRight:
139  vtAssertExpr(ub_ == i.lower() - 1);
140  ub_ = i.upper();
141  break;
142  default:
143  vtAssert(false, "Must have tangent position to join");
144  }
145  }

155 public:
156  /*
157  * Determine if two intervals intersect:
158  *
159  *
160  *. [i1.lower(), ..., i1.upper()]
161  * ^ ^
162  * | |
163  * [i2.lower(), ..., i2.upper()]
164  *
165  * Two intervals intersect iff:
166  * (i2.lower() <= i1.upper()) && (i1.lower() <= i2.upper())
167  */
168  template <typename IntT, typename IntU>
169  static bool intersects(IntT&& i1, IntU&& i2) {
170  bool ret = i1.lower() <= i2.upper() and i2.lower() <= i1.upper();
171  debug_print(
172  gen, node,
173  "Interval intersects: i1={}, i2={}, op:{}<={}={} and {}<={}={} => {}\n",
174  i1, i2,
175  i1.lower(), i2.upper(), i1.lower() <= i2.upper(),
176  i2.lower(), i1.upper(), i2.lower() <= i1.upper(),
177  ret
178  );
179  return ret;
180  }
