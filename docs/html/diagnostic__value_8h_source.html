129  friend DiagnosticValueWrapper<T> operator+(
130  DiagnosticValueWrapper<T> d1, DiagnosticValueWrapper<T> const& d2
131  ) {
132  int32_t N = d1.N_ + d2.N_;
133  double delta = d2.avg_ - d1.avg_;
134  double delta_sur_N = delta / static_cast<double>(N);
135  double delta2_sur_N2 = delta_sur_N * delta_sur_N;
136  int32_t n2 = d1.N_ * d1.N_;
137  int32_t n_c2 = d2.N_ * d2.N_;
138  int32_t prod_n = d1.N_ * d2.N_;
139  int32_t n_c = d2.N_;
140  int32_t n = d1.N_;
141  double M2 = d1.M2_;
142  double M2_c = d2.M2_;
143  double M3 = d1.M3_;
144  double M3_c = d2.M3_;
145 
146  d1.M4_ += d2.M4_
147  + prod_n * ( n2 - prod_n + n_c2 ) * delta * delta_sur_N * delta2_sur_N2
148  + 6. * ( n2 * M2_c + n_c2 * M2 ) * delta2_sur_N2
149  + 4. * ( n * M3_c - n_c * M3 ) * delta_sur_N;
150 
151  d1.M3_ += d2.M3_
152  + prod_n * ( n - n_c ) * delta * delta2_sur_N2
153  + 3. * ( n * M2_c - n_c * M2 ) * delta_sur_N;
154 
155  d1.M2_ += d2.M2_ + prod_n * delta * delta_sur_N;
156  d1.avg_ += n_c * delta_sur_N;
157  d1.N_ = N;
158  d1.min_ = std::min(d1.min_, d2.min_);
159  d1.max_ = std::max(d1.max_, d2.max_);
160  d1.sum_ += d2.sum_;
161 
162  d1.hist_.mergeIn(d2.hist_);
163  return d1;

307  DiagnosticSnapshotValues(int n_snapshots, T in_initial_value)
308  : initial_value_(in_initial_value)
309  {
310  for (int i = 0; i < n_snapshots; i++) {
311  snapshots_.emplace_back(DiagnosticValueWrapper<T>{in_initial_value});
312  }
313  }
314 
320  void update(T val, DiagnosticUpdate updater) {
321  for (auto& v : snapshots_) {
322  switch (updater) {
323  case DiagnosticUpdate::Replace:
324  v.update(val);
325  break;
326  case DiagnosticUpdate::Min:
327  v.update(std::min(v.getRawValue(), val));
328  break;
329  case DiagnosticUpdate::Max:
330  v.update(std::max(v.getRawValue(), val));
331  break;
332  case DiagnosticUpdate::Avg:
333  v.incrementN();
334  [[fallthrough]]; // no break to include update for Avg after increment
335  case DiagnosticUpdate::Sum:
336  v.update(v.getRawValue() + val);
337  break;
338  default:
339  vtAbort("Unknown DiagnosticUpdate---should be unreachable");
340  break;
341  }
342  }
343  }
