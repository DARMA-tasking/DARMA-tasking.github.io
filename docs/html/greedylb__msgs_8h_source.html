57 struct GreedyPayload : GreedyLBTypes {
58  GreedyPayload() = default;
59  GreedyPayload(ObjSampleType const& in_sample, LoadType const& in_profile)
60  : sample_(in_sample)
61  {
62  auto const& this_node = theContext()->getNode();
63  auto iter = load_profile_.find(this_node);
64  auto end_iter = load_profile_.end();
65  vtAssert(iter == end_iter, "Must not exist");
66  load_profile_.emplace(
67  std::piecewise_construct,
68  std::forward_as_tuple(this_node),
69  std::forward_as_tuple(in_profile)
70  );
71  }
72 
73  friend GreedyPayload operator+(GreedyPayload ld1, GreedyPayload const& ld2) {
74  auto& sample1 = ld1.sample_;
75  auto const& sample2 = ld2.sample_;
76  for (auto&& elm : sample2) {
77  auto const& bin = elm.first;
78  auto const& entries = elm.second;
79  for (auto&& entry : entries) {
80  sample1[bin].push_back(entry);
81  }
82  }
83  auto& load1 = ld1.load_profile_;
84  auto const& load2 = ld2.load_profile_;
85  for (auto&& elm : load2) {
86  auto const& proc = elm.first;
87  auto const& load = elm.second;
88  vtAssert(load1.find(proc) == load1.end(), "Must not exist");
89  load1[proc] = load;
90  }
91  return ld1;
92  }

105  LoadProfileType load_profile_;
106  ObjSampleType sample_;
