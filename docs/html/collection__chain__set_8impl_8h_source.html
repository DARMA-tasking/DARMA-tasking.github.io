52 template <typename Index>
53 template <typename ProxyT, typename IndexT>
54 CollectionChainSet<Index>::CollectionChainSet(
55  ProxyT proxy, ChainSetLayout layout
56 ) {
57  using ColT = typename ProxyT::CollectionType;
58  using ThisType = CollectionChainSet<Index>;
59  using ListenerType = vrt::collection::listener::ListenFnType<IndexT>;
60  using vrt::collection::listener::ElementEventEnum;
61 
62  static_assert(std::is_same<IndexT, Index>::value, "Must match index type");
63 
64  // Make this chain set an objgroup instance so we can send updates
65  auto p = theObjGroup()->makeCollective<CollectionChainSet<Index>>(
66  this, "CollectionChainSet"
67  );
68 
69  auto const this_node = theContext()->getNode();
70  auto const proxy_bits = proxy.getProxy();
71 
72  ListenerType l = [=](ElementEventEnum event, IndexT idx, NodeType home) {
73  switch (event) {
74  case ElementEventEnum::ElementCreated:
75  case ElementEventEnum::ElementMigratedIn:
76  if (layout == Local or (layout == Home and home == this_node)) {
77  if (chains_.find(idx) == chains_.end()) {
78  addIndex(idx);
79  } else {
80  vtAssert(
81  layout == Home and event == ElementEventEnum::ElementMigratedIn,
82  "Must be home layout and migrated in"
83  );
84  }
85  } else if (event != ElementEventEnum::ElementMigratedIn) {
86  vtAssert(layout == Home, "Must be a home layout");
87  p[home].template send<IdxMsg, &ThisType::addIndexHan>(idx);
88  }
89  break;
90  case ElementEventEnum::ElementDestroyed:
91  if (chains_.find(idx) != chains_.end()) {
92  removeIndex(idx);
93  } else {
94  vtAssert(layout == Home, "Must be a home layout");
95  p[home].template send<IdxMsg, &ThisType::removeIndexHan>(idx);
96  }
97  break;
98  case ElementEventEnum::ElementMigratedOut:
99  if (layout == Local) {
100  removeIndex(idx);
101  }
102  break;
103  }
104  };
105 
106  // Register the listener with the system
107  auto listener = theCollection()->template registerElementListener<ColT>(
108  proxy_bits, l
109  );
110 
111  // Any elements that currently exist must be added as they won't be triggered
112  // by the listener
113  auto const& local_set = theCollection()->getLocalIndices(proxy);
114  for (auto&& idx : local_set) {
115  auto const home = theCollection()->getMappedNode(proxy, idx);
116  if (layout == Local or (layout == Home and home == this_node)) {
117  addIndex(idx);
118  } else {
119  p[home].template send<IdxMsg, &ThisType::addIndexHan>(idx);
120  }
121  }
122 
123  deallocator_ = [=]{
124  theCollection()->template unregisterElementListener<ColT>(
125  proxy_bits, listener
126  );
127  };
128 }
129 
130 }} /* end namespace vt::messaging */
