54 template<typename SerializerT, typename T>
55 void deserializeQueueElems(SerializerT& s, std::queue<T>& q, typename std::queue<T>::size_type size) {
56  using Reconstructor =
57  dispatch::Reconstructor<typename dispatch::CleanType<T>::CleanT>;
58 
59  dispatch::Allocator<T> allocated;
60  for (typename std::queue<T>::size_type i = 0; i < size; ++i) {
61  auto* reconstructed = Reconstructor::construct(allocated.buf);
62  s | *reconstructed;
63  q.push(std::move(*reconstructed));
64  }
65 }

68 void serializeQueueElems(SerializerT& s, std::queue<T> q) {
