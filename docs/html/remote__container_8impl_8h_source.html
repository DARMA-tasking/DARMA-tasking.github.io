94 typename RemoteContainer<MsgT,TupleT>::template IsNotVoidType<MsgU>
95 RemoteContainer<MsgT,TupleT>::triggerDirect(CallbackT cb, MsgU* data) {
96  auto const& pid = BaseContainer<MsgT>::getPipe();
97  auto const& multi_callback = std::tuple_size<decltype(trigger_list_)>() > 0;
98  vt_debug_print(
99  terse, pipe,
100  "RemoteContainer: (typed) invoke trigger: pipe={:x}, multi={}, ptr={}\n",
101  pid, multi_callback, print_ptr(data)
102  );
103 
104  if (multi_callback) {
105  /*
106  * Make a copy of the message for each callback trigger because the callback
107  * trigger may send/bcast messages, thus a copy must be made
108  */
109  auto cur_msg = makeMessage<MsgT>(*data);
110  cb.trigger(cur_msg.get(),pid);
111  } else {
112  cb.trigger(data,pid);
113  }
114 }

116 template <typename MsgT, typename TupleT>
117 template <typename MsgU>
118 void RemoteContainer<MsgT,TupleT>::trigger(MsgU* data) {
119  auto const& pipe = BaseContainer<MsgT>::getPipe();
120  bool const& is_send_back = isSendBack();
121  vt_debug_print(
122  terse, pipe,
123  "RemoteContainer: pipe={:x}, send_back={}, size={}\n",
124  pipe, is_send_back, std::tuple_size<decltype(trigger_list_)>()
125  );
126  if (is_send_back) {
127  auto const& retpipe = this->getPipe();
128  triggerSendBack<MsgT>(retpipe,data);
129  } else {
130  triggerDirect(data);
131  }
132 }
