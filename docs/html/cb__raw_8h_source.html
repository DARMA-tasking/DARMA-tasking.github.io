219  template <typename SerializerT>
220  void serialize(SerializerT& s) {
221  s | active_;
222 
223  // serialize actual content of the union and account for leftovers and
224  // padding bytes when footprinting
225  auto union_size = sizeof(this->u_);
226  auto ser = [&](auto& cb){
227  s | cb;
228  s.addBytes(union_size - sizeof(cb));
229  };
230  s.skip(u_);
231  switch (active_) {
232  case CallbackEnum::AnonCB:
233  ser(u_.anon_cb_);
234  break;
235  case CallbackEnum::SendMsgCB:
236  ser(u_.send_msg_cb_);
237  break;
238  case CallbackEnum::SendColMsgCB:
239  ser(u_.send_col_msg_cb_);
240  break;
241  case CallbackEnum::BcastMsgCB:
242  ser(u_.bcast_msg_cb_);
243  break;
244  case CallbackEnum::BcastColMsgCB:
245  ser(u_.bcast_col_msg_cb_);
246  break;
247  case CallbackEnum::BcastColDirCB:
248  ser(u_.bcast_col_dir_cb_);
249  break;
250  case CallbackEnum::BcastObjGrpCB:
251  ser(u_.bcast_obj_cb_);
252  break;
253  case CallbackEnum::SendObjGrpCB:
254  ser(u_.send_obj_cb_);
255  break;
256  case CallbackEnum::NoCB:
257  // Serializing empty callback!
258  s.addBytes(sizeof(u_) - sizeof(active_));
259  break;
260  default:
261  vtAssert(0, "Should be unreachable");
262  break;
263  }
264  }
265

267  bool valid() const { return !null(); }
