95 template <typename EntityID>
96 void EntityLocationCoord<EntityID>::registerEntity(
97  EntityID const& id, NodeType const& home, LocMsgActionType msg_action,
98  bool const& migrated
99 ) {
100  auto const& this_node = theContext()->getNode();
101  auto reg_iter = local_registered_.find(id);
102 
103  vtAssert(
104  reg_iter == local_registered_.end(),
105  "EntityLocationCoord entity should not already be registered"
106  );
107 
108  vt_debug_print(
109  terse, location,
110  "EntityLocationCoord: registerEntity: inst={}, home={}, migrated={}, "
111  "id={}\n",
112  this_inst, home, migrated, id
113  );
114 
115  local_registered_.insert(id);
116 
117  recs_.insert(id, home, LocRecType{id, eLocState::Local, this_node});
118 
119  if (msg_action != nullptr) {
120  // vtAssert(
121  // local_registered_msg_han_.find(id) == local_registered_msg_han_.end(),
122  // "Entitiy should not exist in local registered msg handler"
123  // );
124  local_registered_msg_han_.emplace(
125  std::piecewise_construct,
126  std::forward_as_tuple(id),
127  std::forward_as_tuple(LocEntityMsg{id, msg_action})
128  );
129  }
130 
131  // trigger any pending actions upon registration
132  auto pending_lookup_iter = pending_lookups_.find(id);
133 
134  vt_debug_print(
135  normal, location,
136  "EntityLocationCoord: registerEntity: pending lookups size={}, this={}, "
137  "id={}\n",
138  pending_lookups_.size(), print_ptr(this), id
139  );
140 
141  if (pending_lookup_iter != pending_lookups_.end()) {
142  auto const& node = theContext()->getNode();
143  int action = 0;
144  for (auto&& pending_action : pending_lookup_iter->second) {
145  vt_debug_print(
146  verbose, location,
147  "EntityLocationCoord: registerEntity: running pending action {}\n",
148  action
149  );
150  action++;
151  pending_action(node);
152  }
153  pending_lookups_.erase(pending_lookup_iter);
154  }
155 
156  if (!migrated) {
157  /*
158  * This is the case where the entity is *not* migrated here but gets
159  * constructed in an alternative non-default location. Thus we need to
160  * inform the home so that messages can be forwarded.
161  */
162  vtAssert(home != uninitialized_destination, "Must have home node info");
163  if (home != this_node) {
164  vt_debug_print(
165  normal, location,
166  "EntityLocationCoord: registerEntity: updating id={}, home={}: "
167  "not migrated\n",
168  id, home
169  );
170 
171  auto const& ask_node = uninitialized_destination;
172  auto msg = makeMessage<LocMsgType>(
173  this_inst, id, no_location_event_id, ask_node, home
174  );
175  msg->setResolvedNode(this_node);
176  theMsg()->markAsLocationMessage(msg);
177  theMsg()->sendMsg<LocMsgType, updateLocation>(home, msg);
178  }
179  }
180 }

314 template <typename MessageT>
315 void EntityLocationCoord<EntityID>::routeMsgEager(
316  bool const is_serialized, EntityID const& id, NodeType const& home_node,
317  MsgSharedPtr<MessageT> msg
318 ) {
319  auto const& this_node = theContext()->getNode();
320  NodeType route_to_node = uninitialized_destination;
321 
322  auto reg_iter = local_registered_.find(id);
323  bool const found = reg_iter != local_registered_.end();
324 
325  vt_debug_print(
326  normal, location,
327  "EntityLocationCoord: routeMsgEager: found={}, home_node={}, "
328  "route_to_node={}, is_serialized={}, id={}\n",
329  found, home_node, route_to_node, is_serialized, id
330  );
331 
332  if (found) {
333  recs_.insert(id, home_node, LocRecType{id, eLocState::Local, this_node});
334  route_to_node = this_node;
335  } else {
336  bool const& rec_exists = recs_.exists(id);
337 
338  if (not rec_exists) {
339  if (home_node != this_node) {
340  route_to_node = home_node;
341  } else {
342  route_to_node = this_node;
343  }
344  } else {
345  auto const& rec = recs_.get(id);
346 
347  if (rec.isLocal()) {
348  route_to_node = this_node;
349  } else if (rec.isRemote()) {
350  route_to_node = rec.getRemoteNode();
351  }
352  }
353  }
354 
355  vtAssert(
356  route_to_node != uninitialized_destination,
357  "Node to route to must be set by this point"
358  );
359 
360  vt_debug_print(
361  normal, location,
362  "EntityLocationCoord: routeMsgEager: home_node={}, route_node={}, "
363  "is_serialized={}, id={}\n",
364  home_node, route_to_node, is_serialized, id
365  );
366 
367  return routeMsgNode<MessageT>(is_serialized,id,home_node,route_to_node,msg);

370 template <typename EntityID>
371 void EntityLocationCoord<EntityID>::getLocation(
372  EntityID const& id, NodeType const& home_node, NodeActionType const& action
373 ) {
374  auto const& this_node = theContext()->getNode();
375 
376  auto reg_iter = local_registered_.find(id);
377 
378  if (reg_iter != local_registered_.end()) {
379  vt_debug_print(
380  normal, location,
381  "EntityLocationCoord: getLocation: entity is local\n"
382  );
383 
384  recs_.insert(id, home_node, LocRecType{id, eLocState::Local, this_node});
385  action(this_node);
386  return;
387  } else {
388  bool const& rec_exists = recs_.exists(id);
389 
390  vt_debug_print(
391  normal, location,
392  "EntityLocationCoord: getLocation: home_node={}, rec_exists={}, "
393  "msg size={}\n",
394  home_node, print_bool(rec_exists), sizeof(LocMsgType)
395  );
396 
397  if (not rec_exists) {
398  if (home_node != this_node) {
399  auto const& event_id = fst_location_event_id++;
400  auto msg = makeMessage<LocMsgType>(
401  this_inst, id, event_id, this_node, home_node
402  );
403  theMsg()->markAsLocationMessage(msg);
404  theMsg()->sendMsg<LocMsgType, getLocationHandler>(home_node, msg);
405  // save a pending action when information about location arrives
406  pending_actions_.emplace(
407  std::piecewise_construct,
408  std::forward_as_tuple(event_id),
409  std::forward_as_tuple(PendingType{id, action})
410  );
411  } else {
412  // this is the home node and there is no record on this entity
413  insertPendingEntityAction(id, action);
414  }
415  } else {
416  auto const& rec = recs_.get(id);
417 
418  if (rec.isLocal()) {
419  vtAssert(0, "Should be registered if this is the case!");
420  action(this_node);
421  } else if (rec.isRemote()) {
422  vt_debug_print(
423  normal, location,
424  "EntityLocationCoord: getLocation: entity is remote\n"
425  );
426 
427  action(rec.getRemoteNode());
428  }
429  }
430  }

447

449  recs_.insert(id, home_node, LocRecType{id, eLocState::Remote, deliver_node});

451  auto ask_iter = loc_asks_.find(id);
452  if (ask_iter != loc_asks_.end()) {
453  for (auto&& ask_node : ask_iter->second) {
454  sendEagerUpdate(id, ask_node, home_node, deliver_node);
455  }
456  loc_asks_.erase(ask_iter);
457  }

481 template <typename EntityID>
482 template <typename MessageT>
483 void EntityLocationCoord<EntityID>::routeMsgNode(
484  bool const is_serialized, EntityID const& id, NodeType const& home_node,
485  NodeType const& to_node, MsgSharedPtr<MessageT> msg
486 ) {
487  auto const& this_node = theContext()->getNode();
488  auto const epoch = theMsg()->getEpochContextMsg(msg);
489 
490  vt_debug_print(
491  normal, location,
492  "EntityLocationCoord: routeMsgNode: to_node={}, this_node={}: inst={}, "
493  "is_serialized={}, home_node={}, id={}, ref={}, from={}, msg={}, epoch={:x}\n",
494  to_node, this_node, this_inst, is_serialized, home_node, id,
495  envelopeGetRef(msg->env), msg->getLocFromNode(), print_ptr(msg.get()),
496  epoch
497  );
498 
499  theMsg()->markAsLocationMessage(msg);
500 
501  if (to_node != this_node) {
502  // Get the current ask node, which is the from node for the first hop
503  auto ask_node = msg->getAskNode();
504  if (ask_node != uninitialized_destination) {
505  // Insert into the ask list for a later update when information is known
506  loc_asks_[id].insert(ask_node);
507  }
508 
509  // Update the new asking node, as this node is will be the next to ask
510  msg->setAskNode(this_node);
511 
512  // set the instance on the message to deliver to the correct manager
513  msg->setLocInst(this_inst);
514 
515  // send to the node discovered by the location manager
516  theMsg()->sendMsg<MessageT, routedHandler>(to_node, msg);
517  } else {
518  vt_debug_print(
519  normal, location,
520  "EntityLocationCoord: routeMsgNode: to_node={}, this_node={}: "
521  "home_node={}, ref={}, from={}, epoch={:x}: apply here\n",
522  to_node, this_node, home_node, envelopeGetRef(msg->env),
523  msg->getLocFromNode(), epoch
524  );
525 
526  theTerm()->produce(epoch);
527 
528  auto trigger_msg_handler_action = [=](EntityID const& hid) {
529  bool const& has_handler = msg->hasHandler();
530  auto const& from = msg->getLocFromNode();
531  if (has_handler) {
532  auto const handler = msg->getHandler();
533 
534  vt_debug_print(
535  verbose, location,
536  "EntityLocationCoord: apply direct handler action: "
537  "id={}, from={}, handler={}, ref={}\n",
538  hid, from, handler, envelopeGetRef(msg->env)
539  );
540 
541  runnable::makeRunnable(msg, true, handler, from)
542  .withTDEpochFromMsg()
543  .run();
544  } else {
545  auto reg_han_iter = local_registered_msg_han_.find(hid);
546  vtAssert(
547  reg_han_iter != local_registered_msg_han_.end(),
548  "Message handler must exist for location manager routed msg"
549  );
550  vt_debug_print(
551  verbose, location,
552  "EntityLocationCoord: no direct handler: id={}\n", hid
553  );
554  reg_han_iter->second.applyRegisteredActionMsg(msg.get());
555  }
556 
557  auto ask_node = msg->getAskNode();
558 
559  if (ask_node != uninitialized_destination) {
560  auto delivered_node = theContext()->getNode();
561  sendEagerUpdate(hid, ask_node, home_node, delivered_node);
562  }
563  };
564 
565  auto reg_iter = local_registered_.find(id);
566 
567  vt_debug_print(
568  verbose, location,
569  "EntityLocationCoord: routeMsgNode: size={}\n",
570  local_registered_.size()
571  );
572 
573  if (reg_iter != local_registered_.end()) {
574  vt_debug_print(
575  normal, location,
576  "EntityLocationCoord: routeMsgNode: epoch={:x} running actions\n",
577  epoch
578  );
579 
580  theMsg()->pushEpoch(epoch);
581  trigger_msg_handler_action(id);
582  theMsg()->popEpoch(epoch);
583  theTerm()->consume(epoch);
584  } else {
585  vt_debug_print(
586  normal, location,
587  "EntityLocationCoord: routeMsgNode: ref={}, buffering\n",
588  envelopeGetRef(msg->env)
589  );
590 
591  EntityID id_ = id;
592  // buffer the message here, the entity will be registered in the future
593  insertPendingEntityAction(id_, [=](NodeType resolved) {
594  auto const& my_node = theContext()->getNode();
595 
596  vt_debug_print(
597  normal, location,
598  "EntityLocationCoord: routeMsgNode: trigger action: resolved={}, "
599  "this_node={}, id={}, ref={}, epoch={:x}\n",
600  resolved, my_node, id_, envelopeGetRef(msg->env), epoch
601  );
602 
603  theMsg()->pushEpoch(epoch);
604  if (resolved == my_node) {
605  trigger_msg_handler_action(id_);
606  } else {
607  /*
608  * Recurse with the new updated node information. This occurs
609  * typically when an non-migrated registration occurs off the home
610  * node and messages are buffered, awaiting forwarding information.
611  */
612  routeMsgNode<MessageT>(is_serialized,id_,home_node,resolved,msg);
613  }
614  theMsg()->popEpoch(epoch);
615  theTerm()->consume(epoch);
616  });
617  }
618  }
619 }
