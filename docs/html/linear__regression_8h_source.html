60 struct LinearRegression {
61 
68  LinearRegression(std::vector<double> const& in_x, std::vector<double> const& in_y)
69  : x_(in_x),
70  y_(in_y)
71  { }
72 
76  void compute() {
77  vtAssert(x_.size() == y_.size(), "Sizes must be the same");
78  vtAssert(x_.size() != 0, "Sizes must not be zero");
79  auto const n = x_.size();
80  auto const sum_x = std::accumulate(x_.begin(), x_.end(), 0.0);
81  auto const sum_y = std::accumulate(y_.begin(), y_.end(), 0.0);
82  auto const p_xx = std::inner_product(x_.begin(), x_.end(), x_.begin(), 0.0);
83  auto const p_xy = std::inner_product(x_.begin(), x_.end(), y_.begin(), 0.0);
84  // numerator: sum over i of (x[i] - X_mean) * (y[i] - Y_mean)
85  // denominator: sum over i of (x[i] - X_mean) * (x[i] - X_mean)
86  auto const numerator = p_xy * n - sum_x * sum_y;
87  auto const denominator = p_xx * n - sum_x * sum_x;
88 
89  vtAssert(denominator != 0, "Denominator must not be zero");
90 
91  slope_ = numerator / denominator;
92  intercept_ = (sum_y - slope_ * sum_x) / n;
93  computed_ = true;
94  }
95 
101  double getSlope() const { return slope_; }
102 
108  double getIntercept() const { return intercept_; }
109 
117  double predict(double in_x) {
118  if (not computed_) {
119  compute();
120  }
121  return intercept_ + slope_ * in_x;
122  }
123 
124 private:
125  std::vector<double> const& x_;
126  std::vector<double> const& y_;
127  double slope_ = 0.;
128  double intercept_ = 0.;
129  bool computed_ = false;
130 };
