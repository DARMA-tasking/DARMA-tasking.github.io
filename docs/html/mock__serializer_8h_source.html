96 template <typename T, typename BufferT>
97 static inline std::unique_ptr<BufferT> serializeByte(
98  T* target, bool hasSize = false, SizeType inSize = 0
99 ) {
100  using PtrType = std::unique_ptr<BufferT>;
101 
102  SizeType size = 0;
103  PtrType buf = nullptr;
104 
105  auto tptr = reinterpret_cast<SerialByteType*>(target);
106 
107  if (hasSize) {
108  size = inSize + sizeof(SizeType);
109  buf = std::make_unique<BufferT>(size);
110  SerialByteType* cur = buf->getBuffer();
111  std::memcpy(cur, reinterpret_cast<SerialByteType*>(&inSize), sizeof(SizeType));
112  std::memcpy(cur + sizeof(SizeType), tptr, inSize);
113  } else {
114  size = sizeof(T);
115  buf = std::make_unique<BufferT>(size);
116  std::memcpy(buf->getBuffer(), tptr, size);
117  }
118 
119  return buf;
120 }

239  SizeType* buf_as_size = reinterpret_cast<SizeType*>(buf);
240  SizeType vec_bytes_size = *buf_as_size;

257 template <typename T>
258 SerializedReturnType serialize(T& target, BufferCallbackType fn) {
259  auto buf = ByteCopyableTraits<ByteSerializedBuffer, T>::apply(&target, 1);
260  if (fn != nullptr) {
261  auto user_buf = fn(buf->getSize());
262  std::memcpy(user_buf, buf->getBuffer(), buf->getSize());
263  }
264  std::unique_ptr<SerializedInfo> base_ptr(
265  static_cast<SerializedInfo*>(buf.release())
266  );
267  return base_ptr;
268 }
