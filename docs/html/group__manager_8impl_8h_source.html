107 template <typename T>
108 /*static*/ void GroupManagerT<T>::triggerWaitingContinuations(
109  RemoteOperationIDType const op
110 ) {
111  auto action_iter = continuation_actions_t_.find(op);
112  auto const found = action_iter != continuation_actions_t_.end();
113 
114  if (found) {
115  auto iter = waiting_cont_.find(op);
116  if (iter != waiting_cont_.end()) {
117  for (auto&& elm : iter->second) {
118  for (auto&& action : action_iter->second) {
119  action(elm);
120  }
121  }
122  waiting_cont_.clear();
123  }
124  }

132  auto const& found = iter != continuation_actions_t_.end();
133 
134  vt_debug_print(

138  );
139 
140  if (found) {
141  for (auto&& elm : iter->second) {
142  elm(t);
143  }
144  } else {
145  auto iter_wait = waiting_cont_.find(op);
146  if (iter_wait == waiting_cont_.end()) {
147  waiting_cont_.emplace(

151  );
152  } else {
