99 template <typename MsgU>
100 void msgSetPriorityImpl(
101  [[maybe_unused]] MsgU new_msg,
102  [[maybe_unused]] PriorityType new_priority,
103  [[maybe_unused]] PriorityType old_priority,
104  [[maybe_unused]] PriorityLevelType level
105 ) {
106 # if vt_check_enabled(priorities)
107 
108  bool const is_breadth_first = new_priority == sched::Priority::BreadthFirst;
109  vt_debug_print(
110  verbose, gen,
111  "msgSetPriorityImpl: new_priority={:x}, old_priority={:x}, level={}, "
112  "is_breadth_first={}\n",
113  new_priority, old_priority, level, is_breadth_first
114  );
115  for (PriorityType l = level + 1; l < sched::priority_num_levels; l++) {
116  if (is_breadth_first) {
117  sched::PriorityManip::setPriority(old_priority, l, sched::priority_all_set);
118  } else {
119  sched::PriorityManip::setPriority(old_priority, l, 0);
120  }
121  }
122  sched::PriorityManip::setPriority(old_priority, level, new_priority);
123  envelopeSetPriorityLevel(new_msg->env, level);
124  envelopeSetPriority(new_msg->env, old_priority);
125 # endif
