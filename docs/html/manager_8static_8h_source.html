54 template <typename MsgT>
55 void send(MsgSharedPtr<MsgT> msg, HandlerType han, NodeType dest_node) {
56  auto const num_nodes = theContext()->getNumNodes();
57  auto const this_node = theContext()->getNode();
58  vtAssert(dest_node < num_nodes, "Invalid node (must be < num_nodes)");
59  if (dest_node != this_node) {
60  theMsg()->sendMsg<MsgT>(dest_node, han,msg, no_tag);
61  } else {
62  // Get the current epoch for the message
63  auto const cur_epoch = theMsg()->setupEpochMsg(msg);
64  // Schedule the work of dispatching the message handler for later
65  scheduleMsg(msg.template toVirtual<ShortMessage>(),han,cur_epoch);
66  }
67 }

69 template <typename MsgT>
70 void broadcast(MsgSharedPtr<MsgT> msg, HandlerType han) {
71  // Get the current epoch for the message
72  auto const cur_epoch = theMsg()->setupEpochMsg(msg);
73  // Broadcast the message
74  auto msg_hold = promoteMsg(msg.get()); // for scheduling
75  theMsg()->broadcastMsg<MsgT>(han, msg, no_tag);
76  // Schedule delivery on this node for the objgroup
77  scheduleMsg(msg_hold.template toVirtual<ShortMessage>(), han, cur_epoch);
78 }
