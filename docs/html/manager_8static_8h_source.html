56 template <typename MsgT>
57 messaging::PendingSend send(MsgSharedPtr<MsgT> msg, HandlerType han, NodeType dest_node) {
58  auto const num_nodes = theContext()->getNumNodes();
59  auto const this_node = theContext()->getNode();
60  vtAssert(dest_node < num_nodes, "Invalid node (must be < num_nodes)");
61  if (dest_node != this_node) {
62  return theMsg()->sendMsg<MsgT>(dest_node, han,msg, no_tag);
63  } else {
64  theMsg()->setupEpochMsg(msg);
65  envelopeSetHandler(msg->env, han);
66  return messaging::PendingSend{msg, [](MsgSharedPtr<BaseMsgType>& inner_msg){
67  dispatchObjGroup(
68  inner_msg.template to<ShortMessage>(),
69  envelopeGetHandler(inner_msg->env),
70  theContext()->getNode(),
71  nullptr
72  );
73  }};
74  }

125 template <typename MsgT>
126 void dispatch(
127  MsgSharedPtr<MsgT> msg, HandlerType han, NodeType from_node,
128  ActionType cont
129 ) {
130  // Extract the control-bit sequence from the handler
131  auto const ctrl = HandlerManager::getHandlerControl(han);
132  vt_debug_print(
133  verbose, objgroup,
134  "dispatch: ctrl={:x}, han={:x}\n", ctrl, han
135  );
136  auto const node = 0;
137  auto const proxy = proxy::ObjGroupProxy::create(ctrl, node, true);
138  auto& objs = getObjs();
139  auto obj_iter = objs.find(proxy);
140  vt_debug_print(
141  normal, objgroup,
142  "dispatch: try ctrl={:x}, han={:x}, has dispatch={}\n",
143  ctrl, han, obj_iter != objs.end()
144  );
145  if (obj_iter == objs.end()) {
146  auto const epoch = envelopeGetEpoch(msg->env);
147  if (epoch != no_epoch) {
148  theTerm()->produce(epoch);
149  }
150  auto& pending = getPending();
151  pending[proxy].emplace_back([=]{
152  auto& objs2 = getObjs();
153  auto obj_iter2 = objs2.find(proxy);
154  vtAssert(obj_iter2 != objs2.end(), "Obj must exist");
155  detail::dispatchImpl(msg, han, from_node, cont, obj_iter2->second->getPtr());
156  if (epoch != no_epoch) {
157  theTerm()->consume(epoch);
158  }
159  });
160  } else {
161  detail::dispatchImpl(msg, han, from_node, cont, obj_iter->second->getPtr());
162  }
163 }

165 } /* end namespace detail */
