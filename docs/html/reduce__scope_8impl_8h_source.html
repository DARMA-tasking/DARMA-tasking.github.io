58 inline std::string stringizeStamp(ReduceStamp const& stamp) {

61  } else if (stamp.is<TagPair>()) {

91 template <typename T>
92 template <typename U>
93 T& ReduceScopeHolder<T>::getOnDemand(U&& scope) {
94  auto iter = scopes_.find(scope);
95  if (iter == scopes_.end()) {
96  vtAssert(
97  not scope.get().template is<StrongGroup>(),
98  "Group reducers cannot be on-demand created -- needs spanning tree"
99  );
100 
101  iter = scopes_.emplace(
102  std::piecewise_construct,
103  std::forward_as_tuple(scope),
104  std::forward_as_tuple(default_creator_(scope))
105  ).first;
106  }
107 
108  vtAssert(iter != scopes_.end(), "Scope must exist");
109 
110  return iter->second;
111 }
