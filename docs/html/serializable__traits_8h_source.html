98  // Regular serialize detection
99  template <typename U>
100  using serialize_t = decltype(
101  std::declval<U>().serialize(std::declval<Serializer&>())
102  );
103  using has_serialize = detection::is_detected<serialize_t, T>;
104 
105  template <typename U>
106  using nonintrustive_serialize_t = decltype(serialize(
107  std::declval<Serializer&>(),
108  std::declval<U&>()
109  ));
110  using has_nonintrustive_serialize =
111  detection::is_detected<nonintrustive_serialize_t, T>;
112 
113  // serialize{Parent,This} detection
114  template <typename U>
115  using serializeParent_t =
116  decltype(std::declval<U>().serializeParent(std::declval<Serializer&>()));
117  using has_serializeParent = detection::is_detected<serializeParent_t, T>;
118 
119  template <typename U>
120  using serializeThis_t =
121  decltype(std::declval<U>().serializeThis(std::declval<Serializer&>()));
122  using has_serializeThis = detection::is_detected<serializeThis_t, T>;
123 
124  // This defines what it means to have an intrusive serialize
125  static constexpr auto const has_serialize_instrusive = has_serialize::value;
126 
127  // This defines what it means to have a non-intrusive serialize
128  static constexpr auto const has_serialize_noninstrusive =
129  has_nonintrustive_serialize::value;
130 
131  // This defines what it means to have some type of serialize function
132  static constexpr auto const has_serialize_function =
133  has_serialize_instrusive or has_serialize_noninstrusive;
134 
135  // This defines what it means to have parent serializability
136  static constexpr auto const has_parent_serialize = has_serializeParent::value;
137 
138  // This defines what it means to have "this" serializability
139  static constexpr auto const has_this_serialize = has_serializeThis::value;
140 
141  // This defines what it means to have both parent and "this" serializability
142  static constexpr auto const has_split_serialize =
143  has_parent_serialize and has_this_serialize;
144 
149  using has_byteCopyTraitTrue = isByteCopyable<T>;
150

152  using has_isArith = std::is_arithmetic<U>;
153 
154  // This defines what it means to be serializable without a serialize method

157 
162  template <typename U>

165

174  template <typename U>

178  std::declval<void*>()
179  )
180  );
181  using has_nonintrusive_reconstruct =
182  detection::is_detected<nonintrustive_reconstruct_t, T>;
183 
184  // This defines what it means to be default constructible
185  static constexpr auto const is_default_constructible =
186  has_default_constructor::value;
187 
188  // If the class has tagged constructor especially for constructing
189  static constexpr auto const is_tagged_constructible =

191

197  has_nonintrusive_reconstruct::value /*and not has_reconstruct::value*/;

201  has_nonintrusive_reconstruct::value or has_reconstruct::value;
202

211  static constexpr auto const is_serializable =
