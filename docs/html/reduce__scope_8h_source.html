85  std::string str() const {
86  if (l0_.is<StrongObjGroup>()) {
87  return fmt::format("objgroup[{:x}]", l0_.get<StrongObjGroup>().get());
88  } else if (l0_.is<StrongVrtProxy>()) {
89  return fmt::format("vrtproxy[{:x}]", l0_.get<StrongVrtProxy>().get());
90  } else if (l0_.is<StrongGroup>()) {
91  return fmt::format("group[{:x}]", l0_.get<StrongGroup>().get());
92  } else if (l0_.is<StrongCom>()) {
93  return fmt::format("component[{}]", l0_.get<StrongCom>().get());
94  } else if (l0_.is<StrongUserID>()) {
95  return fmt::format("userID[{}]", l0_.get<StrongUserID>().get());
96  } else {
97  return "<unknown-type>";
98  }
99  }
