64 template <typename IndexT>
65 void Holder<IndexT>::insert(IndexT const& idx, InnerHolder&& inner) {
66  vtAssert(!is_destroyed_, "Must not be destroyed to insert a new element");
67  vtAssert(!exists(idx), "Should not exist to insert element");
68 
69  auto const& lookup = idx;
70  auto& container = vc_container_;
71 
72  auto iter = container.find(lookup);
73  if (iter != container.end()) {
74  vtAssert(iter->second.erased_, "Must be in erased state");
75  num_erased_not_removed_--;
76  container.erase(iter);
77  }
78 
79  /*
80  * This assertion no longer valid due to delayed erasure. In fact, the inner
81  * holder VC pointer may be nullptr but set to erased. The move should deal
82  * with this problem.
83  *
84  * auto iter = container.find(lookup);
85  * vtAssert(
86  * iter == container.end(),
87  * "Entry must not exist in holder when inserting"
88  * );
89  */
90  container.emplace(
91  std::piecewise_construct,
92  std::forward_as_tuple(lookup),
93  std::forward_as_tuple(std::move(inner))
94  );
95 }

120  auto owned_ptr = std::move(iter->second.vc_ptr_);
121  vtAssert(iter->second.erased_ == false, "Must not be erased already");
122  iter->second.erased_ = true;

153 template <typename IndexT>
154 void Holder<IndexT>::foreach(FuncApplyType fn) {
155  static uint64_t num_reentrant = 0;
156 
157  num_reentrant++;
158  auto& container = vc_container_;
159  for (auto& elm : container) {
160  if (!elm.second.erased_) {
161  auto const& idx = elm.first;

164  fn(idx, col_ptr);
