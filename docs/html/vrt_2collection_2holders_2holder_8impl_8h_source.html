65 template <typename ColT, typename IndexT>
66 void Holder<ColT, IndexT>::insert(IndexT const& idx, InnerHolder&& inner) {
67  vtAssert(!is_destroyed_, "Must not be destroyed to insert a new element");
68  vtAssert(!exists(idx), "Should not exist to insert element");
69 
70  auto const& lookup = idx;
71  auto& container = vc_container_;
72 
73  auto iter = container.find(lookup);
74  if (iter != container.end()) {
75  vtAssert(iter->second.erased_, "Must be in erased state");
76  num_erased_not_removed_--;
77  container.erase(iter);
78  }
79 
80  /*
81  * This assertion no longer valid due to delayed erasure. In fact, the inner
82  * holder VC pointer may be nullptr but set to erased. The move should deal
83  * with this problem.
84  *
85  * auto iter = container.find(lookup);
86  * vtAssert(
87  * iter == container.end(),
88  * "Entry must not exist in holder when inserting"
89  * );
90  */
91  container.emplace(
92  std::piecewise_construct,
93  std::forward_as_tuple(lookup),
94  std::forward_as_tuple(std::move(inner))
95  );
96 }

154 template <typename ColT, typename IndexT>
155 bool Holder<ColT, IndexT>::foreach(FuncApplyType fn) {
156  static uint64_t num_reentrant = 0;
157 
158  num_reentrant++;
159  auto& container = vc_container_;
160  for (auto& elm : container) {
161  if (!elm.second.erased_) {
162  auto const& idx = elm.first;
163  auto const& holder = elm.second;
164  auto const col_ptr = holder.getCollection();
165  fn(idx,col_ptr);
166  }
167  }
168  num_reentrant--;
169 
170  if (num_reentrant == 0) {
171  cleanupExists();
172  }
173  return true;
174 }
