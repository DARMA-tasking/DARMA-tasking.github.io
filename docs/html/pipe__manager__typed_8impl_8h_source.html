204 interface::CallbackDirectSendMulti<
205  MsgT,
206  typename RepeatNImpl<sizeof...(f),callback::CallbackSend<MsgT>>::ResultType
207 >
208 PipeManagerTyped::makeCallbackMultiSendTyped(
209  bool const is_persist, NodeType const& send_to_node
210 ) {
211  using CBSendT = callback::CallbackSend<MsgT>;
212  using ConsT = std::tuple<NodeType>;
213  using TupleConsT = typename RepeatNImpl<sizeof...(f),ConsT>::ResultType;
214  using ConstructMeta = ConstructCallbacks<CBSendT,TupleConsT,MsgT,f...>;
215  using TupleCBType = typename ConstructMeta::ResultType;
216 
217  auto const& new_pipe_id = makePipeID(is_persist,false);
218  std::array<NodeType,sizeof...(f)> send_node_array;
219  send_node_array.fill(send_to_node);
220  auto const cons = TupleConsT{send_node_array};
221  auto const tuple = ConstructMeta::make(cons);
222  auto rcm = interface::CallbackDirectSendMulti<MsgT,TupleCBType>(
223  interface::CallbackDirectSendMultiTag,new_pipe_id,tuple
224  );
225  return rcm;
226 }
