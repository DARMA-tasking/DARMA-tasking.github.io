101 template <typename MessageT, ActiveTypedFnType<MessageT>* f>
102 template <typename T>
103 /*static*/ auto SeqMatcher<MessageT, f>::getMatchingAnyTagged(
104  SeqStateTaggedContType<T>& tagged_lst, TagType const& tag
105 ) {
106  vtAssert(hasMatchingAnyTagged(tagged_lst, tag), "Must have matching elem");
107 
108  auto iter = tagged_lst.find(tag);
109  auto elm = getFirstElem(iter->second);
110  if (iter->second.size() == 0) {
111  tagged_lst.erase(iter);
112  }
113  return elm;
114 }
