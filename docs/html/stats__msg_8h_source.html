89  auto delta2_sur_N2 = delta_sur_N * delta_sur_N;
90  int32_t n2 = a1.N_ * a1.N_;
91  int32_t n_c2 = a2.N_ * a2.N_;
92  int32_t prod_n = a1.N_ * a2.N_;

95  auto M2 = a1.M2_;

99 
100  a1.M4_ += a2.M4_
101  + prod_n * ( n2 - prod_n + n_c2 ) * delta * delta_sur_N * delta2_sur_N2
102  + 6. * ( n2 * M2_c + n_c2 * M2 ) * delta2_sur_N2
103  + 4. * ( n * M3_c - n_c * M3 ) * delta_sur_N;
104

107  + 3. * ( n * M2_c - n_c * M2 ) * delta_sur_N;
