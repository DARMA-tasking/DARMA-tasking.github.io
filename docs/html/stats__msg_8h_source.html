72  vt_debug_print(lb, node, "operator+: a1.N_={}, a2.N_={}\n", a1.N_, a2.N_);
73  vt_debug_print(lb, node, "operator+: a1.avg_={}, a2.avg_={}\n", a1.avg_, a2.avg_);
74 
75  int32_t N = a1.N_ + a2.N_;
76  double delta = a2.avg_ - a1.avg_;
77  double delta_sur_N = delta / static_cast<double>(N);
78  double delta2_sur_N2 = delta_sur_N * delta_sur_N;
79  int32_t n2 = a1.N_ * a1.N_;
80  int32_t n_c2 = a2.N_ * a2.N_;
81  int32_t prod_n = a1.N_ * a2.N_;
82  int32_t n_c = a2.N_;
83  int32_t n = a1.N_;
84  double M2 = a1.M2_;
85  double M2_c = a2.M2_;
86  double M3 = a1.M3_;
87  double M3_c = a2.M3_;
88 
89  a1.M4_ += a2.M4_
90  + prod_n * ( n2 - prod_n + n_c2 ) * delta * delta_sur_N * delta2_sur_N2
91  + 6. * ( n2 * M2_c + n_c2 * M2 ) * delta2_sur_N2
92  + 4. * ( n * M3_c - n_c * M3 ) * delta_sur_N;
93 
94  a1.M3_ += a2.M3_
95  + prod_n * ( n - n_c ) * delta * delta2_sur_N2
96  + 3. * ( n * M2_c - n_c * M2 ) * delta_sur_N;
97 
98  a1.M2_ += a2.M2_ + prod_n * delta * delta_sur_N;
99  a1.avg_ += n_c * delta_sur_N;
100  a1.N_ = N;
101  a1.min_ = std::min(a1.min_, a2.min_);
102  a1.max_ = std::max(a1.max_, a2.max_);
103  a1.sum_ += a2.sum_;
104  a1.P_ += a2.P_;
105 
106  return a1;
107  }
