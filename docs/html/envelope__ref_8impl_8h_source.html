52 template <typename Env>
53 inline void envelopeRef(Env& env) {
54  Envelope* envp = reinterpret_cast<Envelope*>(&env);
55 
56  vtAssert(
57  envp->ref != not_shared_message,
58  "'Not shared message' encountered on message ref-increment. "
59  "This is can be caused by explicitly using 'new Message(..)' "
60  "instead of a MsgPtr/makeMessage/makeSharedMessage construct."
61  );
62  vtAssertInfo(
63  envp->ref >= 0 and envp->ref < 100,
64  "Bad ref-count on message ref-increment. "
65  "Message ref-count must never be negative and cannnot exceed limit (100).",
66  static_cast<RefType>(envp->ref)
67  );
68 
69  envp->ref++;
70 }
