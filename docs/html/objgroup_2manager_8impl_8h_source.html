127 template <typename ObjT>
128 void ObjGroupManager::destroyCollective(ProxyType<ObjT> proxy) {
129  auto const proxy_bits = proxy.getProxy();
130  auto derived_iter = derived_to_bases_.find(proxy_bits);
131  vt_debug_print(
132  objgroup, node,
133  "destroyCollective: proxy={:x}, num bases={}\n", proxy_bits,
134  derived_iter != derived_to_bases_.end() ? derived_iter->second.size() : 0
135  );
136  if (derived_iter != derived_to_bases_.end()) {
137  auto base_set = derived_iter->second;
138  for (auto&& base_proxy : base_set) {
139  auto iter = dispatch_.find(base_proxy);
140  if (iter != dispatch_.end()) {
141  dispatch_.erase(iter);
142  }
143  }
144  derived_to_bases_.erase(derived_iter);
145  }
146  auto iter = dispatch_.find(proxy_bits);
147  if (iter != dispatch_.end()) {
148  auto ptr = iter->second->objPtr();
149  auto obj_iter = obj_to_proxy_.find(ptr);
150  if (obj_iter != obj_to_proxy_.end()) {
151  obj_to_proxy_.erase(obj_iter);
152  }
153  dispatch_.erase(iter);
154  }
155  auto obj_iter = objs_.find(proxy_bits);
156  if (obj_iter != objs_.end()) {
157  objs_.erase(obj_iter);
158  }
159 }

161 template <typename ObjT>
162 void ObjGroupManager::regObjProxy(ObjT* obj, ObjGroupProxyType proxy) {
163  auto iter = dispatch_.find(proxy);
164  vtAssertExpr(iter == dispatch_.end());
165  vt_debug_print(
166  objgroup, node,
167  "regObjProxy: obj={}, proxy={:x}\n",
168  print_ptr(obj), proxy
169  );
170  DispatchBasePtrType b = std::make_unique<dispatch::Dispatch<ObjT>>(proxy,obj);
171  dispatch_.emplace(
172  std::piecewise_construct,
173  std::forward_as_tuple(proxy),
174  std::forward_as_tuple(std::move(b))
175  );
176  auto pending_iter = pending_.find(proxy);
177  if (pending_iter != pending_.end()) {
178  for (auto&& msg : pending_iter->second) {
179  theSched()->enqueue([msg]{
180  auto const handler = envelopeGetHandler(msg->env);
181  auto const epoch = envelopeGetEpoch(msg->env);
182  theObjGroup()->dispatch(msg,handler);
183  if (epoch != no_epoch) {
184  theTerm()->consume(epoch);
185  }
186  });
187  }
188  pending_.erase(pending_iter);
189  }
190 }

367 template <typename ObjT, typename... Args>
368 void ObjGroupManager::update(ProxyElmType<ObjT> proxy, Args&&... args) {
369  auto const this_node = theContext()->getNode();
370  vtAssert(this_node == proxy.getNode(), "You can only update a local obj");
371  auto const proxy_bits = proxy.getProxy();
372  auto iter = objs_.find(proxy_bits);
373  vtAssert(iter != objs_.end(), "Obj must exist on this node");
374  HolderBaseType* holder = iter->second.get();
375  auto obj_holder = static_cast<holder::Holder<ObjT>*>(holder);
376 
377  auto const cur_obj_ptr = obj_holder->get();
378  auto map_iter = obj_to_proxy_.find(cur_obj_ptr);
379  vtAssert(map_iter != obj_to_proxy_.end(), "Object pointer must exist in map");
380  obj_to_proxy_.erase(map_iter);
381 
382  obj_holder->reset(std::forward<Args>(args)...);
383 
384  auto const new_obj_ptr = obj_holder->get();
385  obj_to_proxy_[new_obj_ptr] = proxy_bits;
386 }
