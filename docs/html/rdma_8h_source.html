784  template <
785  RDMAManager::RDMA_TypeType rdma_type,
786  typename MsgT, typename FunctionT, ActiveTypedRDMAGetFnType<MsgT>* f
787  >
788  RDMA_HandlerType associateRdmaGetFunction(
789  MsgT* msg, RDMA_HandleType const& han, FunctionT const& fn,
790  bool const& any_tag, TagType const& tag
791  ) {
792  auto const& this_node = theContext()->getNode();
793  auto const handler_node = RDMA_HandleManagerType::getRdmaNode(han);
794  auto const& is_collective = RDMA_HandleManagerType::isCollective(han);
795 
796  vtAssert(
797  (is_collective or handler_node == this_node)
798  , "Handle must be local to this node"
799  );
800 
801  auto holder_iter = holder_.find(han);
802  vtAssert(
803  holder_iter != holder_.end(), "Holder for handler must exist here"
804  );
805 
806  auto& state = holder_iter->second;
807 
808  if (rdma_type == RDMAManager::RDMA_TypeType::Get) {
809  return state.setRDMAGetFn<MsgT,FunctionT,f>(msg, fn, any_tag, tag);
810  } else {
811  vtAssert(0, "Should be unreachable");
812  return no_rdma_handle;
813  }
814  }
