123  template <typename U>
124  static std::string getValueWithUnits(
125  U val, DiagnosticUnit unit, std::string default_spec, bool align
126  ) {
127  using util::memory::MemoryUnitEnum;
128  using util::memory::getMemoryUnitName;
129 
130  switch (unit) {
131  case DiagnosticUnit::Bytes: {
132  // Start with the largest unit, testing if its appropriate for the value,
133  // and if not, downgrade it until we find one that works
134  auto multiplier = static_cast<int8_t>(MemoryUnitEnum::Yottabytes);
135  for ( ; multiplier > 0; multiplier--) {
136  auto value_tmp = static_cast<double>(val);
137  for (int8_t i = 0; i < static_cast<int8_t>(multiplier); i++) {
138  value_tmp /= 1024.0;
139  }
140  if (value_tmp > 1.) {
141  goto found_appropiate_memory_multiplier;
142  }
143  }
144 
145  // We found a multiplier that results in a value over 1.0, use it
146  found_appropiate_memory_multiplier:
147  vtAssert(
148  multiplier <= static_cast<int8_t>(MemoryUnitEnum::Yottabytes) and
149  multiplier >= 0,
150  "Must be a valid memory unit"
151  );
152 
153  auto unit_name = getMemoryUnitName(static_cast<MemoryUnitEnum>(multiplier));
154 
155  // Use the default spec if we have base units (multiplier of 1)---this
156  // means that integer types won't have a decimal place (as desired)
157  // Otherwise, add decimal places since we used a multiplier
158 
159  if (multiplier == 0) {
160  return fmt::format(
161  default_spec + " {}", val, unitFormat(unit_name, align)
162  );
163  } else {
164  // Compute the new value with multiplier as a double
165  auto new_value = static_cast<double>(val);
166  for (int8_t i = 0; i < static_cast<int8_t>(multiplier); i++) {
167  new_value /= 1024.0;
168  }
169 
170  auto decimal = std::string{decimal_format};
171  return fmt::format(
172  decimal + " {}", new_value, unitFormat(unit_name, align)
173  );
174  }
175  break;
176  }
177  case DiagnosticUnit::Units: {
178 
179  // Start with the largest unit, testing if its appropriate for the value,
180  // and if not, downgrade it until we find one that works
181  auto multiplier = static_cast<int8_t>(UnitMultiplier::Quintillions);
182  for ( ; multiplier > 0; multiplier--) {
183  auto value_tmp = static_cast<double>(val);
184  for (int8_t i = 0; i < static_cast<int8_t>(multiplier); i++) {
185  value_tmp /= 1000.0;
186  }
187  if (value_tmp > 1.) {
188  goto found_appropiate_unit_multiplier;
189  }
190  }
191 
192  // We found a multiplier that results in a value over 1.0, use it
193  found_appropiate_unit_multiplier:
194  vtAssert(
195  multiplier <= static_cast<int8_t>(UnitMultiplier::Quintillions) and
196  multiplier >= 0,
197  "Must be a valid unit multiplier"
198  );
199 
200  auto unit_name =
201  diagnosticMultiplierString(static_cast<UnitMultiplier>(multiplier));
202 
203  // Use the default spec if we have base units (multiplier of 1)---this
204  // means that integer types won't have a decimal place (as desired)
205  // Otherwise, add decimal places since we used a multiplier
206 
207  if (multiplier == 0) {
208  return fmt::format(
209  default_spec + " {}", val, unitFormat(unit_name, align)
210  );
211  } else {
212  // Compute the new value with multiplier as a double
213  auto new_value = static_cast<double>(val);

216  }
217 
218  auto decimal = std::string{decimal_format};
219  return fmt::format(
220  decimal + " {}", new_value, unitFormat(unit_name, align)

222  }

224  break;
225  }

229  // and if not, upgrade it until we find one that works

232  auto value_tmp = static_cast<double>(val);

234  value_tmp *= 1000.0;

238  }
