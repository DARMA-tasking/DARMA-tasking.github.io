96 template <typename>
97 void CollectionManager::cleanupAll() {
98  /*
99  * Run the cleanup functions for type-specific cleanup that can not be
100  * performed without capturing the type of each collection
101  */
102  int num_cleanup = 0;
103  int tot_cleanup = static_cast<int>(cleanup_fns_.size());
104  auto iter = cleanup_fns_.begin();
105  while (iter != cleanup_fns_.end()) {
106  auto lst = std::move(iter->second);
107  iter = cleanup_fns_.erase(iter);
108  num_cleanup++;
109  for (auto fn : lst) {
110  fn();
111  }
112  }
113  vtAssertExpr(cleanup_fns_.size() == 0);
114  vtAssertExpr(num_cleanup == tot_cleanup);

262  trace::TraceEventIDType trace_event = trace::no_trace_event;

273  * other end the sender produces p units for each broadcast to the default
274  * group

276  vt_debug_print(
277  verbose, vrt_coll,

282  );

337  auto const pfrom = msg->getSenderElm();
338 
339  if (pfrom.id == elm::no_element_id) {
340  return;
341  }
342 
343  auto const pto = col_ptr->getElmID();
344  auto& lb_data = col_ptr->getLBData();

352  static_cast<typename std::underlying_type<elm::CommCategory>::type>(cat)
353  );
354  if (
355  cat == elm::CommCategory::SendRecv or
356  cat == elm::CommCategory::Broadcast

358  bool bcast = cat == elm::CommCategory::SendRecv ? false : true;
359  lb_data.recvObjData(pto, pfrom, msg_size, bcast);

361  cat == elm::CommCategory::NodeToCollection or
362  cat == elm::CommCategory::NodeToCollectionBcast
363  ) {
364  bool bcast = cat == elm::CommCategory::NodeToCollection ? false : true;
365  auto nfrom = msg->getFromNode();
366  lb_data.recvFromNode(pto, nfrom, msg_size, bcast);
367  }

1515 template <typename ColT>
1516 void CollectionManager::finishModification(
1517  CollectionProxyWrapType<ColT> const& proxy, ModifierToken&& token
1518 ) {
1519  using IndexType = typename ColT::IndexType;
1520 
1521  auto untyped_proxy = proxy.getProxy();
1522 
1523  auto const epoch = token.modifyEpoch();
1524 
1525  vt_debug_print(
1526  normal, vrt_coll,
1527  "finishModification: epoch={:x}\n", epoch
1528  );
1529 
1530  theTerm()->finishedEpoch(epoch);
1531  runSchedulerThrough(epoch);
1532 
1533  // Compute the proper reduce stamp for the insertions that took place
1534  using StrongSeq = collective::reduce::detail::StrongSeq;
1535  using SeqType = typename StrongSeq::Type;
1536 
1537  auto elm_holder = findElmHolder<IndexType>(untyped_proxy);
1538  SeqType min_seq = std::numeric_limits<SeqType>::max();
1539  elm_holder->foreach([&](IndexType const&, Indexable<IndexType>* c) {
1540  // skip zeros since they are insertions that just happened
1541  if (*c->reduce_stamp_ != 0) {
1542  min_seq = std::min(*c->reduce_stamp_, min_seq);
1543  }
1544  });
1545 
1546  // Compute the global min stamp with a reduction
1547  auto r = theCollection()->reducer();
1548 
1549  using collective::reduce::makeStamp;
1550  using collective::reduce::StrongUserID;
1551 
1552  NodeType collective_root = 0;
1553  auto stamp = makeStamp<StrongUserID>(untyped_proxy);
1554  auto msg = makeMessage<CollectionStampMsg>(untyped_proxy, min_seq);
1555  auto cb = theCB()->makeBcast<&CollectionManager::computeReduceStamp>();
1556  r->reduce<collective::MinOp<SeqType>>(collective_root, msg.get(), cb, stamp);
1557 
1558  theSched()->runSchedulerWhile([untyped_proxy]{
1559  return theCollection()->reduce_stamp_.find(untyped_proxy) ==
1560  theCollection()->reduce_stamp_.end();
1561  });
1562 
1563  auto iter = reduce_stamp_.find(untyped_proxy);
1564  vtAssert(iter != reduce_stamp_.end(), "Must have value");
1565 
1566  SeqType const global_min_stamp = iter->second;
1567  reduce_stamp_.erase(iter);
1568 
1569  // set all new insertions to proper value
1570  elm_holder->foreach([&](IndexType const&, Indexable<IndexType>* c) {
1571  if (*c->reduce_stamp_ == 0) {
1572  *c->reduce_stamp_ = global_min_stamp;
1573  }
1574  });
1575 
1576  vt_debug_print(
1577  verbose, vrt_coll,
1578  "finishedInserting: creating new group\n"
1579  );
1580 
1581  runInEpochCollective([&]{ constructGroup<ColT>(untyped_proxy); });
1582 }

1615 template <typename ColT, typename MsgT>
1616 void CollectionManager::insert(
1617  CollectionProxyWrapType<ColT> const& proxy, typename ColT::IndexType idx,
1618  NodeType const node, ModifierToken& token, MsgSharedPtr<MsgT> insert_msg,
1619  bool pinged_home_already
1620 ) {
1621  using IndexType = typename ColT::IndexType;
1622 
1623  auto const modify_epoch = token.modifyEpoch();
1624  auto const untyped_proxy = proxy.getProxy();
1625  vtAssert(modify_epoch != no_epoch, "Modification epoch should be valid");
1626 
1627  vt_debug_print(
1628  normal, vrt_coll,
1629  "insert: proxy={:x}\n",
1630  untyped_proxy
1631  );
1632 
1633  theMsg()->pushEpoch(modify_epoch);
1634 
1635  auto const mapped_node = getMappedNode<ColT>(proxy, idx);
1636  auto const has_explicit_node = node != uninitialized_destination;
1637  auto const insert_node = has_explicit_node ? node : mapped_node;
1638  auto const this_node = theContext()->getNode();
1639 
1640  bool proceed_with_insertion = true;
1641 
1642  vt_debug_print(
1643  normal, vrt_coll,
1644  "insert: insert_node={}, mapped_node={}\n",
1645  insert_node, mapped_node
1646  );
1647 
1648  if (insert_node == this_node and not pinged_home_already) {
1649  // Case 0--insertion from home node onto home node (or message sent from
1650  // another node to insert on home node)
1651  if (mapped_node == this_node) {
1652  auto elm_holder = findElmHolder<IndexType>(untyped_proxy);
1653  auto const elm_exists = elm_holder->exists(idx);
1654  if (elm_exists) {
1655  // element exists here and is live--return
1656  proceed_with_insertion = false;
1657  } else {
1658  auto lm = theLocMan()->getCollectionLM<IndexType>(untyped_proxy);
1659  auto elm_lives_somewhere = lm->isCached(idx);
1660  if (elm_lives_somewhere) {
1661  // element exists somewhere in the system and since we are the home
1662  // we check the cache to determine if it has been inserted
1663  proceed_with_insertion = false;
1664  }
1665  }
1666  } else {
1667  // Case 1: insertion from the non-home node---we must check if the home
1668  // has a reserved entry from another insertion. If so, we cancel the
1669  // insertion---otherwise, we reserve for this insertion

1671  proxy, idx, insert_node, mapped_node, modify_epoch, insert_msg
1672  );

1675  mapped_node, msg

1678  }

1682  auto cons_fn = detail::InsertMsgDispatcher<MsgT, ColT>::makeCons(insert_msg);
1683  makeCollectionElement<ColT>(untyped_proxy, idx, mapped_node, cons_fn);
1684 
1685  auto elm_holder = findElmHolder<IndexType>(untyped_proxy);
1686  auto raw_ptr = elm_holder->lookup(idx).getRawPtr();
1687  raw_ptr->getLBData().updatePhase(thePhase()->getCurrentPhase());
1688  } else if (insert_node != this_node) {
1689  auto msg = makeMessage<InsertMsg<ColT, MsgT>>(
1690  proxy, idx, insert_node, mapped_node, modify_epoch, insert_msg
1691  );

1693  theMsg()->sendMsg<InsertMsg<ColT, MsgT>,insertHandler<ColT, MsgT>>(
1694  insert_node, msg
1695  );
1696  }
1697 
1698  theMsg()->popEpoch(modify_epoch);
1699 }
