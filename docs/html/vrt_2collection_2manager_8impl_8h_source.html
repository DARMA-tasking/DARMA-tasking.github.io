155 template <typename SysMsgT>
156 /*static*/ void CollectionManager::distConstruct(SysMsgT* msg) {
157  using ColT = typename SysMsgT::CollectionType;
158  using IndexT = typename SysMsgT::IndexType;
159  using BaseIdxType = vt::index::BaseIndex;
160 
161  auto const num_nodes = theContext()->getNumNodes();
162 
163  auto& info = msg->info;
164 
165  // The VirtualProxyType for this construction
166  auto proxy = info.getProxy();
167  // The insert epoch for this collection
168  auto const insert_epoch = info.getInsertEpoch();
169  // Count the number of elements locally created here
170  int64_t num_elements_created = 0;
171  // Get the mapping function handle
172  auto const map_han = msg->map;
173  // Get the range for the construction
174  auto range = msg->info.range_;
175 
176  theCollection()->insertCollectionInfo(proxy,msg->map,insert_epoch);
177 
178  // Add the cleanup function for this node
179  theCollection()->addCleanupFn<ColT>(proxy);
180 
181  if (info.immediate_) {
182  // Get the handler function
183  auto fn = auto_registry::getHandlerMap(map_han);
184  // Total count across the statically sized collection
185  std::size_t num_elms = info.range_.getSize();
186 
187  vt_debug_print(
188  vrt_coll, node,
189  "running foreach: size={}, range={}, map_han={}\n",
190  num_elms, range, map_han
191  );
192 
193  range.foreach([&](IndexT cur_idx) mutable {
194  vt_debug_print_verbose(
195  vrt_coll, node,
196  "running foreach: before map: cur_idx={}, range={}\n",
197  cur_idx.toString(), range.toString()
198  );
199 
200  auto const cur = static_cast<BaseIdxType*>(&cur_idx);
201  auto const max = static_cast<BaseIdxType*>(&range);
202 
203  auto mapped_node = fn(cur, max, num_nodes);
204 
205  vt_debug_print(
206  vrt_coll, node,
207  "construct: foreach: node={}, cur_idx={}, max_range={}\n",
208  mapped_node, cur_idx.toString(), range.toString()
209  );
210 
211  if (theContext()->getNode() == mapped_node) {
212  using IdxContextHolder = InsertContextHolder<IndexT>;
213 
214  // Actually construct the element. If the detector is enabled, call the
215  // detection-based overloads to invoke the constructor with the
216  // optionally-positional index. Otherwise, invoke constructor without
217  // the index as a parameter
218 
219  // Set the current context index to `cur_idx`. This enables the user to
220  // query the index of their collection element in the constructor, which
221  // is often very handy
222  IdxContextHolder::set(&cur_idx,proxy);
223 
224  #if vt_check_enabled(detector) && vt_check_enabled(cons_multi_idx)
225  auto new_vc = DerefCons::derefTuple<ColT, IndexT, decltype(msg->tup)>(
226  num_elms, cur_idx, &msg->tup
227  );
228  #else
229  using Args = typename SysMsgT::ArgsTupleType;
230  static constexpr auto num_args = std::tuple_size<Args>::value;
231  auto new_vc = CollectionManager::runConstructor<ColT, IndexT>(
232  num_elms, cur_idx, &msg->tup, std::make_index_sequence<num_args>{}
233  );
234  #endif
235 
236  /*
237  * Set direct attributes of the newly constructed element directly on
238  * the user's class
239  */
240  CollectionTypeAttorney::setup(new_vc, num_elms, cur_idx, proxy);
241 
242  // Insert the element into the managed holder for elements
243  theCollection()->insertCollectionElement<ColT, IndexT>(
244  std::move(new_vc), cur_idx, msg->info.range_, map_han, proxy,
245  info.immediate_, mapped_node
246  );
247 
248  // Clear the current index context
249  IdxContextHolder::clear();
250 
251  // Increment the number of elements created locally
252  num_elements_created++;
253  }
254  });
255  } else {
256  vtAssert(num_elements_created == 0, "Elements should not be created");
257  }
258 
259  if (num_elements_created == 0) {
260  insertMetaCollection<ColT>(proxy,map_han,range,info.immediate_);
261  }
262 
263  // Reduce construction of the collection to release dependencies when
264  // construction has finsihed
265  reduceConstruction<ColT>(proxy);
266 
267  // Construct a underlying group for the collection
268  groupConstruction<ColT>(proxy,info.immediate_);
269 }

549  VirtualProxyType const& proxy, EpochType const& epoch, MsgT* msg
550 ) {

552  if (proxy_iter == details::Broadcasts<ColT>::m_.end()) {

561  )
562  );
563  } else {

565  if (epoch_iter == proxy_iter->second.end()) {
566  proxy_iter->second.emplace(
567  std::piecewise_construct,
568  std::forward_as_tuple(epoch),
569  std::forward_as_tuple(msg)
570  );
571  } else {
572  epoch_iter->second = msg;
573  }
574  }
