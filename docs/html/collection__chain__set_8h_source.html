122  void nextStep(

124  for (auto& entry : chains_) {

128  // The parameter `true` here tells VT to use an efficient rooted DS-epoch

133 
134  chain.add(new_epoch, step_action(idx));

136  vt::theMsg()->popEpoch(new_epoch);

238  static void mergeStepCollective(
239  std::string const& label, CollectionChainSet& a, CollectionChainSet& b,
240  std::function<PendingSend(Index)> step_action) {
241  auto epoch = theTerm()->makeEpochCollective(label);
242  vt::theMsg()->pushEpoch(epoch);
243 
244  for (auto& entry : a.chains_) {
245  auto& idx = entry.first;
246  auto& chaina = entry.second;
247  auto chainb_pos = b.chains_.find(entry.first);
248  vtAssert(
249  chainb_pos != b.chains_.end(),
250  fmt::format("index {} must be present in chainset b", entry.first));
251 
252  auto& chainb = chainb_pos->second;
253  DependentSendChain::mergeChainStep(
254  chaina, chainb, epoch, step_action(idx));
255  }
256 
257  vt::theMsg()->popEpoch(epoch);
258  theTerm()->finishedEpoch(epoch);
259  }
