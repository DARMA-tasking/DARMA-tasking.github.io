118  void nextStep(
119  std::string const& label, std::function<PendingSend(Index)> step_action
120  ) {
121  for (auto &entry : chains_) {
122  auto& idx = entry.first;
123  auto& chain = entry.second;
124 
125  // The parameter `true` here tells VT to use an efficient rooted DS-epoch
126  // by default. This can still be overridden by command-line flags
127  EpochType new_epoch = theTerm()->makeEpochRooted(label, term::UseDS{true});
128  vt::theMsg()->pushEpoch(new_epoch);
129 
130  chain.add(new_epoch, step_action(idx));
131 
132  vt::theMsg()->popEpoch(new_epoch);
133  theTerm()->finishedEpoch(new_epoch);
134  }
135  }
