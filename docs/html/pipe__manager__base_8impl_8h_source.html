109 template <typename SignalT, typename ListenerT>
110 void PipeManagerBase::registerCallback(
111  PipeType const& pipe, ListenerT&& listener, bool update_state
112 ) {
113  auto const holder_id = signal_holder_<SignalT>.getID();
114 
115  auto cleanup_iter = signal_cleanup_fns_.find(holder_id);
116  if (cleanup_iter == signal_cleanup_fns_.end()) {
117  signal_cleanup_fns_[holder_id] = []{
118  signal_holder_<SignalT>.clearAll();
119  };
120  }
121 
122  /*
123  * Save the new listener in the typed signal holder container for delivery
124  * when a signal arrives
125  */
126  signal_holder_<SignalT>.addListener(pipe, std::move(listener));
127  /*
128  * Update the number of registered listeners in the pipe state
129  */
130  if (update_state) {
131  auto iter = pipe_state_.find(pipe);
132  vtAssert(iter != pipe_state_.end(), "Pipe state must exist");
133  iter->second.listenerReg();
134  }
135 }

137 template <typename MsgT, typename ListenerT>
138 PipeType PipeManagerBase::makeCallbackAny(
139  bool const& persist, ListenerT&& listener, bool const& dispatch,
140  PipeRefType num_signals, PipeRefType num_listeners
141 ) {
142  using SignalType = signal::Signal<MsgT>;
143 
144  bool const& send_back = false;
145  auto const& pipe_id = makePipeID(persist,send_back);
146  /*
147  * Create a dispatch function so any inputs can be dispatched without any
148  * type information on the sending side
149  */
150  auto dispatch_fn = [pipe_id](void* input) {
151  auto data = reinterpret_cast<typename SignalType::DataPtrType>(input);
152  signal_holder_<SignalType>.deliverAll(pipe_id,data);
153  };
154  DispatchFuncType dfn = nullptr;
155  if (dispatch) {
156  dfn = dispatch_fn;
157  }
158  /*
159  * Add a new entry for pipe state to track it on this node
160  */
161  newPipeState(pipe_id,persist,dispatch,num_signals,num_listeners,1,dfn);
162  /*
163  * Do not update the pipe state since this newly registered listener is
164  * already counted by the `1' in the newPipeState(...,1)
165  */
166  registerCallback<SignalType>(pipe_id,std::move(listener),false);
167  return pipe_id;
168 }

187 template <typename MsgT, typename ListenerT>
188 void PipeManagerBase::addListenerAny(PipeType const& pipe, ListenerT&& fn) {
189  using SignalType = signal::Signal<MsgT>;
190  auto iter = pipe_state_.find(pipe);
191  vtAssert(iter != pipe_state_.end(), "Pipe state must exist");
192  auto& state = iter->second;
193  if (!state.hasDispatch()) {
194  auto fn2 = [pipe](void* input) {
195  auto data = reinterpret_cast<typename SignalType::DataPtrType>(input);
196  signal_holder_<SignalType>.deliverAll(pipe,data);
197  };
198  state.setDispatch(fn2);
199  }
200  return registerCallback<SignalType>(pipe, std::move(fn));
201 }

204 void PipeManagerBase::addListener(PipeType const& pipe, FuncMsgType<MsgT> fn) {
205  using SignalType = signal::Signal<MsgT>;

207  vtAssert(iter != pipe_state_.end(), "Pipe state must exist");
