65 template <typename T, HandleEnum E>
66 void Holder<T,E>::allocateDataWindow(std::size_t const in_len) {
67  std::size_t len = in_len == 0 ? count_ : in_len;
68  vt_debug_print(
69  terse, rdma,
70  "allocate: len={}, in_len={}, count_={}\n", len, in_len, count_
71  );
72  // Allocate data window
73  MPI_Comm comm = theContext()->getComm();
74  MPI_Alloc_mem(len * sizeof(T), MPI_INFO_NULL, &data_base_);
75  MPI_Win_create(
76  data_base_, len * sizeof(T), sizeof(T), MPI_INFO_NULL, comm,
77  &data_window_
78  );
79  if (not uniform_size_) {
80  // Allocate control window
81  MPI_Alloc_mem(sizeof(uint64_t), MPI_INFO_NULL, &control_base_);
82  MPI_Win_create(
83  control_base_, sizeof(uint64_t), sizeof(uint64_t), MPI_INFO_NULL, comm,
84  &control_window_
85  );
86  {
87  auto this_node = theContext()->getNode();
88  LockMPI _scope_lock(Lock::Exclusive, this_node, control_window_, false);
89  auto mpi_type = TypeMPI<uint64_t>::getType();
90  MPI_Put(&count_, 1, mpi_type, this_node, 0, 1, mpi_type, control_window_);
91  vt_debug_print(
92  verbose, rdma,
93  "setting allocate size: size={}, window={}\n",
94  count_, print_ptr(&control_window_)
95  );
96  }
97  }
98  ready_ = true;
99 }
