56 /*
57  * Discrete interval tree for encoding a set of partially tangent,
58  * non-overlapping integral values into interval ranges seamlessly in a
59  * tree. Maintains a "DIET" data structure: Discrete Interval Encoding
60  * Tree. When inserting contiguous elements in order, the storage (non-bit) and
61  * time is constant O(1).
62  */
63

178  void erase(DomainT const& val) {
179  vtAssert(existsGlobal(val), "This element must exist in the set");
180  IntervalType j(val);
181  auto iter = set_.find(j);
182  bool in_set = iter != set_.end();
183  vt_debug_print(
184  normal, gen,
185  "OrderedSet: erase: bounds=[{:x},{:x}] size={}, comp={}, val={}, in={}\n",
186  lb_, ub_, size(), compressedSize(), j, in_set
187  );
188  vtAssert(in_set, "The element must exist in a interval bucket");
189  if (in_set) {
190  if (iter->width() == 1) {
191  bool invalid_hint = iter == hint_;
192  auto ret = set_.erase(iter);
193  if (invalid_hint) {
194  hint_ = ret;
195  }
196  } else if (iter->upper() == val) {
197  auto& to_split = const_cast<IntervalType&>(*iter);
198  to_split.setUpper(iter->upper() - 1);
199  } else if (iter->lower() == val) {
200  auto& to_split = const_cast<IntervalType&>(*iter);
201  to_split.setLower(iter->lower() + 1);
202  } else {
203  // Slice the interval into two interval nodes
204  vtAssert(iter->width() > 2, "Interval width must be greater than 2");
205  auto ub = iter->upper();
206  auto& to_split = const_cast<IntervalType&>(*iter);
207  to_split.setUpper(val - 1);
208  IntervalType i(val+1, ub);
209  insertSet(iter,std::move(i));
210  }
211  eraseGlobal(val);
212  }
213  }

290  void eraseGlobal(DomainT const& val) {
291  bool is_lower = lb_ == val;
292  bool is_upper = ub_ == val;
293  if (is_lower) {
294  if (set_.size() > 0) {
295  lb_ = set_.begin()->lower();
296  } else {
297  lb_ = 0;
298  }
299  }
300  if (is_upper) {
301  if (set_.size() > 0) {
302  ub_ = std::prev(set_.end())->upper();
303  } else {
304  ub_ = 0;
305  }
306  }
307  // Decrement count of non-compressed elements
308  elms_--;

311  void insertGlobal(IntervalType const& i) {
312  vt_debug_print(
313  verbose, gen,
314  "OrderedSet: insertGlobal: bounds=[{:x},{:x}] elms_={}, i={}\n",
315  lb_, ub_, elms_, i
316  );
317 
318  if (elms_ == 0) {
319  lb_ = i.lower();
320  ub_ = i.upper();
321  } else {
322  // Expand the global bounds in this interval set
323  lb_ = std::min<DomainT>(lb_, i.lower());
324  ub_ = std::max<DomainT>(ub_, i.upper());
325  }
326  // Increment count of non-compressed elements
327  elms_ += i.width();
328  }

330  IteratorType join(IteratorType it) {
331  vt_debug_print(
332  verbose, gen,
333  "OrderedSet: join: bounds=[{:x},{:x}] elms_={}\n",
334  lb_, ub_, elms_
335  );
336 
337  auto it2 = joinLeft(it);
338  return joinRight(it2);
339  }
340 
341  IteratorType joinLeft(IteratorType it) {
342  vtAssertExpr(it not_eq set_.end());
343  if (it not_eq set_.begin()) {
344  auto prev = std::prev(it);
345  if (it->tangent(*prev) == PositionType::TangentLeft) {
346  auto tmp = *prev;
347  set_.erase(prev);
348  auto& to_fuse = const_cast<IntervalType&>(*it);
349  to_fuse.join(tmp, PositionType::TangentLeft);
350  }
351  }
352  return it;
353  }
354 
355  IteratorType joinRight(IteratorType it) {
356  vtAssertExpr(it not_eq set_.end());
357  auto next = std::next(it);
358  if (next not_eq set_.end()) {
359  if (it->tangent(*next) == PositionType::TangentRight) {
360  auto tmp = *next;
361  set_.erase(next);
362  auto& to_fuse = const_cast<IntervalType&>(*it);
363  to_fuse.join(tmp, PositionType::TangentRight);
364  }
365  }
366  return it;
367  }

395 
396  Iter& operator++() {
397  if (impl_->lower() + static_cast<DomainT>(cur_) < impl_->upper()) {
398  cur_++;
399  } else {
400  cur_ = 0;
401  impl_ = std::next(impl_);
402  }

404  }

406  Iter& operator--() {
407  if (cur_ == 0) {

412  }

414  }
