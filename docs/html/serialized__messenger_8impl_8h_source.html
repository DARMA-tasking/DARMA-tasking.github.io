98 template <typename UserMsgT>
99 /*static*/ void SerializedMessenger::serialMsgHandler(
100  SerialWrapperMsgType<UserMsgT>* sys_msg
101 ) {
102  auto const handler = sys_msg->handler;
103  auto const& recv_tag = sys_msg->data_recv_tag;
104  auto const epoch = envelopeGetEpoch(sys_msg->env);
105 
106  vt_debug_print(
107  serial_msg, node,
108  "serialMsgHandler: non-eager, recvDataMsg: msg={}, handler={}, "
109  "recv_tag={}, epoch={}\n",
110  print_ptr(sys_msg), handler, recv_tag, epoch
111  );
112 
113  bool const is_valid_epoch = epoch != no_epoch;
114 
115  if (is_valid_epoch) {
116  theTerm()->produce(epoch);
117  }
118 
119  auto node = sys_msg->from_node;
120  theMsg()->recvDataMsg(
121  recv_tag, sys_msg->from_node,
122  [handler,recv_tag,node,epoch,is_valid_epoch]
123  (RDMA_GetType ptr, ActionType action){
124  // be careful here not to use "sys_msg", it is no longer valid
125  auto msg_data = reinterpret_cast<SerialByteType*>(std::get<0>(ptr));
126  auto msg = deserializeFullMessage<UserMsgT>(msg_data);
127 
128  vt_debug_print(
129  serial_msg, node,
130  "serialMsgHandler: recvDataMsg finished: handler={}, recv_tag={},"
131  "epoch={}\n",
132  handler, recv_tag, envelopeGetEpoch(msg->env)
133  );
134 
135  if (is_valid_epoch) {
136  theMsg()->pushEpoch(epoch);
137  }
138  runnable::Runnable<UserMsgT>::run(handler, nullptr, msg.get(), node);
139  action();
140 
141  if (is_valid_epoch) {
142  theMsg()->popEpoch(epoch);
143  theTerm()->consume(epoch);
144  }
145  }
146  );
147 }

194 template <typename MsgT, typename BaseT>
195 /*static*/ messaging::PendingSend
196  SerializedMessenger::broadcastSerialMsg(
197  MsgT* msg_ptr, HandlerType han
198 ) {
199  using PayloadMsg = SerialEagerPayloadMsg<MsgT, BaseT>;
200 
201  auto msg = promoteMsg(msg_ptr);
202 
203  MsgSharedPtr<SerialEagerPayloadMsg<MsgT,BaseT>> payload_msg = nullptr;
204  MsgSharedPtr<SerialWrapperMsgType<MsgT>> sys_msg = nullptr;
205  SizeType ptr_size = 0;
206  auto sys_size = sizeof(typename decltype(sys_msg)::MsgType);
207 
208  envelopeSetHasBeenSerialized(msg->env, false);
209 
210  auto serialized_msg = checkpoint::serialize(
211  *msg.get(), [&](SizeType size) -> SerialByteType* {
212  ptr_size = size;
213  if (size >= serialized_msg_eager_size) {
214  sys_msg = makeMessageSz<SerialWrapperMsgType<MsgT>>(ptr_size);
215  return reinterpret_cast<char*>(sys_msg.get()) + sys_size;
216  } else {
217  payload_msg = makeMessage<PayloadMsg>(
218  static_cast<NumBytesType>(size)
219  );
220  return payload_msg->payload.data();
221  }
222  }
223  );
224 
225  vtAssertInfo(
226  envelopeHasBeenSerialized(msg->env),
227  "A message was serialized that did not correctly serialize parents."
228  " All serialized messages are required to call serialize on the"
229  " message's parent type. See 'msg_serialize_parent'.",
230  typeid(MsgT).name()
231  );
232 
233  auto const& group_ = envelopeGetGroup(msg->env);
234 
235  if (ptr_size < serialized_msg_eager_size) {
236  vtAssert(
237  ptr_size < serialized_msg_eager_size,
238  "Must be smaller for eager protocol"
239  );
240 
241  // move serialized msg envelope to system envelope to preserve info
242  auto cur_ref = envelopeGetRef(payload_msg->env);
243  payload_msg->env = msg->env;
244  payload_msg->handler = han;
245  payload_msg->from_node = theContext()->getNode();
246  envelopeSetGroup(payload_msg->env, group_);
247  envelopeSetRef(payload_msg->env, cur_ref);
248 
249  vt_debug_print(
250  serial_msg, node,
251  "broadcastSerialMsg (eager): han={}, size={}, "
252  "serialized_msg_eager_size={}, group={:x}\n",
253  han, ptr_size, serialized_msg_eager_size, group_
254  );
255 
256  theMsg()->markAsSerialMsgMessage(payload_msg);
257  return theMsg()->broadcastMsg<PayloadMsg,payloadMsgHandler>(
258  payload_msg.get()
259  );
260  } else {
261  auto const& total_size = ptr_size + sys_size;
262 
263  auto traceable_han = han;
264 
265 # if vt_check_enabled(trace_enabled)
266  // Since we aren't sending the message (just packing it into a buffer, we
267  // need to transfer whether the handler should be traced on that message
268  auto_registry::HandlerManagerType::setHandlerTrace(
269  traceable_han, envelopeGetTraceRuntimeEnabled(msg->env)
270  );
271 # endif
272 
273  auto cur_ref = envelopeGetRef(sys_msg->env);
274  sys_msg->handler = traceable_han;
275  sys_msg->env = msg->env;
276  sys_msg->from_node = theContext()->getNode();
277  sys_msg->ptr_size = ptr_size;
278  envelopeSetGroup(sys_msg->env, group_);
279  envelopeSetRef(sys_msg->env, cur_ref);
280 
281  vt_debug_print(
282  serial_msg, node,
283  "broadcastSerialMsg (non-eager): container: han={}, sys_size={}, "
284  "ptr_size={}, total_size={}, group={:x}\n",
285  traceable_han, sys_size, ptr_size, total_size, group_
286  );
287 
288  using MsgType = SerialWrapperMsgType<MsgT>;
289  theMsg()->markAsSerialMsgMessage(sys_msg);
290  return theMsg()->broadcastMsgSz<MsgType,serialMsgHandlerBcast>(
291  sys_msg.get(), total_size, no_tag
292  );
293  }

296 template <typename MsgT, typename BaseT>
297 /*static*/ messaging::PendingSend SerializedMessenger::sendSerialMsgSendImpl(
298  MsgT* msg_ptr, HandlerType typed_handler,
299  ActionEagerSend<MsgT, BaseT> eager_sender, ActionDataSend data_sender
300 ) {
301 #ifndef vt_quirked_serialize_method_detection
302  static_assert(
303  ::vt::messaging::has_own_serialize<MsgT>,
304  "Messages sent via SerializedMessenger must have a serialization function."
305  );
306 #endif
307  static_assert(
308  ::vt::messaging::msg_defines_serialize_mode<MsgT>::value,
309  "Messages that define a serialization function must specify serialization mode."
310  );
311  static_assert(
312  ::vt::messaging::msg_serialization_mode<MsgT>::required,
313  "Messages sent via SerializedMessenger should require serialization."
314  );
315 
316  auto msg = promoteMsg(msg_ptr);
317 
318  MsgSharedPtr<SerialEagerPayloadMsg<MsgT,BaseT>> payload_msg = nullptr;
319  SerialByteType* ptr = nullptr;
320  SizeType ptr_size = 0;
321 
322  envelopeSetHasBeenSerialized(msg->env, false);
323 
324  auto serialized_msg = checkpoint::serialize(
325  *msg.get(), [&](SizeType size) -> SerialByteType* {
326  ptr_size = size;
327 
328  if (size > serialized_msg_eager_size) {
329  ptr = static_cast<SerialByteType*>(std::malloc(size));
330  return ptr;
331  } else {
332  payload_msg = makeMessage<SerialEagerPayloadMsg<MsgT, BaseT>>(
333  static_cast<NumBytesType>(ptr_size)
334  );
335  return payload_msg->payload.data();
336  }
337  }
338  );
339 
340  vtAssertInfo(
341  envelopeHasBeenSerialized(msg->env),
342  "A message was serialized that did not correctly serialize parents."
343  " All serialized messages are required to call serialize on the"
344  " message's parent type. See 'msg_serialize_parent'.",
345  typeid(MsgT).name()
346  );
347 
348  //fmt::print("ptr_size={}\n", ptr_size);
349 
350  vt_debug_print(
351  serial_msg, node,
352  "sendSerialMsgHandler: ptr_size={}, han={}, eager={}, epoch={}\n",
353  ptr_size, typed_handler, ptr_size <= serialized_msg_eager_size,
354  envelopeGetEpoch(msg_ptr->env)
355  );
356 
357  if (ptr_size > serialized_msg_eager_size) {
358  vt_debug_print(
359  serial_msg, node,
360  "sendSerialMsg: non-eager: ptr_size={}\n", ptr_size
361  );
362 
363  vtAssertExpr(payload_msg == nullptr && data_sender != nullptr);
364 
365  auto send_data = [=](NodeType dest) -> messaging::PendingSend {
366  auto const& node = theContext()->getNode();
367  if (node != dest) {
368  auto sys_msg = makeMessage<SerialWrapperMsgType<MsgT>>();
369  auto send_serialized = [=](Active::SendFnType send){
370  auto ret = send(RDMA_GetType{ptr, ptr_size}, dest, no_tag);
371  EventType event = std::get<0>(ret);
372  theEvent()->attachAction(event, [=]{ std::free(ptr); });
373  sys_msg->data_recv_tag = std::get<1>(ret);
374  };
375  auto cur_ref = envelopeGetRef(sys_msg->env);
376  sys_msg->env = msg->env;
377  sys_msg->handler = typed_handler;
378  sys_msg->from_node = theContext()->getNode();
379  envelopeSetRef(sys_msg->env, cur_ref);
380 
381  vt_debug_print(
382  serial_msg, node,
383  "sendSerialMsg: non-eager: dest={}, sys_msg={}, handler={}\n",
384  dest, print_ptr(sys_msg.get()), typed_handler
385  );
386 
387  theMsg()->markAsSerialMsgMessage(sys_msg);
388  return theMsg()->sendMsg<SerialWrapperMsgType<MsgT>, serialMsgHandler>(
389  dest, sys_msg.get(), send_serialized
390  );
391  } else {
392  auto msg_data = ptr;
393  auto user_msg = deserializeFullMessage<MsgT>(msg_data);
394 
395  vt_debug_print(
396  serial_msg, node,
397  "serialMsgHandler: local msg: handler={}\n", typed_handler
398  );
399 
400  auto base_msg = msg.template to<BaseMsgType>();
401  ByteType msg_sz = sizeof(MsgT);
402  return messaging::PendingSend(base_msg, msg_sz, [=](MsgPtr<BaseMsgType> in){
403  runnable::Runnable<MsgT>::run(typed_handler,nullptr,msg.get(),node);
404  });
405  }
406  };
407 
408  return data_sender(send_data);
409  } else {
410  vt_debug_print(
411  serial_msg, node,
412  "sendSerialMsg: eager: ptr_size={}\n", ptr_size
413  );
414 
415  vtAssertExpr(payload_msg != nullptr && eager_sender != nullptr);
416 
417  // move serialized msg envelope to system envelope to preserve info
418  auto cur_ref = envelopeGetRef(payload_msg->env);
419  payload_msg->env = msg->env;
420  payload_msg->handler = typed_handler;
421  payload_msg->from_node = theContext()->getNode();
422  envelopeSetRef(payload_msg->env, cur_ref);
423 
424  return eager_sender(payload_msg);
425  }
426 }
