94 template <typename Tuple>
95 static void dataMessageHandler(DataMsg<Tuple>* msg) {
96  vt_debug_print(
97  param, node,
98  "dataMessageHandler: id={}\n", msg->sub_han
99  );
100 
101 #if vt_check_enabled(trace_enabled)
102  trace::TraceProcessingTag processing_tag;
103  {
104  trace::TraceEntryIDType ep = auto_registry::handlerTraceID(
105  msg->sub_han, auto_registry::RegistryTypeEnum::RegGeneral
106  );
107  trace::TraceEventIDType event = envelopeGetTraceEvent(msg->env);
108 
109  size_t msg_size = sizeof(*msg);
110  NodeType const& from_node = theMsg()->getFromNodeCurrentHandler();
111 
112  processing_tag =
113  theTrace()->beginProcessing(ep, msg_size, event, from_node);
114  }
115 #endif
116 
117  if (HandlerManagerType::isHandlerFunctor(msg->sub_han)) {
118  auto fn = auto_registry::getAutoHandlerFunctor(msg->sub_han);
119  invokeCallableTuple(msg->tup, fn, true);
120  } else {
121  // regular active function
122  auto fn = auto_registry::getAutoHandler(msg->sub_han);
123  invokeCallableTuple(msg->tup, fn, false);
124  }
125 
126 #if vt_check_enabled(trace_enabled)
127  theTrace()->endProcessing(processing_tag);
128 #endif
129 }
