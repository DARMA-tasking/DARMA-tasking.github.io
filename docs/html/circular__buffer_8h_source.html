51 template <typename T, int capacity>
52 struct CircularBuffer {
53 
54  CircularBuffer() = default;
55 
56 private:
57  int getNextEntry() const {
58  int next_entry = head_ + 1;
59  if (next_entry == capacity) {
60  next_entry = 0;
61  }
62  return next_entry;
63  }
64 
65 public:
66  template <typename... Args>
67  void emplace(Args&&... args) {
68  int const next_entry = getNextEntry();
69  elms_[head_] = T{std::forward<Args>(args)...};
70  head_ = next_entry;
71  }
72 
73  void push(T&& t) {
74  int const next_entry = getNextEntry();
75  elms_[head_] = std::move(t);
76  head_ = next_entry;
77  }
78 
79  void push(T const& t) {
80  int const next_entry = getNextEntry();
81  elms_[head_] = t;
82  head_ = next_entry;
83  }
84 
85  T pop() {
86  T elm = std::move(elms_[tail_]);
87  ++tail_;
88  if (tail_ == capacity) {
89  tail_ = 0;
90  }
91  return elm;
92  }
93 
94  int numFree() const {
95  if (head_ == tail_) {
96  return capacity - 1;
97  } else if (head_ < tail_) {
98  return tail_ - head_ - 1;
99  } else {
100  return capacity + tail_ - head_ - 1;
101  }
102  }
103 
104  bool empty() const { return head_ == tail_; }
105  bool full() const { return numFree() == 0; }
106  int size() const { return capacity - 1 - numFree(); }
107 
108  template <typename SerializerT>
109  void serialize(SerializerT& s) {
110  s | head_;
111  s | tail_;
112  s | elms_; // this is inefficient, but its use is footprinting
113  }
114 
115 private:
116  int head_ = 0;
117  int tail_ = 0;
118  std::array<T, capacity> elms_;
119 };
