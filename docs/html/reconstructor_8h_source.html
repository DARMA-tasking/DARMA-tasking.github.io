100  // Default-construct as lowest priority in reconstruction preference
101  template <typename U = T>
102  static T* constructDefault(void* buf, isDefaultConsType<U>* = nullptr) {
103  debug_checkpoint("DeserializerDispatch: default constructor: buf=%p\n", buf);
104  T* t_ptr = new (buf) T{};
105  return t_ptr;
106  }

108  // Fail, no valid option to constructing T
109  template <typename U = T>
110  static T* constructDefault(void* buf, isNotDefaultConsType<U>* = nullptr) {
111  static_assert(
112  SerializableTraits<U,void>::is_tagged_constructible or
113  SerializableTraits<U,void>::is_reconstructible or
114  std::is_default_constructible<U>::value,
115  "Either a default constructor, reconstruct() function, or tagged "
116  "constructor are required for de-serialization"
117  );
118  return nullptr;
119  }

121  /*
122  * Try to reconstruct with the following precedence:
123  *
124  * - Tagged constructor: T(SERIALIZE_CONSTRUCT_TAG{})
125  * - Reconstruct: T::reconstruct(buf) or reconstruct(t, buf)
126  * - Default constructor: T()
127  *
128  * If none of these options work, static assert failure
129  */
