59  template <typename U = T>
60  static T* constructDefault(void* buf, isDefaultConsType<U>* = nullptr) {
61  debug_checkpoint(
62  "DeserializerDispatch: default constructor: buf=%p\n", buf

66  }

70  static T* constructDefault(void* buf, isNotDefaultConsType<U>* = nullptr) {
71  static_assert(
72  SerializableTraits<U, void>::is_tagged_constructible or
73  SerializableTraits<U, void>::is_reconstructible or
74  std::is_default_constructible<U>::value,
75  "Either a default constructor, reconstruct() function, or tagged "
76  "constructor are required for de-serialization"

78  return nullptr;

81  /*
82  * Try to reconstruct with the following precedence:
83  *
84  * - Tagged constructor: T(SERIALIZE_CONSTRUCT_TAG{})
85  * - Reconstruct: T::reconstruct(buf) or reconstruct(t, buf)
86  * - Default constructor: T()
87  *
88  * If none of these options work, static assert failure
89  */
90 
91  // Intrusive reconstruct
92  template <typename U = T>
93  static T* constructReconstruct(void* buf, isReconstructibleType<U>* = nullptr) {
94  debug_checkpoint("DeserializerDispatch: T::reconstruct(): buf=%p\n", buf);
95  auto& t = T::reconstruct(buf);
96  return &t;
97  }
98 
99  // Non-intrusive reconstruct
100  template <typename U = T>
101  static T* constructReconstruct(void* buf, isNonIntReconstructibleType<U>* = nullptr) {
102  debug_checkpoint(
103  "DeserializerDispatch: non-int reconstruct(): buf=%p\n", buf
104  );
105  T* t = nullptr;
106  // Explicitly call bare to invoke ADL
107  reconstruct(t, buf);
108  return t;
109  }
