124  void stageObject(jsonlib const& in) {
125  if (staged_ == nullptr) {
126  staged_ = std::make_unique<jsonlib>();
127  }
128  auto &st = *staged_;
129  if (in.is_object()) {
130  for (auto &kv : in.items()) {
131  st[kv.key()] = kv.value();
132  }
133  } else {
134  vtAbort("Appender::stageObject: cannot stage a non-object");
135  }
136  }

141  void commitStaged() {
142  if (staged_) {

151  StreamLike finish() {
152  oa_->write_character(']');
153  oa_->write_character('}');
154  // causes the final flush to happen
155  oa_ = nullptr;
156  os_.flush();
157  // will close automatically when out of scope
158  finished_ = true;
159  return std::move(os_);
160  }
161 
162  virtual ~Appender() {
163  if (not finished_) {
164  finish();
165  }
166  }
