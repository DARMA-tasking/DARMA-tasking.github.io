74 template <
75  typename T,
76  typename... StartupDeps,
77  typename... RuntimeDeps,
78  typename... Cons
79 >
80 registry::AutoHandlerType ComponentPack::registerComponent(
81  T** ref,
82  typename BaseComponent::StartupDepsPack<StartupDeps...>,
83  typename BaseComponent::RuntimeDepsPack<RuntimeDeps...>,
84  Cons&&... cons
85 ) {
86  ComponentRegistry::dependsOn<T, StartupDeps...>(true);
87  ComponentRegistry::dependsOn<T, RuntimeDeps...>(false);
88 
89  auto idx = registry::makeIdx<T>();
90  //fmt::print("registerComponent name={} idx={}\n", typeid(T).name(), idx);
91 
92  if (registered_set_.find(idx) == registered_set_.end()) {
93  registered_set_.insert(idx);
94  registered_components_.push_back(idx);
95  }
96 
97  // Be very careful here. Don't use std::make_tuple or we loose l-val refs
98  // (which initially was a "feature", but I realized later was probably not
99  // ideal)
100  std::tuple<Cons...> cons_tuple{std::forward<Cons>(cons)...};
101 
102  auto fn = makeCallable(
103  [ref, this, tup = std::move(cons_tuple)]() mutable {
104  auto ptr = tupleCons<T>(std::move(tup));
105 
106  vt_debug_print(
107  verbose, runtime,
108  "ComponentPack: constructed component={}, pollable={}\n",
109  ptr->name(), ptr->pollable()
110  );
111 
112  // Set the reference for access outside
113  if (ref != nullptr) {
114  *ref = ptr.get();
115  }
116 
117  // Add to pollable for the progress function to be triggered
118  if (ptr->pollable()) {
119  pollable_components_.emplace_back(ptr.get());
120  }
121 
122  ptr->initialize();
123  live_components_.emplace_back(std::move(ptr));
124  }
125  );
126 
127  construct_components_.emplace(
128  std::piecewise_construct,
129  std::forward_as_tuple(idx),
130  std::forward_as_tuple(std::move(fn))
131  );
132  return idx;
133 }

143 ComponentPack::extractComponent(std::string const& name) {
144  for (auto iter = live_components_.begin(); iter != live_components_.end(); ++iter) {
145  if ((*iter) != nullptr and (*iter)->name() == name) {
146  T* raw = static_cast<T*>((*iter).release());
147  *iter = nullptr;
148  return std::unique_ptr<T>(raw);
149  }
150  }
151  return std::unique_ptr<T>(nullptr);
152 }
