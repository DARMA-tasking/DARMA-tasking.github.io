73 template <typename T, typename... Deps, typename... Cons>
74 registry::AutoHandlerType ComponentPack::registerComponent(
75  T** ref, typename BaseComponent::DepsPack<Deps...>, Cons&&... cons
76 ) {
77  ComponentRegistry::dependsOn<T, Deps...>();
78  auto idx = registry::makeIdx<T>();
79 
80  if (registered_set_.find(idx) == registered_set_.end()) {
81  registered_set_.insert(idx);
82  registered_components_.push_back(idx);
83  }
84 
85  // Be very careful here. Don't use std::make_tuple or we loose l-val refs
86  // (which initially was a "feature", but I realized later was probably not
87  // ideal)
88  std::tuple<Cons...> cons_tuple{std::forward<Cons>(cons)...};
89 
90  auto fn = makeCallable(
91  [ref, this, tup = std::move(cons_tuple)]() mutable {
92  auto ptr = tupleCons<T>(std::move(tup));
93 
94  vt_debug_print(
95  runtime, node,
96  "ComponentPack: constructed component={}, pollable={}\n",
97  ptr->name(), ptr->pollable()
98  );
99 
100  // Set the reference for access outside
101  if (ref != nullptr) {
102  *ref = ptr.get();
103  }
104 
105  // Add to pollable for the progress function to be triggered
106  if (ptr->pollable()) {
107  pollable_components_.emplace_back(ptr.get());
108  }
109 
110  ptr->initialize();
111  live_components_.emplace_back(std::move(ptr));
112  }
113  );
114 
115  construct_components_.emplace(
116  std::piecewise_construct,
117  std::forward_as_tuple(idx),
118  std::forward_as_tuple(std::move(fn))
119  );
120  return idx;
121 }

129 template <typename T>
130 std::unique_ptr<T>
131 ComponentPack::extractComponent(std::string const& name) {
132  for (auto iter = live_components_.begin(); iter != live_components_.end(); ++iter) {
133  if ((*iter) != nullptr and (*iter)->name() == name) {
134  T* raw = static_cast<T*>((*iter).release());
135  *iter = nullptr;
136  return std::unique_ptr<T>(raw);
137  }
138  }
139  return std::unique_ptr<T>(nullptr);
140 }
