52 template <typename T, typename... Deps, typename... Cons>
53 registry::AutoHandlerType ComponentPack::registerComponent(
54  T** ref, typename BaseComponent::DepsPack<Deps...>, Cons&&... cons
55 ) {
56  ComponentRegistry::dependsOn<T, Deps...>();
57  auto idx = registry::makeIdx<T>();
58 
59  if (registered_set_.find(idx) == registered_set_.end()) {
60  registered_set_.insert(idx);
61  registered_components_.push_back(idx);
62  }
63 
64  construct_components_.emplace(
65  std::piecewise_construct,
66  std::forward_as_tuple(idx),
67  std::forward_as_tuple(
68  [ref,this,cons...]() mutable {
69  auto ptr = T::template staticInit<Cons...>(std::forward<Cons>(cons)...);
70 
71  debug_print(
72  runtime, node,
73  "ComponentPack: constructed component={}, pollable={}\n",
74  ptr->name(), ptr->pollable()
75  );
76 
77  // Set the reference for access outside
78  if (ref != nullptr) {
79  *ref = ptr.get();
80  }
81 
82  // Add to pollable for the progress function to be triggered
83  if (ptr->pollable()) {
84  pollable_components_.emplace_back(ptr.get());
85  }
86 
87  ptr->initialize();
88  live_components_.emplace_back(std::move(ptr));
89  }
90  )
91  );
92  return idx;
93 }
