114 template <typename T>
115 detail::DiagnosticValue<T>* Diagnostic::registerDiagnostic(
116  std::string const& key, std::string const& desc, DiagnosticUpdate update,
117  DiagnosticUnit unit, DiagnosticTypeEnum type, T initial_value
118 ) {
119 # if vt_check_enabled(diagnostics)
120  // If diagnostics are runtime disabled, return a null value
121  if (not theConfig()->vt_diag_enable) {
122  return nullptr;
123  }
124 
125  vtAssert(values_.find(key) == values_.end(), "Key must not exist");
126  values_.emplace(
127  std::piecewise_construct,
128  std::forward_as_tuple(key),
129  std::forward_as_tuple(
130  std::make_unique<detail::DiagnosticValue<T>>(
131  key, desc, update, unit, type, initial_value
132  )
133  )
134  );
135  auto iter = values_.find(key);
136  vtAssert(iter != values_.end(), "Diagnostic key must exist");
137  return static_cast<detail::DiagnosticValue<T>*>(iter->second.get());
138 # else
139  debug::useVars(key, desc, update, unit, type, initial_value);
140  return nullptr;
141 # endif
142 }
