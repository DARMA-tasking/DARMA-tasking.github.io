157 template <typename MsgT>
158 ActiveMessenger::PendingSendType ActiveMessenger::sendMsgCopyableImpl(
159  NodeType dest,
160  HandlerType han,
161  MsgSharedPtr<MsgT>& msg,
162  ByteType msg_size,
163  TagType tag
164 ) {
165  static_assert(
166  vt::messaging::is_byte_copyable_t<MsgT>::value,
167  "Message sent without serialization must be byte-copyable."
168  );

171  "Message sent without serialization must be trivially destructible."
172  );
173 
174  MsgT* rawMsg = msg.get();
175 
176  bool is_term = envelopeIsTerm(rawMsg->env);
177  const bool is_bcast = dest == broadcast_dest;
178 
179  if (!is_term || vt_check_enabled(print_term_msgs)) {
180  vt_debug_print(

185  print_ptr(rawMsg), typeid(MsgT).name()
186  );
187  }
188 
189  if (is_bcast) {
190  dest = theContext()->getNode();
191  }
192  if (msg_size == msgsize_not_specified) {
193  msg_size = sizeof(MsgT);
194  }
195  if (tag != no_tag) {
196  envelopeSetTag(rawMsg->env, tag);
197  }
198  envelopeSetup(rawMsg->env, dest, han);
199  setupEpochMsg(rawMsg);
200 
201  envelopeSetIsLocked(rawMsg->env, true);
202

495  /*
496  * setGlobalEpoch() is a shortcut for both pushing and popping epochs on the

520 inline void ActiveMessenger::pushEpoch(EpochType const& epoch) {
521  /*
522  * pushEpoch(epoch) pushes any epoch onto the local stack iff epoch !=
523  * no_epoch; the epoch stack includes all locally pushed epochs and the
524  * current contexts pushed, transitively causally related active message
525  * handlers.
526  */

531  );
532  if (epoch != no_epoch) {
533  epoch_stack_.push(epoch);

537 inline EpochType ActiveMessenger::popEpoch(EpochType const& epoch) {
538  /*
539  * popEpoch(epoch) shall remove the top entry from epoch_size_, iif the size
540  * is non-zero and the `epoch' passed, if `epoch != no_epoch', is equal to the
541  * top of the `epoch_stack_.top()'; else, it shall remove any entry from the
542  * top of the stack.
543  */
544  auto const& non_zero = epoch_stack_.size() > 0;
545  vtAssertExprInfo(
546  non_zero and (epoch_stack_.top() == epoch or epoch == no_epoch),
547  epoch, non_zero, epoch_stack_.top()
548  );
549  if (epoch == no_epoch) {
550  return non_zero ? epoch_stack_.pop(),epoch_stack_.top() : no_epoch;
551  } else {
552  return non_zero && epoch == epoch_stack_.top() ?
553  epoch_stack_.pop(),epoch :
554  no_epoch;
555  }
556 }

562 template <typename MsgT>
563 inline EpochType ActiveMessenger::getEpochContextMsg(MsgT* msg) {
564  auto const is_epoch = envelopeIsEpochType(msg->env);
565  auto const any_epoch = term::any_epoch_sentinel;
566 
567  if (is_epoch) {
568  auto const msg_epoch = envelopeGetEpoch(msg->env);
569 
570  // Prefer the epoch on the msg before the current handler epoch
571  if (msg_epoch != no_epoch and msg_epoch != any_epoch) {
572  // It has a valid (possibly global) epoch, use it
573  return msg_epoch;
574  } else {
575  // Otherwise, use the active messenger's current epoch on the stack
576  auto const ctx_epoch = getEpoch();
577  vtAssertInfo(
578  ctx_epoch != no_epoch, "Must have a valid epoch here",
579  ctx_epoch, msg_epoch, no_epoch, any_epoch
580  );
581  return ctx_epoch;
582  }
583  } else {
584  return any_epoch;
585  }
586 }
