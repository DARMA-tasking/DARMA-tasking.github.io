129 struct LoadSummary {
130  TimeType whole_phase_load = 0.0;
131  std::vector<TimeType> subphase_loads = {};
132 
133  TimeType get(PhaseOffset when) const
134  {
135  if (when.subphase == PhaseOffset::WHOLE_PHASE)
136  return whole_phase_load;
137  else
138  return subphase_loads.at(when.subphase);
139  }
140 
141  void operator += (const LoadSummary& rhs) {
142  vtAssert(subphase_loads.size() == rhs.subphase_loads.size(),
143  "Subphase counts must match");
144 
145  whole_phase_load += rhs.whole_phase_load;
146  for (size_t i = 0; i < subphase_loads.size(); ++i)
147  subphase_loads[i] += rhs.subphase_loads[i];
148  }
149 
150  template <typename SerializerT>
151  void serialize(SerializerT& s) {
152  s | whole_phase_load
153  | subphase_loads;
154  }
155 };
