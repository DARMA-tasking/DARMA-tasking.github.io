83 struct LoadSummary {
84  TimeType whole_phase_load = 0.0;
85  std::vector<TimeType> subphase_loads = {};
86 
87  TimeType get(PhaseOffset when) const
88  {
89  if (when.subphase == PhaseOffset::WHOLE_PHASE)
90  return whole_phase_load;
91  else
92  return subphase_loads.at(when.subphase);
93  }
94 
95  void operator += (const LoadSummary& rhs) {
96  vtAssert(subphase_loads.size() == rhs.subphase_loads.size(),
97  "Subphase counts must match");
98 
99  whole_phase_load += rhs.whole_phase_load;
100  for (size_t i = 0; i < subphase_loads.size(); ++i)
101  subphase_loads[i] += rhs.subphase_loads[i];
102  }
103 
104  template <typename SerializerT>
105  void serialize(SerializerT& s) {
106  s | whole_phase_load
107  | subphase_loads;
108  }
109 };
