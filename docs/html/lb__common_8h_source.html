86 struct LoadSummary {
87  TimeType whole_phase_load = 0.0;
88  std::vector<TimeType> subphase_loads = {};
89 
90  TimeType get(PhaseOffset when) const
91  {
92  if (when.subphase == PhaseOffset::WHOLE_PHASE)
93  return whole_phase_load;
94  else
95  return subphase_loads.at(when.subphase);
96  }
97 
98  void operator += (const LoadSummary& rhs) {
99  vtAssert(subphase_loads.size() == rhs.subphase_loads.size(),
100  "Subphase counts must match");
101 
102  whole_phase_load += rhs.whole_phase_load;
103  for (size_t i = 0; i < subphase_loads.size(); ++i)
104  subphase_loads[i] += rhs.subphase_loads[i];
105  }
106 
107  template <typename SerializerT>
108  void serialize(SerializerT& s) {
109  s | whole_phase_load
110  | subphase_loads;
111  }
112 };
