126 template <typename IndexElmType, typename PhysicalType>
127 inline NodeType blockMapDenseFlatIndex(
128  IndexElmType* flat_idx_ptr, IndexElmType* num_elems_ptr,
129  PhysicalType num_resources
130 ) {
131  IndexElmType flat_idx = *flat_idx_ptr;
132  IndexElmType num_elems = *num_elems_ptr;
133 
134  double const& elms_as_dbl = static_cast<double>(num_elems);
135  double const& res_as_dbl = static_cast<double>(num_resources);
136  double const& bin_floor_dbl = std::floor(elms_as_dbl / res_as_dbl);
137  IndexElmType const& bin_size_floor = static_cast<IndexElmType>(bin_floor_dbl);
138  IndexElmType const& rem_elms = num_elems % num_resources;
139  IndexElmType const& num_first_set = rem_elms * (bin_size_floor + 1);
140 
141  if (flat_idx < num_first_set) {
142  return (flat_idx / (bin_size_floor + 1));
143  } else if (flat_idx < num_elems) {
144  return (rem_elms + (flat_idx - num_first_set) / bin_size_floor);
145  } else {
146  return flat_idx % num_resources;
147  }
148 }

150 template <typename Idx, index::NumDimensionsType ndim>
151 Idx linearizeDenseIndexColMajor(
152  DenseIndex <Idx, ndim> *idx, DenseIndex <Idx, ndim> *max_idx
153 ) {
154  // @todo: Do we have a defined behaviour when index is out of max_idx?
155  // @todo: This might be useful mechanism to express a boundary condition
156 
157  auto const& idx_ = *idx;
158  auto const& max_idx_ = *max_idx;
159 
160  for (size_t dim = 0; dim < ndim; dim++) {
161  vtAssert(idx_[dim] < max_idx_[dim], "Out of range index!");
162  }
163 
164  Idx val = 0;
165  Idx dim_size = 1;
166  for (auto i = ndim - 1; i >= 0; i--) {
167  val += dim_size * idx_[i];
168  dim_size *= max_idx_[i];
169  }
170  return val;
171 }
