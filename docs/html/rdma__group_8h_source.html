83  auto const& ret = map.elm_map(lo, num_total_elems, num_blocks);
84  blk_lo = std::get<0>(ret);
85  elm_hi = std::get<2>(ret);
86  cur_node = map.block_map(blk_lo, num_blocks);
87 
88  walk(cur_node, ret, lo, std::min(elm_hi,hi));
89 
90  lo = elm_hi;
91  }
92  }
