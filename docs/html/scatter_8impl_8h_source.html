59 void Scatter::scatter(
60  std::size_t const& total_size, std::size_t const& max_proc_size,
61  FuncSizeType size_fn, FuncDataType data_fn
62 ) {
63  auto const& num_nodes = theContext()->getNumNodes();
64  auto const& elm_size = max_proc_size;
65  auto const& combined_size = num_nodes * elm_size;
66  auto scatter_msg = makeMessageSz<ScatterMsg>(
67  combined_size, combined_size, elm_size
68  );
69  vtAssert(total_size == combined_size, "Sizes must be consistent");
70  auto ptr = reinterpret_cast<char*>(scatter_msg.get()) + sizeof(ScatterMsg);
71 #if backend_check_enabled(memory_pool)
72  auto remaining_size = thePool()->remainingSize(
73  reinterpret_cast<void*>(scatter_msg.get())
74  );
75  vtAssertInfo(
76  remaining_size >= combined_size,
77  "Remaining size must be sufficient",
78  total_size, combined_size, remaining_size, elm_size
79  );
80 #else
81  auto remaining_size = elm_size;
82 #endif
83  debug_print(
84  scatter, node,
85  "Scatter::scatter: total_size={}, elm_size={}, ScatterMsg={}, msg-ptr={}, "
86  "ptr={}, remaining_size={}\n",
87  total_size, elm_size, sizeof(ScatterMsg), print_ptr(scatter_msg.get()),
88  print_ptr(ptr), remaining_size
89  );
90  auto const& root_node = 0;
91  auto nptr = applyScatterRecur(root_node, ptr, elm_size, size_fn, data_fn);
92  debug_print(
93  scatter, node,
94  "Scatter::scatter: incremented size={}\n",
95  nptr-ptr
96  );
97  vtAssert(nptr == ptr + combined_size, "nptr must match size");
98  auto const& handler = auto_registry::makeAutoHandler<MessageT,f>(nullptr);
99  auto const& this_node = theContext()->getNode();
100  scatter_msg->user_han = handler;
101  if (this_node != root_node) {
102  theMsg()->sendMsgSz<ScatterMsg,scatterHandler>(
103  root_node, scatter_msg.get(), sizeof(ScatterMsg) + combined_size
104  );
105  } else {
106  scatterIn(scatter_msg.get());
107  }
108 }
