72 template <typename ColT>
73 /*static*/ void ElementStats::syncNextPhase(PhaseMsg<ColT>* msg, ColT* col) {
74  auto& stats = col->stats_;
75 
76  debug_print(
77  lb, node,
78  "ElementStats: syncNextPhase ({}) (idx={}): stats.getPhase()={}, "
79  "msg->getPhase()={}\n",
80  print_ptr(col), col->getIndex(), stats.getPhase(), msg->getPhase()
81  );
82 
83  vtAssert(stats.getPhase() == msg->getPhase(), "Phases must match");
84  stats.updatePhase(1);
85 
86  auto const& cur_phase = msg->getPhase();
87  auto const& proxy = col->getCollectionProxy();
88  auto const& untyped_proxy = col->getProxy();
89  auto const& total_load = stats.getLoad(cur_phase);
90  auto const& comm = stats.getComm(cur_phase);
91  auto const& idx = col->getIndex();
92  auto const& elm_proxy = proxy[idx];
93 
94  ProcStats::addProcStats<ColT>(elm_proxy, col, cur_phase, total_load, comm);
95 
96  auto const before_ready = theCollection()->numReadyCollections();
97  theCollection()->makeCollectionReady(untyped_proxy);
98  auto const after_ready = theCollection()->numReadyCollections();
99  auto const ready = theCollection()->readyNextPhase();
100 
101  debug_print(
102  lb, node,
103  "ElementStats: syncNextPhase: before_ready={}, after_ready={}, ready={}\n",
104  before_ready, after_ready, ready
105  );
106 
107  using MsgType = InvokeReduceMsg;
108 
109  auto lb_man = LBManager::getProxy();
110 
111  auto const single_node = theContext()->getNumNodes() == 1;
112  auto const lb = lb_man.get()->decideLBToRun(cur_phase);
113  bool const must_run_lb = lb != LBType::NoLB and not single_node;
114  auto const num_collections = theCollection()->numCollections<>();
115  auto const do_sync = msg->doSync();
116  auto nmsg = makeMessage<MsgType>(cur_phase,lb,msg->manual(),num_collections);
117 
118  if (must_run_lb) {
119  auto cb = theCB()->makeBcast<LBManager,MsgType,&LBManager::sysLB<MsgType>>(lb_man);
120  proxy.reduce(nmsg.get(),cb);
121  } else {
122 
123  // Preemptively release the element directly, doing cleanup later after a
124  // collection reduction. This allows work to start early while still
125  // releasing the node-level LB continuations needed for cleanup
126  if (lb == LBType::NoLB and not do_sync) {
127  theCollection()->elmFinishedLB(elm_proxy,cur_phase);
128  }
129 
130  auto cb = theCB()->makeBcast<LBManager,MsgType,&LBManager::sysReleaseLB<MsgType>>(lb_man);
131  proxy.reduce(nmsg.get(),cb);
132  }
133 }
