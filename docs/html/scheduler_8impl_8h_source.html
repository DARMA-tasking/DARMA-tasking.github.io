54 void runInEpoch(EpochType ep, Callable&& fn) {
55  theSched()->triggerEvent(sched::SchedulerEvent::PendingSchedulerLoop);
56 
57  theMsg()->pushEpoch(ep);
58  fn();
59  theMsg()->popEpoch(ep);
60  theTerm()->finishedEpoch(ep);
61  runSchedulerThrough(ep);
62 }

98 template <typename MsgT, typename RunT>
99 void Scheduler::enqueue(MsgT* msg, RunT r) {
100  bool const is_term = envelopeIsTerm(msg->env);
101 
102  if (is_term) {
103  num_term_msgs_++;
104  }
105 
106 # if vt_check_enabled(priorities)
107  auto priority = envelopeGetPriority(msg->env);
108  work_queue_.emplace(UnitType(is_term, std::move(r), priority));
109 # else
110  work_queue_.emplace(UnitType(is_term, std::move(r)));
111 # endif
112 }

114 template <typename MsgT, typename RunT>
115 void Scheduler::enqueue(MsgSharedPtr<MsgT> const& msg, RunT r) {
116 # if vt_check_enabled(priorities)
117  //
118  // Assume that MsgSharedPtr<MsgT> is already captured in the action.
119  //
120  // To speed this up, in the future, we could have a pure message queue that
121  // could be dispatched directly based on type/state-bits
122  //
123  enqueue<MsgT>(msg.get(), std::move(r));
124 #else
125  bool const is_term = envelopeIsTerm(msg->env);
126  enqueue<RunT>(is_term, r);
127 #endif
128 }
