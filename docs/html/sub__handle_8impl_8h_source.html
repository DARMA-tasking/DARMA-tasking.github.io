66 template <typename T, HandleEnum E, typename IndexT>
67 void SubHandle<T,E,IndexT>::makeSubHandles(bool initial) {
68  if (ordered_opt_) {
69  auto this_node = theContext()->getNode();
70  auto num_nodes = theContext()->getNumNodes();
71  auto const& map_fn = auto_registry::getHandlerMap(map_han_);
72  range_.foreach ([&](IndexT cur_idx) {
73  auto home_node = map_fn->dispatch(&cur_idx, &range_, num_nodes);
74  auto iter = sub_handles_staged_.find(cur_idx);
75  if (home_node == this_node and iter != sub_handles_staged_.end()) {
76  stageLocalIndex(iter->first, iter->second);
77  }
78  });
79  } else {
80  for (auto&& h : sub_handles_staged_) {
81  stageLocalIndex(h.first, h.second);
82  }
83  }
84  sub_handles_staged_.clear();
85 
86  auto const total = totalLocalCount();
87  auto const num_local = sub_handles_.size();
88  vt_debug_print(
89  terse, rdma,
90  "total={}, num_local={}, is_migratable_={}\n",
91  total, num_local, is_migratable_
92  );
93  for (uint64_t i = 0; i < sub_prefix_.size(); i++) {
94  vt_debug_print(
95  verbose, rdma,
96  "i={} prefix={} layout={}\n",
97  i, sub_prefix_[i], sub_layout_[i]
98  );
99  }
100  data_handle_ = proxy_.template makeHandleRDMA<T>(total, false);
101  waitForHandleReady(data_handle_);
102  if (initial) {
103  // Handle case when the local size is zero
104  auto loc_len =
105  num_local > 0 ? (is_migratable_ ? num_local * 4 : (num_local + 1) * 2) : 0;
106  vt_debug_print(
107  verbose, rdma,
108  "total={}, num_local={}, is_migratable_={}, loc_len={}\n",
109  total, num_local, is_migratable_, loc_len
110  );
111  loc_handle_ = proxy_.template makeHandleRDMA<uint64_t>(loc_len, false);
112  waitForHandleReady(loc_handle_);
113  vtAssertExpr(sub_prefix_.size() == num_local);
114  vtAssertExpr(sub_layout_.size() == num_local);
115  if (loc_len > 0) {
116  if (not is_migratable_) {
117  loc_handle_.modifyExclusive([&](uint64_t* t, std::size_t) {
118  uint64_t i = 0;
119  for (i = 0; i < num_local * 2; i += 2) {
120  t[i+0] = linearize(sub_layout_[i/2]);
121  t[i+1] = sub_prefix_[i/2];
122  }
123  t[i+0] = 0;
124  t[i+1] = sub_handles_[sub_layout_[sub_layout_.size()-1]].count_ + sub_prefix_[(i-2)/2];
125  });
126  } else {
127  auto this_node = theContext()->getNode();
128  loc_handle_.modifyExclusive([&](uint64_t* t, std::size_t) {
129  uint64_t i = 0;
130  for (i = 0; i < num_local * 4; i += 4) {
131  t[i+0] = linearize(sub_layout_[i/4]);
132  t[i+1] = sub_prefix_[i/4];
133  t[i+2] = this_node;
134  t[i+3] = sub_handles_[sub_layout_[i/4]].count_;
135  }
136  });
137  }
138  }
139  }
140  ready_ = true;
141 }

162 template <typename T, HandleEnum E, typename IndexT>
163 int SubHandle<T, E, IndexT>::getOrderedOffset(IndexT idx, NodeType home_node) {
164  auto iter = ordered_local_offset_.find(idx);
165  int found_offset = 0;
166  if (iter == ordered_local_offset_.end()) {
167  auto num_nodes = theContext()->getNumNodes();
168  auto const& map_fn = auto_registry::getHandlerMap(map_han_);
169  int offset = 0;
170  range_.foreach ([&](IndexT cur_idx) {
171  auto map_node = map_fn->dispatch(&cur_idx, &range_, num_nodes);
172  if (home_node == map_node) {
173  if (cur_idx == idx) {
174  found_offset = offset;
175  }
176  offset++;
177  }
178  });
179  ordered_local_offset_[idx] = found_offset;
180  } else {
181  found_offset = iter->second;
182  }
183  return found_offset;
184 }

186 template <typename T, HandleEnum E, typename IndexT>
187 void SubHandle<T,E,IndexT>::updateInfo(
188  IndexT const& idx, IndexInfo info, NodeType home
189 ) {
190  vtAssertExpr(not ordered_opt_);
191  vtAssertExpr(is_migratable_);
192  auto this_node = theContext()->getNode();
193  auto home_size = loc_handle_.getCount(home);
194  vt_debug_print(
195  normal, rdma,
196  "updateInfo: idx={}, this_node={}, home={}, home_size={}\n",
197  idx, this_node, home, home_size
198  );
199  auto lin_idx = linearize(idx);
200  auto ptr = std::make_unique<uint64_t[]>(home_size);
201  // These can probably be relaxed to Shared locks because lin_idx is never
202  // modified
203  loc_handle_.get(home, ptr.get(), home_size, 0, Lock::Exclusive);
204  for (uint64_t i = 0; i < home_size; i += 4) {
205  vt_debug_print(
206  verbose, rdma,
207  "updateInfo: idx={}, node={}, home={}, ptr[{}]={},{},{},{}\n",
208  idx, this_node, home, i, ptr[i+0], ptr[i+1], ptr[i+2], ptr[i+3]
209  );
210  if (ptr[i] == static_cast<uint64_t>(lin_idx)) {
211  vt_debug_print(
212  verbose, rdma,
213  "updateInfo: idx={}, node={}, home={}, "
214  "ptr[{}]={},{},{},{}\n",
215  idx, this_node, home, i,
216  ptr[i+0], ptr[i+1], ptr[i+2], ptr[i+3]
217  );
218  int offset = static_cast<int>(i);
219  auto pptr = std::make_unique<uint64_t[]>(2);
220  pptr[0] = info.getOffset();
221  pptr[1] = info.getNode();
222  loc_handle_.put(home, pptr.get(), 2, offset + 1, Lock::Exclusive);
223  }
224  }

227 template <typename T, HandleEnum E, typename IndexT>
228 IndexInfo SubHandle<T,E,IndexT>::fetchInfo(IndexT const& idx) {
229  auto this_node = theContext()->getNode();
230  auto home_node = getHomeNode(idx);
231  if (ordered_opt_) {
232  auto offset = getOrderedOffset(idx, home_node);
233  if (not is_migratable_) {
234  auto ptr = std::make_unique<uint64_t[]>(4);
235  loc_handle_.get(home_node, ptr.get(), 4, offset*2, Lock::Exclusive);
236  auto lin_idx = linearize(idx);
237  vt_debug_print(
238  verbose, rdma,
239  "fetchInfo: ordered: offset={}, idx={}, node={}, home={}, "
240  "ptr={},{}, {},{}\n",
241  offset, idx, this_node, home_node, ptr[0], ptr[1], ptr[2], ptr[3]
242  );
243  vtAssertExpr(ptr[0] == static_cast<uint64_t>(lin_idx));
244  return IndexInfo(home_node, ptr[1], ptr[3]-ptr[1]);
245  } else {
246  auto ptr = std::make_unique<uint64_t[]>(4);
247  loc_handle_.get(home_node, ptr.get(), 4, offset*4, Lock::Exclusive);
248  auto lin_idx = linearize(idx);
249  vt_debug_print(
250  verbose, rdma,
251  "fetchInfo: ordered: offset={}, idx={}, node={}, home={},"
252  " ptr={},{},{},{}\n",
253  offset, idx, this_node, home_node, ptr[0], ptr[1], ptr[2], ptr[3]
254  );
255  vtAssertExpr(ptr[0] == static_cast<uint64_t>(lin_idx));
256  return IndexInfo(ptr[2], ptr[1], ptr[3]);
257  }
258  } else {
259  auto home_size = loc_handle_.getCount(home_node);
260  vt_debug_print(
261  normal, rdma,
262  "fetchInfo: idx={}, this_node={}, home_node={}, home_size={}\n",
263  idx, this_node, home_node, home_size
264  );
265  auto lin_idx = linearize(idx);
266  auto ptr = std::make_unique<uint64_t[]>(home_size);
267  loc_handle_.get(home_node, ptr.get(), home_size, 0, Lock::Exclusive);
268  if (not is_migratable_) {
269  for (uint64_t i = 0; i < home_size; i += 2) {
270  vt_debug_print(
271  verbose, rdma,
272  "fetchInfo: idx={}, node={}, home={}, ptr[{}]={},{}\n",
273  idx, this_node, home_node, i, ptr[i+0], ptr[i+1]
274  );
275  if (ptr[i] == static_cast<uint64_t>(lin_idx)) {
276  vt_debug_print(
277  verbose, rdma,
278  "fetchInfo: idx={}, node={}, home={}, ptr[{}]={},{}, {},{}\n",
279  idx, this_node, home_node, i, ptr[i+0], ptr[i+1],
280  ptr[i+2], ptr[i+3]
281  );
282  return IndexInfo(home_node, ptr[i+1], ptr[i+3]-ptr[i+1]);
283  }
284  }
285  } else {
286  for (uint64_t i = 0; i < home_size; i += 4) {
287  vt_debug_print(
288  verbose, rdma,
289  "fetchInfo: idx={}, node={}, home={}, ptr[{}]={},{},{},{}\n",
290  idx, this_node, home_node, i, ptr[i+0], ptr[i+1], ptr[i+2], ptr[i+3]
291  );
292  if (ptr[i] == static_cast<uint64_t>(lin_idx)) {
293  vt_debug_print(
294  verbose, rdma,
295  "fetchInfo: idx={}, node={}, home={}, "
296  "ptr[{}]={},{},{},{}\n",
297  idx, this_node, home_node, i,
298  ptr[i+0], ptr[i+1], ptr[i+2], ptr[i+3]
299  );
300  return IndexInfo(ptr[i+2], ptr[i+1], ptr[i+3]);
301  }
302  }
303  }
304  }
305  vtAssert(false, "Could not find location info");
306  return IndexInfo{};
307 }

309 template <typename T, HandleEnum E, typename IndexT>
310 IndexInfo SubHandle<T,E,IndexT>::resolveLocation(IndexT const& idx) {
311  auto iter = sub_handles_.find(idx);
312  if (iter != sub_handles_.end()) {
313  auto const this_node = vt::theContext()->getNode();
314  return IndexInfo(this_node, iter->second.offset_, iter->second.count_);
315  } else {
316  if (cache_.hasIndex(idx)) {
317  return cache_.getInfo(idx);
318  } else {
319  auto info = fetchInfo(idx);
320  cache_.saveInfo(idx, info);
321  return info;
322  }
323  }

433 template <typename T, HandleEnum E, typename IndexT>
434 void SubHandle<T,E,IndexT>::stageLocalIndex(
435  IndexT index, uint64_t count
436 ) {
437  vt_debug_print(
438  verbose, rdma,
439  "addLocalIndex: idx={}, count={}, range={}\n",
440  index, count, range_
441  );
442  sub_layout_.push_back(index);
443  if (sub_prefix_.size() > 0) {
444  vtAssertExpr(sub_layout_.size() >= 2);
445  // Compute a prefix as elements are added
446  auto last_idx = sub_layout_[sub_layout_.size()-2];
447  auto last_count_iter = sub_handles_.find(last_idx);
448  vtAssertExpr(last_count_iter != sub_handles_.end());
449  auto last_count = last_count_iter->second.count_;
450  sub_prefix_.push_back(sub_prefix_[sub_prefix_.size()-1] + last_count);
451  } else {
452  sub_prefix_.push_back(0);
453  }
454  auto const offset = sub_prefix_[sub_prefix_.size()-1];
455  sub_handles_[index] = SubInfo(count, offset);
456  if (uniform_size_) {
457  count_if_uniform_ = static_cast<std::size_t>(count);
458  }
459 }

577  bool local_changed = migrate_in_.size() != 0 or migrate_out_.size() != 0;
578  auto cb = theCB()->makeBcast<
579  ThisType,impl::ReduceLBMsg,&ThisType::checkChanged
580  >(proxy_);
581  auto epoch = theTerm()->makeEpochCollective("SubHandle::afterLB");
582  theMsg()->pushEpoch(epoch);
583  auto msg = makeMessage<impl::ReduceLBMsg>(local_changed);
584  proxy_.template reduce<collective::OrOp<bool>>(msg.get(),cb);
585  theMsg()->popEpoch(epoch);
586  theTerm()->finishedEpoch(epoch);
587 
588  runSchedulerThrough(epoch);
589 }

591 template <typename T, HandleEnum E, typename IndexT>
592 void SubHandle<T,E,IndexT>::checkChanged(impl::ReduceLBMsg* msg) {
593  auto global_changed = msg->getVal();
594 
595  vt_debug_print(
596  normal, rdma,
597  "checkChanged: global_changed={}\n", global_changed
598  );
599 
600  // Must re-configure all windows
601  if (global_changed) {
602  std::vector<IndexT> cur_handles;
603  for (auto&& h : sub_handles_) {
604  cur_handles.push_back(h.first);
605  }
606 
607  vt_debug_print(
608  verbose, rdma,
609  "checkChanged: cur_handles={}, in={}, out={}\n",
610  cur_handles.size(), migrate_in_.size(), migrate_out_.size()
611  );
612 
613  std::vector<IndexT> all_handles;
614  // Union current with migrated in
615  std::set_union(
616  cur_handles.begin(), cur_handles.end(),
617  migrate_in_.begin(), migrate_in_.end(),
618  std::back_inserter(all_handles)
619  );
620 
621  vt_debug_print(
622  verbose, rdma,
623  "checkChanged: all_handles={}, in={}, out={}\n",
624  all_handles.size(), migrate_in_.size(), migrate_out_.size()
625  );
626 
627  std::vector<IndexT> new_handles;
628  std::sort(all_handles.begin(), all_handles.end());
629  std::sort(migrate_out_.begin(), migrate_out_.end());
630  // Minus off the ones migrated out
631  std::set_difference(
632  all_handles.begin(), all_handles.end(),
633  migrate_out_.begin(), migrate_out_.end(),
634  std::inserter(new_handles, new_handles.begin())
635  );
636 
637  vt_debug_print(
638  verbose, rdma,
639  "checkChanged: new_handles={}, in={}, out={}\n",
640  new_handles.size(), migrate_in_.size(), migrate_out_.size()
641  );
642 
643  vt_debug_print(
644  verbose, rdma,
645  "checkChanged: in.size()={}, out.size()={}\n",
646  migrate_in_.size(), migrate_out_.size()
647  );
648 
649  // Now, we need the sizes to re-create the handles that belong here
650  std::unordered_map<IndexT, std::size_t> new_handles_sized;
651  for (auto&& h : new_handles) {
652  vt_debug_print(
653  verbose, rdma,
654  "checkChanged: fetching size: idx={}\n", h
655  );
656 
657  // Fetch the size, which may be remote if it migrated
658  new_handles_sized[h] = getCount(h);
659  }
660 
661  vt_debug_print(
662  verbose, rdma,
663  "checkChanged: fetched all sizes\n"
664  );
665 
666  // Fetch all the data for the new handles into buffers
667  // @todo: possibly we could do a local memcpy for the purely local
668  // transfers, but it would be very difficult
669  std::unordered_map<IndexT, std::unique_ptr<T[]>> idx_data;
670  std::vector<RequestHolder> reqs;
671  for (auto&& h : new_handles_sized) {
672  idx_data[h.first] = std::make_unique<T[]>(h.second);
673  reqs.emplace_back(
674  rget(h.first, vt::Lock::Shared, &idx_data[h.first][0], h.second, 0)
675  );
676  }
677 
678  // Wait for all the fetches to complete
679  for (auto&& r : reqs) {
680  r.wait();
681  }
682  reqs.clear();
683 
684  // All must finish before we start destroying the hold sub-handle info
685  theCollective()->barrier();
686 
687  // Clear out all the incremental building data structures
688  sub_handles_.clear();
689  sub_layout_.clear();
690  sub_prefix_.clear();
691  if (ordered_opt_) {
692  ordered_local_offset_.clear();
693  }
694 
695  // New we have everything to re-build the sub-handles
696  for (auto&& h : new_handles_sized) {
697  addLocalIndex(h.first, h.second);
698  }
699 
700  vt_debug_print(
701  verbose, rdma,
702  "checkChanged: new_handles_sized={}, staged={}\n",
703  new_handles_sized.size(), sub_handles_staged_.size()
704  );
705 
706  // Destroy the old data handle
707  proxy_.destroyHandleRDMA(data_handle_);
708 
709  // Make the new sub-handles, with the new local indices now mapped here
710  makeSubHandles(false);
711 
712  // Invalidate the cache, since handles have migrated and the cached
713  // locations may not be valid anymore
714  cache_.invalidate();
715 
716  // Make sure everyone has invalidated, as to not get bad values
717  theCollective()->barrier();
718 
719  // Update the locations for all the handles since they are been
720  // re-configured
721  for (auto&& m : new_handles_sized) {
722  auto home = getHomeNode(m.first);
723  // This resolveLocation better be a local operation!
724  auto info = resolveLocation(m.first);
725  updateInfo(m.first, info, home);
726  }
727 
728  // Make sure everyone has updated their location data
729  theCollective()->barrier();
730 
731  // Put all the old data where it goes
732  for (auto&& h : new_handles_sized) {
733  reqs.emplace_back(
734  rput(h.first, vt::Lock::Shared, &idx_data[h.first][0], h.second, 0)
735  );
736  }
737 
738  // Wait for all the puts to complete
739  for (auto&& r : reqs) {
740  r.wait();
741  }
742  reqs.clear();
743 
744  // Clear the migrate buffers
745  migrate_in_.clear();
746  migrate_out_.clear();
747 
748  // Make sure all the puts are globally complete before we exit this block
749  theCollective()->barrier();
750  }
751 }
