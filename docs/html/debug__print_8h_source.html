286 template <CatEnum cat, ModeEnum mod, typename Arg, typename... Args>
287 static inline void debugPrintImpl(NodeType node, Arg&& arg, Args&&... args) {
288  constexpr auto mask = ModeEnum::terse | ModeEnum::normal | ModeEnum::verbose;
289  constexpr auto level = mod & mask;
290  if (level <= vt::debug::preConfig()->vt_debug_level_val) {
291  auto user = fmt::format(std::forward<Arg>(arg),std::forward<Args>(args)...);
292  std::string debug_level = "";
293  if (level == ModeEnum::terse) {
294  debug_level = "(t)";
295  } else if (level == ModeEnum::normal) {
296  debug_level = "(n)";
297  } else if (level == ModeEnum::verbose) {
298  debug_level = "(v)";
299  }
300  fmt::print(
301  "{} {} {} {} {}",
302  vt_print_colorize,
303  vt_proc_print_colorize(node),
304  vt_print_colorize_impl(::vt::debug::yellow(), debug_level, ""),
305  vt_print_colorize_impl(
306  ::vt::debug::green(), PrettyPrintCat<cat>::print(), ":"
307  ),
308  user
309  );
310  bool const flush_enabled = vt_option_check_enabled(mod, ModeEnum::flush);
311  if (flush_enabled or vt::debug::preConfig()->alwaysFlush()) {
312  fflush(stdout);
313  }
314  }
315 }
