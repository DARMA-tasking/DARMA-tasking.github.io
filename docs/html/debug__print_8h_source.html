172 template <CatEnum cat, ModeEnum mod, typename Arg, typename... Args>
173 static inline void debugPrintImpl(NodeType node, Arg&& arg, Args&&... args) {
174  bool const verb = vt_option_check_enabled(mod, ModeEnum::verbose);
175  if ((verb and ::vt::arguments::ArgConfig::vt_debug_verbose) or not verb) {
176  auto user = fmt::format(std::forward<Arg>(arg),std::forward<Args>(args)...);
177  fmt::print(
178  "{} {} {} {}",
179  vt_print_colorize,
180  vt_proc_print_colorize(node),
181  vt_print_colorize_impl(::vt::debug::green(), PrettyPrintCat<cat>::print(), ":"),
182  user
183  );
184  if (vt_option_check_enabled(mod, ModeEnum::flush) or arguments::alwaysFlush()) {
185  fflush(stdout);
186  }
187  }
188 }

191 struct DebugPrintOp<cat, CtxEnum::node, mod> {
192  template <typename Arg, typename... Args>
193  void operator()(bool const rt_option, Arg&& arg, Args&&... args) {
