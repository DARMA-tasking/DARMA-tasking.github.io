181 template <CatEnum cat, ModeEnum mod, typename Arg, typename... Args>
182 static inline void debugPrintImpl(NodeType node, Arg&& arg, Args&&... args) {
183  constexpr auto mask = ModeEnum::terse | ModeEnum::normal | ModeEnum::verbose;
184  constexpr auto level = mod & mask;
185  if (level <= vt::debug::preConfig()->vt_debug_level_val) {
186  auto user = fmt::format(std::forward<Arg>(arg),std::forward<Args>(args)...);
187  std::string debug_level = "";
188  if (level == ModeEnum::terse) {
189  debug_level = "(t)";
190  } else if (level == ModeEnum::normal) {
191  debug_level = "(n)";
192  } else if (level == ModeEnum::verbose) {
193  debug_level = "(v)";
194  }
195  fmt::print(
196  "{} {} {} {} {}",
197  vt_print_colorize,
198  vt_proc_print_colorize(node),
199  vt_print_colorize_impl(::vt::debug::yellow(), debug_level, ""),
200  vt_print_colorize_impl(
201  ::vt::debug::green(), PrettyPrintCat<cat>::print(), ":"
202  ),
203  user
204  );
205  bool const flush_enabled = vt_option_check_enabled(mod, ModeEnum::flush);
206  if (flush_enabled or vt::debug::preConfig()->alwaysFlush()) {
207  fflush(stdout);
208  }
209  }
210 }
