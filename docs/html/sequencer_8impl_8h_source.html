329 template <typename SeqTag, template <typename> class SeqTrigger>
330 template <typename MessageT, ActiveTypedFnType<MessageT>* f>
331 void TaggedSequencer<SeqTag, SeqTrigger>::wait_on_trigger(
332  TagType const& tag, SeqActionType<MessageT> action
333 ) {
334  theTerm()->produce();
335 
336  assertValidContext();
337 
338  SeqType const seq_id = getSeqID();
339  SeqNodePtrType node = getNode(seq_id);
340 
341  vtAssert(node != nullptr, "Must have node from context");
342 
343  bool const seq_ready = node->isReady();
344 
345  vt_debug_print(
346  verbose, sequence,
347  "Sequencer: wait: tag={}: context seq id={}, node={}, blocked={}, "
348  "ready={}\n",
349  tag, seq_id, PRINT_SEQ_NODE_PTR(node),
350  print_bool(node->isBlockedNode()), print_bool(seq_ready)
351  );
352 
353  auto deferred_wait_action = [tag,action,node,seq_id]() -> bool {
354  bool const has_match = SeqStateMatcherType<MessageT,f>::hasMatchingMsg(tag);
355 
356  if (has_match) {
357  auto msg = SeqStateMatcherType<MessageT, f>::getMatchingMsg(tag);
358  action.runAction(msg);
359  }
360 
361  vt_debug_print(
362  verbose, sequence,
363  "Sequencer: {}: tag={}: node={}, has_match={}, "
364  "is_blocked={}\n",
365  has_match ? "wait ran *immediately*" : "wait registered", tag,
366  PRINT_SEQ_NODE_PTR(node), print_bool(has_match),
367  print_bool(node->isBlockedNode())
368  );
369 
370  bool const should_block = not has_match;
371 
372  node->setBlockedOnNode(eSeqConstructType::WaitConstruct, should_block);
373 
374  if (has_match) {
375  vt_debug_print(
376  verbose, sequence,
377  "Sequencer: activating next node: seq={}, node={}, blocked={}\n",
378  seq_id, PRINT_SEQ_NODE_PTR(node), print_bool(node->isBlockedNode())
379  );
380 
381  if (node->isParallel()) {
382  node->activate();
383  }
384  } else {
385  // buffer the action to wait for a matching message
386 
387  auto msg_recv_trigger = [node,seq_id,action,tag](MessageT* msg){
388  vt_debug_print(
389  verbose, sequence,
390  "Sequencer: msg_recv_trigger: seq={}, tag={}, node={}, blocked={}, "
391  "msg={}\n",
392  seq_id, tag, PRINT_SEQ_NODE_PTR(node),
393  print_bool(node->isBlockedNode()), print_ptr(msg)
394  );
395 
396  auto pmsg = promoteMsg(msg);
397 
398  action.runAction(pmsg, false);
399 
400  vtAssert(node != nullptr, "node must not be nullptr");
401 
402  node->setBlockedOnNode(eSeqConstructType::WaitConstruct, false);
403  node->activate();
404  };
405 
406  // buffer the action because there is not a matching message to trigger
407  // the message
408  SeqStateMatcherType<MessageT, f>::bufferUnmatchedAction(
409  msg_recv_trigger, seq_id, tag
410  );
411  }
412 
413  return should_block;
414  };
415 
416  bool should_suspend = false;
417 
418  if (seq_ready) {
419  // run it here, right now
420  bool const has_match = not deferred_wait_action();
421 
422  vt_debug_print(
423  verbose, sequence,
424  "Sequencer: executed wait: has_match={}: seq_id={}\n",
425  print_bool(has_match), seq_id
426  );
427  should_suspend = not has_match;
428  } else {
429  should_suspend = true;
430 
431  vt_debug_print(
432  verbose, sequence,
433  "Sequencer: deferring wait: seq_id={}\n", seq_id
434  );
435 
436  node->addSequencedClosure(deferred_wait_action);
437  }
438 
439  if (should_suspend and context_->isSuspendable()) {
440  vt_debug_print(
441  verbose, sequence,
442  "Sequencer: should suspend: seq_id={}, context suspendable={}\n",
443  seq_id, print_bool(context_->isSuspendable())
444  );
445 
446  context_->suspend();
447  }
448 }

546 template <typename SeqTag, template <typename> class SeqTrigger>
547 template <typename MessageT, ActiveTypedFnType<MessageT>* f>
548 /*static*/ void TaggedSequencer<SeqTag, SeqTrigger>::sequenceMsg(MessageT* msg) {
549  auto const& is_tag_type = envelopeIsTagType(msg->env);
550  TagType const& msg_tag = is_tag_type ? envelopeGetTag(msg->env) : no_tag;
551 
552  vt_debug_print(
553  verbose, sequence,
554  "sequenceMsg: arrived: msg={}, tag={}\n",
555  print_ptr(msg), msg_tag
556  );
557 
558  bool const has_match =
559  SeqStateMatcherType<MessageT, f>::hasMatchingAction(msg_tag);
560 
561  vt_debug_print(
562  verbose, sequence,
563  "sequenceMsg: arriving: msg={}, has_match={}, tag={}\n",
564  print_ptr(msg), print_bool(has_match), msg_tag
565  );
566 
567  // reference the arrival message to keep it alive past normal lifetime, in
568  // case it's buffered or put into the scheduler
569  auto pmsg = promoteMsg(msg);
570 
571  if (has_match) {
572  auto action = SeqStateMatcherType<MessageT, f>::getMatchingAction(msg_tag);
573  auto handle_msg_action = [this,action,pmsg]{
574  lookupContextExecute(action.seq_id, action.generateCallable(pmsg));
575  };
576 
577  // skip the work queue and directly execute
578  if (seq_skip_queue) {
579  handle_msg_action();
580  } else {
581  theSeq()->enqueue(handle_msg_action);
582  }
583  } else {
584  // nothing was found so the message must be buffered and wait an action
585  // being posted
586 
587  // buffer the unmatched messaged until a trigger is posted for it that
588  // matches
589  SeqStateMatcherType<MessageT, f>::bufferUnmatchedMessage(pmsg, msg_tag);
590  }
591 }
