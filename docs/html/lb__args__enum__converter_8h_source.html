95  std::string getString(T e) const {

105  vtAbort(err);
106  }
107  return it->second;
108  }

116  T getEnum(const std::string &s) const {

123  "option for enum '{}'",
124  par_name_, s, enum_type_
125  );
126  vtAbort(err);
127  }
128  return it->second;
129  }
