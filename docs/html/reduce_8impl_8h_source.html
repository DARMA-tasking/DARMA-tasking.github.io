107 template <typename MessageT, ActiveTypedFnType<MessageT>* f>
108 SequentialIDType Reduce::reduce(
109  NodeType root, MessageT* const msg, TagType tag, SequentialIDType seq,
110  ReduceNumType num_contrib, VirtualProxyType proxy, ObjGroupProxyType objgroup
111 ) {
112  if (group_ != default_group) {
113  envelopeSetGroup(msg->env, group_);
114  }
115  auto const han = auto_registry::makeAutoHandler<MessageT,f>(msg);
116  msg->combine_handler_ = han;
117  msg->reduce_tag_ = tag;
118  msg->reduce_root_ = root;
119  msg->reduce_proxy_ = proxy;
120  msg->reduce_objgroup_ = objgroup;
121  debug_print(
122  reduce, node,
123  "reduce: group={:x}, tag={}, seq={}, vrt={}, objgrp={}, contrib={}, "
124  "msg={}, ref={}\n",
125  group_, msg->reduce_tag_, msg->reduce_seq_, msg->reduce_proxy_,
126  msg->reduce_objgroup_, num_contrib, print_ptr(msg), envelopeGetRef(msg->env)
127  );
128  if (seq == no_seq_id) {
129  auto reduce_seq_lookup = std::make_tuple(proxy,tag,objgroup);
130  auto iter = next_seq_for_tag_.find(reduce_seq_lookup);
131  if (iter == next_seq_for_tag_.end()) {
132  next_seq_for_tag_.emplace(
133  std::piecewise_construct,
134  std::forward_as_tuple(reduce_seq_lookup),
135  std::forward_as_tuple(SequentialIDType{1})
136  );
137  iter = next_seq_for_tag_.find(reduce_seq_lookup);
138  }
139  vtAssert(iter != next_seq_for_tag_.end(), "Must exist now");
140  msg->reduce_seq_ = iter->second++;
141  } else {
142  msg->reduce_seq_ = seq;
143  }
144  reduceAddMsg<MessageT>(msg,true,num_contrib);
145  reduceNewMsg<MessageT>(msg);
146  return msg->reduce_seq_;
147 }

149 template <typename MessageT>
150 void Reduce::reduceAddMsg(
151  MessageT* msg, bool const local, ReduceNumType num_contrib
152 ) {
153  auto lookup = ReduceIdentifierType{
154  msg->reduce_tag_,
155  msg->reduce_seq_,
156  msg->reduce_proxy_,
157  msg->reduce_objgroup_
158  };
159 
160  auto exists = ReduceStateHolder<MessageT>::exists(group_,lookup);
161  if (not exists) {
162  auto num_contrib_state = num_contrib == -1 ? 1 : num_contrib;
163  ReduceState<MessageT> state(
164  msg->reduce_tag_,msg->reduce_seq_,num_contrib_state
165  );
166  ReduceStateHolder<MessageT>::insert(group_,lookup,std::move(state));
167  }
168 
169  auto& state = ReduceStateHolder<MessageT>::find(group_,lookup);
170  auto msg_ptr = promoteMsg(msg);
171  state.msgs.push_back(msg_ptr);
172  if (num_contrib != -1) {
173  state.num_contrib_ = num_contrib;
174  }
175  if (local) {
176  state.num_local_contrib_++;
177  }
178  state.combine_handler_ = msg->combine_handler_;
179  state.reduce_root_ = msg->reduce_root_;
180  debug_print(
181  reduce, node,
182  "reduceAddMsg: group={:x}, msg={}, contrib={}, msgs.size()={}, ref={}\n",
183  group_, print_ptr(msg), state.num_contrib_,
184  state.msgs.size(), envelopeGetRef(msg->env)
185  );
186 }

188 template <typename MessageT>
189 void Reduce::startReduce(
190  TagType tag, SequentialIDType seq, VirtualProxyType proxy,
191  ObjGroupProxyType objgroup, bool use_num_contrib
192 ) {
193  auto lookup = ReduceIdentifierType{tag,seq,proxy,objgroup};
194  auto& state = ReduceStateHolder<MessageT>::find(group_,lookup);
195 
196  std::size_t nmsgs = state.msgs.size();
197  auto const contrib =
198  use_num_contrib ? state.num_contrib_ : state.num_local_contrib_;
199  std::size_t total = getNumChildren() + contrib;
200  bool ready = nmsgs == total;
201 
202  debug_print(
203  reduce, node,
204  "startReduce: group={:x}, tag={}, seq={}, vrt={}, msg={}, children={}, "
205  "contrib_={}, local_contrib_={}, nmsgs={}, ready={}\n",
206  group_, tag, seq, proxy, state.msgs.size(), getNumChildren(),
207  state.num_contrib_, state.num_local_contrib_, nmsgs, ready
208  );
209 
210  if (ready) {
211  // Combine messages
212  if (state.msgs.size() > 1) {
213  auto size = state.msgs.size();
214  for (decltype(size) i = 0; i < size; i++) {
215  bool const has_next = i+1 < size;
216  state.msgs[i]->next_ = has_next ? state.msgs[i+1].get() : nullptr;
217  state.msgs[i]->count_ = size - i;
218  state.msgs[i]->is_root_ = false;
219 
220  debug_print(
221  reduce, node,
222  "i={} next={} has_next={} count={} msgs.size()={}, ref={}\n",
223  i, print_ptr(state.msgs[i]->next_), has_next, state.msgs[i]->count_,
224  size, envelopeGetRef(state.msgs[i]->env)
225  );
226  }
227 
228  debug_print(
229  reduce, node,
230  "msgs.size()={}\n", size
231  );
232 
233  /*
234  * Invoke user handler to run the functor that combines messages,
235  * applying the reduction operator
236  */
237  auto const& handler = state.combine_handler_;
238  auto const& from_node = theMsg()->getFromNodeCurrentHandler();
239  runnable::Runnable<MessageT>::run(
240  handler,nullptr,static_cast<MessageT*>(state.msgs[0].get()),from_node
241  );
242  }
243 
244  // Send to parent
245  auto msg = state.msgs[0];
246  auto typed_msg = static_cast<MessageT*>(msg.get());
247  ActionType cont = nullptr;
248 
249  state.msgs.clear();
250  state.num_contrib_ = 1;
251 
252  if (isRoot()) {
253  auto const& root = state.reduce_root_;
254  auto const& this_node = theContext()->getNode();
255  if (root != this_node) {
256  debug_print(
257  reduce, node,
258  "reduce notify root (send): root={}, node={}\n", root, this_node
259  );
260 
261  theMsg()->sendMsg<MessageT,reduceRootRecv<MessageT>>(root,typed_msg);
262  } else {
263  debug_print(
264  reduce, node,
265  "reduce notify root (deliver directly): root={}, node={}\n",
266  root, this_node
267  );
268  reduceRootRecv(typed_msg);
269  }
270  } else {
271  auto const& parent = getParent();
272  debug_print(
273  reduce, node,
274  "reduce send to parent: parent={}\n", parent
275  );
276  theMsg()->sendMsg<MessageT,reduceUp<MessageT>>(parent,typed_msg);
277  }
278  }
279 }
