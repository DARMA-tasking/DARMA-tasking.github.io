99 Reduce::PendingSendType Reduce::reduce(Node root, Params&&... params) {
100  using Tuple = typename FuncTraits<decltype(f)>::TupleType;
101  using MsgT = ReduceTMsg<Tuple>;

159 template <typename MsgT, ActiveTypedFnType<MsgT>* f>
160 detail::ReduceStamp Reduce::reduceImmediate(
161  NodeType root, MsgT* const msg, detail::ReduceStamp id,
162  ReduceNumType num_contrib
163 ) {
164  if (std::holds_alternative<detail::StrongGroup>(scope_.get())) {
165  envelopeSetGroup(msg->env, std::get<detail::StrongGroup>(scope_.get()).get());
166  }
167 
168  auto default_stamp = detail::ReduceStamp{};
169  auto cur_id = id == default_stamp ? generateNextID() : id;
170 
171  auto const han = auto_registry::makeAutoHandler<MsgT,f>();
172  msg->combine_handler_ = han;
173  msg->reduce_root_ = root;
174  msg->reduce_id_ = detail::ReduceIDImpl{cur_id, scope_};
175 
176  vt_debug_print(
177  terse, reduce,
178  "reduce: scope={}, stamp={}, contrib={}, msg={}, ref={}\n",
179  scope_.str(), detail::stringizeStamp(id), num_contrib, print_ptr(msg),
180  envelopeGetRef(msg->env)
181  );
182 
183  reduceAddMsg<MsgT>(msg,true,num_contrib);
184  reduceNewMsg<MsgT>(msg);
185 
186  return cur_id;

189 template <typename MsgT>
190 void Reduce::reduceAddMsg(
191  MsgT* msg, bool const local, ReduceNumType num_contrib
192 ) {
193  auto lookup = msg->stamp();
194 
195  auto exists = state_.exists(lookup);
196 
197  if (not exists) {
198  auto num_contrib_state = num_contrib == -1 ? 1 : num_contrib;
199  ReduceState state(num_contrib_state);
200  state_.insert(lookup,std::move(state));
201  }
202 
203  auto& state = state_.find(lookup);
204  auto msg_ptr = promoteMsg(msg);
205 
206  // Run-time cast with lasting type info to ReduceMsg for holder
207  state.msgs.push_back(msg_ptr.template to<ReduceMsg>());
208  if (num_contrib != -1) {
209  state.num_contrib_ = num_contrib;
210  }
211  if (local) {
212  state.num_local_contrib_++;
213  }
214  state.combine_handler_ = msg->combine_handler_;
215  state.reduce_root_ = msg->reduce_root_;
216  vt_debug_print(
217  verbose, reduce,
218  "reduceAddMsg: scope={}, stamp={}, msg={}, contrib={}, msgs.size()={}, "
219  "ref={}\n",
220  scope_.str(), detail::stringizeStamp(lookup), print_ptr(msg),
221  state.num_contrib_, state.msgs.size(), envelopeGetRef(msg->env)
222  );
223 }

225 template <typename MsgT>
226 void Reduce::startReduce(detail::ReduceStamp id, bool use_num_contrib) {
227  auto lookup = id;
228  auto& state = state_.find(lookup);
229 
230  std::size_t nmsgs = state.msgs.size();
231  auto const contrib =
232  use_num_contrib ? state.num_contrib_ : state.num_local_contrib_;
233  std::size_t total = getNumChildren() + contrib;
234  bool ready = nmsgs == total;
235 
236  vt_debug_print(
237  normal, reduce,
238  "startReduce: scope={}, stamp={}, msg={}, children={}, "
239  "contrib_={}, local_contrib_={}, nmsgs={}, ready={}\n",
240  scope_.str(), detail::stringizeStamp(id), state.msgs.size(),
241  getNumChildren(), state.num_contrib_, state.num_local_contrib_, nmsgs, ready
242  );
243 
244  if (ready) {
245  // Combine messages
246  if (state.msgs.size() > 1) {
247  auto size = state.msgs.size();
248  for (decltype(size) i = 0; i < size; i++) {
249  bool const has_next = i+1 < size;
250  // Collection is of MsgPtr<ReduceMsg>
251  auto typed_msg = static_cast<MsgT*>(state.msgs[i].get());
252  if (has_next) {
253  typed_msg->next_ = static_cast<MsgT*>(state.msgs[i+1].get());
254  } else {
255  typed_msg->next_ = nullptr;
256  }
257  typed_msg->count_ = size - i;
258  typed_msg->is_root_ = false;
259 
260  vt_debug_print(
261  verbose, reduce,
262  "scope={}, stamp={}: i={} next={} has_next={} count={} msgs.size()={} "
263  "ref={}\n",
264  scope_.str(), detail::stringizeStamp(id),
265  i, print_ptr(typed_msg->next_), has_next, typed_msg->count_,
266  size, envelopeGetRef(typed_msg->env)
267  );
268  }
269 
270  vt_debug_print(
271  verbose, reduce,
272  "scope={}, stamp={}, msgs.size()={}\n",
273  scope_.str(), detail::stringizeStamp(id), size
274  );
275 
276  /*
277  * Invoke user handler to run the functor that combines messages,
278  * applying the reduction operator
279  */
280  auto const handler = state.combine_handler_;
281  auto const from_node = theContext()->getFromNodeCurrentTask();
282 
283  // this needs to run inline.. threaded not allowed for reduction
284  // combination
285  runnable::makeRunnable(state.msgs[0], false, handler, from_node)
286  .withTDEpochFromMsg()
287  .run();
288  }
289 
290  // Send to parent
291  // Collection is of MsgPtr<ReduceMsg>, re-type and drop collection owner.
292  auto msg = state.msgs[0];
293  MsgPtr<MsgT> typed_msg = msg.template to<MsgT>();
294  state.msgs.clear();
295  state.num_contrib_ = 1;
296  NodeType const root = state.reduce_root_;
297 
298  // Must erase this before invoking the root function/callback because it
299  // might be re-entrant
300  state_.erase(lookup);
301 
302  if (isRoot()) {
303  auto const& this_node = theContext()->getNode();
304  if (root != this_node) {
305  vt_debug_print(
306  normal, reduce,
307  "reduce notify root (send): scope={}, stamp={}, root={}, node={}\n",
308  scope_.str(), detail::stringizeStamp(id), root, this_node
309  );
310 
311  theMsg()->sendMsg<ReduceManager::reduceRootRecv<MsgT>>(root, typed_msg);
312  } else {
313  vt_debug_print(
314  normal, reduce,
315  "reduce notify root (direct): scope={}, stamp={}, root={}, node={}\n",
316  scope_.str(), detail::stringizeStamp(id), root, this_node
317  );
318  theMsg()->setupEpochMsg(typed_msg.get());
319  reduceRootRecv(typed_msg.get());
320  }
321  } else {
322  auto const& parent = getParent();
323  vt_debug_print(
324  normal, reduce,
325  "reduce send to parent: scope={}, stamp={}, parent={}\n",
326  scope_.str(), detail::stringizeStamp(id), parent
327  );
328 
329  theMsg()->sendMsg<ReduceManager::reduceUpHan<MsgT>>(parent, typed_msg);
330  }
331  }
332 }
