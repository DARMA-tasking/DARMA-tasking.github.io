117 template <typename IndexType, NumDimensionsType ndim>
118 void DenseIndexArray<IndexType, ndim>::foreach(
119  ThisIndexType in_max, ApplyType fn
120 ) const {
121  ThisIndexType max = in_max;
122  auto size = max.getSize();
123  std::array<IndexType, ndim> vec = {{0}};
124  for (decltype(size) sz = 0; sz < size; sz++) {
125  fn(ThisIndexType(vec));
126  for (auto i = 0; i < ndim; i++) {
127  if (vec[i] + 1 < max[i]) {
128  vec[i]++;
129  for (auto j = 0; j < i; j++) {
130  vec[j] = 0;
131  }
132  break;
133  }
134  }
135  }
136 }
