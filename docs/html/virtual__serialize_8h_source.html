101 struct SerializeVirtualTypeIfNeeded<
102  T,
103  SerializerT,
104  typename std::enable_if_t<
105  dispatch::vrt::VirtualSerializeTraits<T>::has_virtual_serialize
106  >
107 >
108 {
109  static typename dispatch::vrt::TypeIdx apply(SerializerT& s, T* target) {
110  dispatch::vrt::TypeIdx entry = dispatch::vrt::no_type_idx;
111 
112  if (not s.isUnpacking()) {
113  entry = target->_checkpointDynamicTypeIndex();
114  }
115 
116  // entry doesn't count as part of a footprint
117  if (not s.isFootprinting()) {
118  s | entry;
119  }
120 
121  if (target != nullptr) {
122  // Support deserialization in place, and make sure it's safe
123  checkpointAssert(entry == target->_checkpointDynamicTypeIndex(),
124  "Trying to deserialize in place over a mismatched type");
125  }
126 
127  return entry;
128  }
