53 template <typename T>
54 struct MutexTraits {
55  template <typename U, typename... Vs>
56  using constructor_t = decltype(U(std::declval<Vs>()...));
57  using has_constructor = detection::is_detected<constructor_t, T>;
58 
59  template <typename U>
60  using copy_constructor_t = decltype(U(std::declval<U const&>()));
61  using has_copy_constructor = detection::is_detected<copy_constructor_t, T>;
62 
63  template <typename U>
64  using lock_t = decltype(std::declval<U>().lock());
65  using has_lock = detection::is_detected<lock_t, T>;
66 
67  template <typename U>
68  using unlock_t = decltype(std::declval<U>().unlock());
69  using has_unlock = detection::is_detected<unlock_t, T>;
70 
71  template <typename U>
72  using try_lock_t = decltype(std::declval<U>().try_lock());
73  using has_try_lock = detection::is_detected_exact<bool, try_lock_t, T>;
74 
75  // This defines what it means to be an `mutex'
76  static constexpr auto const is_mutex =
77  // default constructor and copy constructor
78  has_constructor::value and not has_copy_constructor::value and
79  // methods: lock(), unlock(), try_lock()
80  has_lock::value and has_unlock::value and
81  has_try_lock::value;
82 };
