90 typename Proxy<ObjT>::PendingSendType Proxy<ObjT>::reduce(
91  MsgPtrT inmsg, Callback<MsgT> cb, ReduceStamp stamp
92 ) const {
93  auto proxy = Proxy<ObjT>(*this);
94  // Force promoteMsg(Msg*) to avoid special-case overload behavior
95  // which is now deprecated. The issue is caused because inmsg can be either
96  // Msg* or MsgPtr<Msg> based on call sites.
97  MsgPtr<MsgT> msg = promoteMsg(static_cast<MsgT*>(inmsg));
98  msg->setCallback(cb);
99  return theObjGroup()->reduce<ObjT, MsgT, f>(proxy,msg,stamp);
100 }
