90 template <typename Readable>
91 std::size_t Decompressor<Readable>::read(
92  uint8_t* output_buffer, std::size_t bytes_to_output
93 ) {
94  uint8_t* next_out = output_buffer;
95  std::size_t avail_out = bytes_to_output;
96 
97  if (avail_in_ == 0) {
98  getMoreInput();
99  }
100 
101  bool success = false;
102  // While we have input data and room to output decompressed data...
103  while (avail_out > 0 and not success) {
104  BrotliDecoderResult res = BrotliDecoderDecompressStream(
105  dec_, &avail_in_, &next_in_, &avail_out, &next_out, nullptr
106  );
107  switch (res) {
108  case BROTLI_DECODER_RESULT_NEEDS_MORE_INPUT:
109  {
110  // We need to read more to continue decoding
111  vtAssert(avail_in_ == 0, "Brotli asked for more input even though we still had some.");
112  bool const has_more_input = getMoreInput();
113  if (not has_more_input) {
114  vtAbort("Brotli asked for more input but the file terminated early.");
115  }
116  break;
117  }
118  case BROTLI_DECODER_RESULT_NEEDS_MORE_OUTPUT:
119  {
120  // this is an expected condition
121  vtAssert(
122  avail_out == 0, "Brotli asked for more output even though we still had some."
123  );
124  break;
125  }
126  case BROTLI_DECODER_RESULT_SUCCESS:
127  {
128  if (avail_in_ != 0 or getMoreInput()) {
129  vtAbort("Brotli terminated early before reading the whole file!");
130  }
131  success = true;
132  break;
133  }
134  case BROTLI_DECODER_RESULT_ERROR:
135  {
136  // we have hit an unknown error, print the code and corresponding message!
137  auto error_code = BrotliDecoderGetErrorCode(dec_);
138  auto error_str = fmt::format(
139  "code={}, msg={}\n", static_cast<int32_t>(error_code), BrotliDecoderErrorString(error_code)
140  );
141  vtAbort(error_str);
142  break;
143  }
144  default:
145  vtAbort("Unknown error condition during decompression");
146  }
147  }
148 
149  // return how many bytes we ended up decoding
150  return bytes_to_output - avail_out;
151 }

153 template <typename Readable>
154 bool Decompressor<Readable>::getMoreInput() {
155  if (avail_in_ == 0) {
156  // read some data, up to our internal temporary buffer length
157  r_.read(reinterpret_cast<char*>(buf_in_.get()), in_buf_len_);
158 
159  // check how much we actually read in
160  avail_in_ = r_.gcount();
161  next_in_ = buf_in_.get();
162 
163  return avail_in_ != 0;
164  } else {
165  return false;
166  }
167 }
