109 template <typename LocType>
110 /*static*/ void LocationManager::applyInstance(
111  LocInstType const inst, ActionLocInstType<LocType> action
112 ) {
113  auto inst_iter = loc_insts.find(inst);
114  if (inst_iter == loc_insts.end()) {
115  auto pending_iter = details::PendingInst<LocType>::pending_inst_.find(inst);
116  if (pending_iter == details::PendingInst<LocType>::pending_inst_.end()) {
117  details::PendingInst<LocType>::pending_inst_.emplace(
118  std::piecewise_construct,
119  std::forward_as_tuple(inst),
120  std::forward_as_tuple(PendingContainerType<LocType>{action})
121  );
122  } else {
123  pending_iter->second.push_back(action);
124  }
125  } else {
126  auto const& inst_ret = LocationManager::getInstance(inst);
127  action(static_cast<LocType*>(inst_ret));
128  }
129 }

140  if (iter != details::PendingInst<LocType>::pending_inst_.end()) {

143  }
144  details::PendingInst<LocType>::pending_inst_.erase(iter);
