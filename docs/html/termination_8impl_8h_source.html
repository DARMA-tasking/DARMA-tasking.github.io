105 inline void TerminationDetector::produceConsume(
106  EpochType epoch, TermCounterType num_units, bool produce, NodeType node
107 ) {
108  vt_debug_print(
109  normal, term,
110  "produceConsume: epoch={:x}, rooted={}, ds={}, count={}, produce={}, "
111  "node={}\n",
112  epoch, isRooted(epoch), isDS(epoch), num_units, produce, node
113  );
114 
115  produceConsumeState(any_epoch_state_, num_units, produce);
116 
117  if (epoch != any_epoch_sentinel) {
118  if (isDS(epoch)) {
119  auto ds_term = getDSTerm(epoch);
120 
121  // If a node is not passed, use the current node (self-prod/cons)
122  if (node == uninitialized_destination) {
123  node = this_node_;
124  }
125 
126  if (produce) {
127  ds_term->msgSent(node,num_units);
128  } else {
129  ds_term->msgProcessed(node,num_units);
130  }
131  } else {
132  auto& state = findOrCreateState(epoch, false);
133  produceConsumeState(state, num_units, produce);
134  }

145  inline void TerminationDetector::pushEpoch(EpochType epoch) {
146  /*
147  * pushEpoch(epoch) pushes any epoch onto the local stack iff epoch !=
148  * no_epoch; the epoch stack includes all locally pushed epochs and the
149  * current contexts pushed, transitively causally related active message
150  * handlers.
151  */
152  vtAssertInfo(
153  epoch != no_epoch, "Do not push no_epoch onto the epoch stack",
154  epoch, no_epoch, epoch_stack_.size(),
155  epoch_stack_.size() > 0 ? EpochType{epoch_stack_.top()} : no_epoch
156  );
157  if (epoch != no_epoch) {
158  epoch_stack_.push(epoch.get());
159  }
160 }

162 inline EpochType TerminationDetector::popEpoch(EpochType epoch) {
163  /*
164  * popEpoch(epoch) shall remove the top entry from epoch_size_, iif the size
165  * is non-zero and the `epoch' passed, if `epoch != no_epoch', is equal to the
166  * top of the `epoch_stack_.top()'; else, it shall remove any entry from the
167  * top of the stack.
168  */
169  auto const& non_zero = epoch_stack_.size() > 0;
170  vtAssertExprInfo(
171  non_zero and (epoch_stack_.top() == epoch.get() or epoch == no_epoch),
172  epoch, non_zero, non_zero ? EpochType{epoch_stack_.top()} : no_epoch
173  );
174  if (epoch == no_epoch) {
175  return non_zero ? epoch_stack_.pop(),EpochType{epoch_stack_.top()} : no_epoch;
176  } else {
177  return non_zero && epoch == EpochType{epoch_stack_.top()} ?
178  epoch_stack_.pop(),epoch :
179  no_epoch;
180  }
181 }
