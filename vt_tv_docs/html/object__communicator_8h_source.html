169  double getMaxVolume() const {
170  // Search for the maximum value in received and sent (0. if sets are empty)
171  double max_recv = !this->received_.empty() ?
172  std::max_element(
173  this->received_.begin(),
174  this->received_.end(),
175  [](const auto& a, const auto& b) { return a.second < b.second; })
176  ->second :
177  0.0;
178 
179  double max_sent = !this->sent_.empty() ?
180  std::max_element(
181  this->sent_.begin(),
182  this->sent_.end(),
183  [](const auto& a, const auto& b) { return a.second < b.second; })
184  ->second :
185  0.0;
186 
187  // Return the max
188  return std::max(max_recv, max_sent);
189  }
