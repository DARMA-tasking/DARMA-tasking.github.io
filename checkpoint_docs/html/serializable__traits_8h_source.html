89 template <typename T, typename S = checkpoint::BaseSerializer>
90 struct SerializableTraits {
100  // Regular serialize detection
101  template <typename U, typename V>
102  using serialize_t = decltype(
103  std::declval<U>().serialize(std::declval<V&>())
104  );
105  using has_serialize = detection::is_detected<serialize_t, T, S>;
106 
107  template <typename U, typename V>
108  using nonintrustive_serialize_t = decltype(serialize(
109  std::declval<V&>(),
110  std::declval<U&>()
111  ));
112  using has_nonintrustive_serialize =
113  detection::is_detected<nonintrustive_serialize_t, T, S>;
114 
115  // serialize{Parent,This} detection
116  template <typename U, typename V>
117  using serializeParent_t =
118  decltype(std::declval<U>().serializeParent(std::declval<V&>()));
119  using has_serializeParent = detection::is_detected<serializeParent_t, T, S>;
120 
121  template <typename U, typename V>
122  using serializeThis_t =
123  decltype(std::declval<U>().serializeThis(std::declval<V&>()));
124  using has_serializeThis = detection::is_detected<serializeThis_t, T, S>;
125 
126  // This defines what it means to have an intrusive serialize
127  static constexpr auto const has_serialize_instrusive = has_serialize::value;
128 
129  // This defines what it means to have a non-intrusive serialize
130  static constexpr auto const has_serialize_noninstrusive =
131  has_nonintrustive_serialize::value;
132 
133  // This defines what it means to have some type of serialize function
134  static constexpr auto const has_serialize_function =
135  has_serialize_instrusive or has_serialize_noninstrusive;
136 
137  // This defines what it means to have parent serializability
138  static constexpr auto const has_parent_serialize = has_serializeParent::value;
139 
140  // This defines what it means to have "this" serializability
141  static constexpr auto const has_this_serialize = has_serializeThis::value;
142 
143  // This defines what it means to have both parent and "this" serializability
144  static constexpr auto const has_split_serialize =
145  has_parent_serialize and has_this_serialize;
146 
151  using has_byteCopyTraitTrue = isByteCopyable<T>;
152 
153  template <typename U>
154  using has_isArith = std::is_arithmetic<U>;
155 
156  // This defines what it means to be serializable without a serialize method
157  static constexpr auto const is_bytecopyable =
158  has_byteCopyTraitTrue::value or has_isArith<T>::value or
159  ByteCopyNonIntrusive<T>::isByteCopyable::value;
160 
165  template <typename U>
166  using constructor_t = decltype(U());
167  using has_default_constructor = detection::is_detected<constructor_t, T>;
168 
169  template <typename U>
170  using tagged_constructor_t = decltype(U(std::declval<SERIALIZE_CONSTRUCT_TAG>()));
171  using has_tagged_constructor = detection::is_detected<tagged_constructor_t, T>;
172 
173  template <typename U>
174  using reconstruct_t = decltype(U::reconstruct(std::declval<void*>()));
175  using has_reconstruct = detection::is_detected_exact<T&, reconstruct_t, T>;
176 
177  template <typename U>
178  using nonintrustive_reconstruct_t = decltype(
179  reconstruct(
180  std::declval<U*&>(),
181  std::declval<void*>()
182  )
183  );
184  using has_nonintrusive_reconstruct =
185  detection::is_detected<nonintrustive_reconstruct_t, T>;
186 
187  template <typename U>
188  using specialized_reconstruct_t = decltype(
189  checkpoint::CheckpointReconstructor<U>::reconstruct(
190  std::declval<U*&>(),
191  std::declval<void*>()
192  )
193  );
194  using has_specialized_reconstruct =
195  detection::is_detected<specialized_reconstruct_t, T>;
196 
197  // This defines what it means to be default constructible
198  static constexpr auto const is_default_constructible =
199  has_default_constructor::value;
200 
201  // If the class has tagged constructor especially for constructing
202  static constexpr auto const is_tagged_constructible =
203  has_tagged_constructor::value;
204 
205  // This defines what it means to be intrusively reconstructible
206  static constexpr auto const is_intrusive_reconstructible = has_reconstruct::value;
207 
208  // This defines what it means to be non-intrusively reconstructible
209  static constexpr auto const is_nonintrusive_reconstructible =
210  has_nonintrusive_reconstruct::value /*and not has_reconstruct::value*/;
211 
212  // This defines what it means to be specialized reconstructible
213  static constexpr auto const is_specialized_reconstructible =
214  has_specialized_reconstruct::value;
215 
216  // This defines what it means to be reconstructible in general
217  static constexpr auto const is_reconstructible =
218  has_nonintrusive_reconstruct::value or
219  has_reconstruct::value or
220  has_specialized_reconstruct::value;
221 
222  // This defines what is means to be generally constructible for
223  // de-serialization
224  static constexpr auto const is_constructible =
225  is_default_constructible or is_reconstructible or is_tagged_constructible;
226 
230  static constexpr auto const is_serializable =
231  has_serialize_function and is_constructible;
232 
233  static constexpr auto const is_traversable =
234  has_serialize_function or is_bytecopyable;
235 };
