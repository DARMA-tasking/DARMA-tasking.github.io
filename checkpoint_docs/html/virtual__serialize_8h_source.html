101 template <typename T, typename SerializerT>
102 struct SerializeVirtualTypeIfNeeded<
103  T,
104  SerializerT,
105  typename std::enable_if_t<
106  dispatch::vrt::VirtualSerializeTraits<T>::has_virtual_serialize
107  >
108 >
109 {
110  static typename dispatch::vrt::TypeIdx apply(SerializerT& s, T* target) {
111  dispatch::vrt::TypeIdx entry = dispatch::vrt::no_type_idx;
112 
113  if (not s.isUnpacking()) {
114  entry = target->_checkpointDynamicTypeIndex();
115  }
116 
117  // entry doesn't count as part of a footprint
118  if (not s.isFootprinting()) {
119  s | entry;
120  }
121 
122  if (target != nullptr) {
123  // Support deserialization in place, and make sure it's safe
124  checkpointAssert(entry == target->_checkpointDynamicTypeIndex(),
125  "Trying to deserialize in place over a mismatched type");
126  }
127 
128  return entry;
129  }
130 };
