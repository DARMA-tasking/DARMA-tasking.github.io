73 template <typename T, typename TraverserT>
74 TraverserT& withTypeIdx(TraverserT& t) {
75  using CleanT = typename CleanType<typeregistry::DecodedIndex>::CleanT;
76  using DispatchType =
77  typename TraverserT::template DispatcherType<TraverserT, CleanT>;
78 
79  auto const thisTypeIdx = typeregistry::getTypeIdx<T>();
80  SerializerDispatch<TraverserT, CleanT, DispatchType> ap;
81  constexpr SerialSizeType len = 1;
82  if (t.isPacking() || t.isSizing()) {
83  auto val = cleanType(&thisTypeIdx);
84  ap(t, val, len);
85  } else if (t.isUnpacking()) {
86  typeregistry::DecodedIndex serTypeIdx = 0;
87  auto val = cleanType(&serTypeIdx);
88  ap(t, val, len);
89 
90  if (
91  typeregistry::validateIndex(serTypeIdx) == false ||
92  thisTypeIdx != serTypeIdx
93  ) {
94  auto const err = std::string("Unpacking wrong type, got=") +
95  typeregistry::getTypeNameForIdx(thisTypeIdx) +
96  " (idx=" + std::to_string(thisTypeIdx) +
97  "), expected=" + typeregistry::getTypeNameForIdx(serTypeIdx) +
98  " (idx=" + std::to_string(serTypeIdx) + ")\n#0 " +
99  typeregistry::getTypeName<T>();
100  throw serialization_error(err);
101  }
102  }
103 
104  return t;
105 }

107 template <typename T, typename TraverserT>
108 TraverserT& withMemUsed(TraverserT& t, SerialSizeType len) {
109  using DispatchType =
110  typename TraverserT::template DispatcherType<TraverserT, SerialSizeType>;
111  SerializerDispatch<TraverserT, SerialSizeType, DispatchType> ap;
112 
113  SerialSizeType const memUsed = sizeof(T) * len;
114  constexpr SerialSizeType memUsedLen = 1;
115  if (t.isPacking() || t.isSizing()) {
116  auto val = cleanType(&memUsed);
117  ap(t, val, memUsedLen);
118  } else if (t.isUnpacking()) {
119  SerialSizeType serMemUsed = 0;
120  auto val = cleanType(&serMemUsed);
121  ap(t, val, memUsedLen);
122 
123  if (memUsed != serMemUsed) {
124  using CleanT = typename CleanType<T>::CleanT;
125  std::string msg = "For type '" + typeregistry::getTypeName<CleanT>() +
126  "' serialization used " + std::to_string(serMemUsed) +
127  "B, but deserialization used " + std::to_string(memUsed) + "B";
128 
129  throw serialization_error(msg);
130  }
131  }
132 
133  return t;
134 }

136 template <typename T, typename TraverserT>
137 TraverserT& Traverse::with(T& target, TraverserT& t, SerialSizeType len) {
138  using CleanT = typename CleanType<T>::CleanT;
139  using DispatchType =
140  typename TraverserT::template DispatcherType<TraverserT, CleanT>;
141 
142  #if defined(SERIALIZATION_ERROR_CHECKING)
143  withTypeIdx<CleanT>(t);
144  #endif
145 
146  auto val = cleanType(&target);
147  SerializerDispatch<TraverserT, CleanT, DispatchType> ap;
148 
149  #if defined(SERIALIZATION_ERROR_CHECKING)
150  try {
151  ap(t, val, len);
152  } catch (serialization_error const& err) {
153  auto const depth = err.depth_ + 1;
154  auto const what = std::string(err.what()) + "\n#" + std::to_string(depth) +
155  " " + typeregistry::getTypeName<T>();
156  throw serialization_error(what, depth);
157  }
158  #else
159  ap(t, val, len);
160  #endif
161 
162  #if defined(SERIALIZATION_ERROR_CHECKING)
163  withMemUsed<CleanT>(t, len);
164  #endif
165 
166  return t;
167 }

250 template <typename T>
251 buffer::ImplReturnType
252 packBuffer(T& target, SerialSizeType size, BufferObtainFnType fn) {
253  SerialByteType* user_buf = fn ? fn(size) : nullptr;
254  if (user_buf == nullptr) {
255  auto p =
256  Standard::pack<T, PackerBuffer<buffer::ManagedBuffer>>(target, size);
257  validatePackerBufferSize<T>(p, size);
258  return std::make_tuple(std::move(p.extractPackedBuffer()), size);
259  } else {
260  auto p = Standard::pack<T, PackerBuffer<buffer::UserBuffer>>(
261  target, size, std::make_unique<buffer::UserBuffer>(user_buf, size)
262  );
263  validatePackerBufferSize<T>(p, size);
264  return std::make_tuple(std::move(p.extractPackedBuffer()), size);
265  }
266 }
