75  >::type;

80  SerializableTraits<U,void>::is_nonintrusive_reconstructible, T

85  typename std::enable_if<not SerializableTraits<U,void>::is_reconstructible, T>::type;

110  // Default-construct as lowest priority in reconstruction preference
111  template <typename U = T>
112  static T* constructDefault(void* buf, isDefaultConsType<U>* = nullptr) {
113  debug_checkpoint("DeserializerDispatch: default constructor: buf=%p\n", buf);
114  T* t_ptr = new (buf) T{};
115  return t_ptr;
116  }
117 
118  // Fail, no valid option to constructing T
119  template <typename U = T>
120  static T* constructDefault(void* buf, isNotDefaultConsType<U>* = nullptr) {
121  #if HAS_DETECTION_COMPONENT
122  static_assert(
123  SerializableTraits<U,void>::is_tagged_constructible or
124  SerializableTraits<U,void>::is_reconstructible or
125  std::is_default_constructible<U>::value,
126  "Either a default constructor, reconstruct() function, or tagged "
127  "constructor are required for de-serialization"
128  );
129  #else
130  static_assert(
131  std::is_default_constructible<U>::value,
132  "A default constructor is required for de-serialization. To enable "
133  "reconstruct or tagged constructors, you must compile with the "
134  "detection component."
135  );
136  #endif
137  return nullptr;
138  }

142  /*
143  * Try to reconstruct with the following precedence:
144  *
145  * - Tagged constructor: T(SERIALIZE_CONSTRUCT_TAG{})
146  * - Reconstruct: T::reconstruct(buf) or reconstruct(t, buf)
147  * - Default constructor: T()
148  *
149  * If none of these options work, static assert failure
150  */

152  // Intrusive reconstruct
153  template <typename U = T>
154  static T* constructReconstruct(void* buf, isReconstructibleType<U>* = nullptr) {
155  debug_checkpoint("DeserializerDispatch: T::reconstruct(): buf=%p\n", buf);
156  auto& t = T::reconstruct(buf);
157  return &t;
158  }
159 
160  // Non-intrusive reconstruct
161  template <typename U = T>
162  static T* constructReconstruct(void* buf, isNonIntReconstructibleType<U>* = nullptr) {
163  debug_checkpoint("DeserializerDispatch: non-int reconstruct(): buf=%p\n", buf);
164  T* t = nullptr;
165  // Explicitly call bare to invoke ADL
166  reconstruct(t,buf);
167  return t;
168  }

171  template <typename U = T>
172  static T* constructReconstruct(void* buf, isNotReconstructibleType<U>* = nullptr) {
173  return constructDefault<U>(buf);
174  }

178  static T* constructTag(void* buf, isTaggedConstructibleType<U>* = nullptr) {
179  debug_checkpoint("DeserializerDispatch: tagged constructor: buf=%p\n", buf);

214  sprintf(
215  &msg[0],
216  "Checkpoint is failing to reconstruct a class %s, due to it being "

223  }
